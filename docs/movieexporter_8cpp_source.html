73 int16_t safeSumInt16(int16_t a, int16_t b)
74 {
75  int32_t a32 = static_cast<int32_t>(a);
76  int32_t b32 = static_cast<int32_t>(b);
77 
78  if ((a32 + b32) > INT16_MAX)
79  {
80  return INT16_MAX;
81  }
82  else if ((a32 + b32) < INT16_MIN)
83  {
84  return INT16_MIN;
85  }
86  return a + b;
87 }

244 Status MovieExporter::assembleAudio(const Object* obj,
245  QString ffmpegPath,
246  std::function<void(float)> progress)
247 {
248  // Quicktime assemble call
249  const int startFrame = mDesc.startFrame;
250  const int endFrame = mDesc.endFrame;
251  const int fps = mDesc.fps;
252 
253  Q_ASSERT(startFrame >= 0);
254  Q_ASSERT(endFrame >= startFrame);
255 
256  float lengthInSec = (endFrame - startFrame + 1) / (float)fps;
257  qDebug() << "Audio Length = " << lengthInSec << " seconds";
258 
259  int32_t audioDataSize = 44100 * 2 * 2 * lengthInSec;
260 
261  std::vector<int16_t> audioData(audioDataSize / sizeof(int16_t));
262 
263  bool audioDataValid = false;
264 
265  QDir dir(mTempWorkDir);
266  Q_ASSERT(dir.exists());
267 
268  QString tempAudioPath = mTempWorkDir + "/tmpaudio0.wav";
269  qDebug() << "TempAudio=" << tempAudioPath;
270 
271  std::vector< SoundClip* > allSoundClips;
272 
273  std::vector< LayerSound* > allSoundLayers = obj->getLayersByType<LayerSound>();
274  for (LayerSound* layer : allSoundLayers)
275  {
276  layer->foreachKeyFrame([&allSoundClips](KeyFrame* key)
277  {
278  allSoundClips.push_back(static_cast<SoundClip*>(key));
279  });
280  }
281 
282  int clipCount = 0;
283 
284  for (SoundClip* clip : allSoundClips)
285  {
286  if (mCanceled)
287  {
288  return Status::CANCELED;
289  }
290 
291  // convert audio file: 44100Hz sampling rate, stereo, signed 16 bit little endian
292  // supported audio file types: wav, mp3, ogg... ( all file types supported by ffmpeg )
293  QString strCmd;
294  strCmd += QString("\"%1\"").arg(ffmpegPath);
295  strCmd += QString(" -i \"%1\" ").arg(clip->fileName());
296  strCmd += "-ar 44100 -acodec pcm_s16le -ac 2 -y ";
297  strCmd += QString("\"%1\"").arg(tempAudioPath);
298 
299  executeFFMpeg(strCmd, [](float){});
300  qDebug() << "audio file: " + tempAudioPath;
301 
302  // Read wav file header
303  WavFileHeader header;
304  QFile file(tempAudioPath);
305  file.open(QIODevice::ReadOnly);
306  file.read((char*)&header, sizeof(WavFileHeader));
307 
308  skipUselessChucks(header, file);
309 
310  int32_t audioSize = header.dataSize;
311 
312  qDebug() << "audio len " << audioSize;
313 
314  // before calling malloc should check: audioSize < max credible value
315  std::vector< int16_t > data(audioSize / sizeof(int16_t));
316  file.read((char*)data.data(), audioSize);
317  audioDataValid = true;
318 
319  float fframe = (float)clip->pos() / (float)fps;
320  int delta = fframe * 44100 * 2;
321  qDebug() << "audio delta " << delta;
322 
323  int indexMax = std::min(audioSize / 2, audioDataSize / 2 - delta);
324 
325  // audio files 'mixing': 'higher' sound layers overwrite 'lower' sound layers
326  for (int i = 0; i < indexMax; i++)
327  {
328  audioData[i + delta] = safeSumInt16(audioData[i + delta], data[i]);
329  }
330 
331  file.close();
332 
333  progress((float)clipCount / allSoundClips.size());
334  clipCount++;
335  }
336 
337  if (!audioDataValid)
338  {
339  return Status::SAFE;
340  }
341 
342  // save mixed audio file ( will be used as audio stream )
343  QFile file(mTempWorkDir + "/tmpaudio.wav");
344  file.open(QIODevice::WriteOnly);
345 
346  WavFileHeader outputHeader;
347  outputHeader.InitWithDefaultValues();
348  outputHeader.dataSize = audioDataSize;
349  outputHeader.chuckSize = 36 + audioDataSize;
350 
351  file.write((char*)&outputHeader, sizeof(outputHeader));
352  file.write((char*)audioData.data(), audioDataSize);
353  file.close();
354 
355  return Status::OK;
356 }

374 Status MovieExporter::generateMovie(
375  const Object* obj,
376  QString ffmpegPath,
377  QString strOutputFile,
378  std::function<void(float)> progress)
379 {
380  if (mCanceled)
381  {
382  return Status::CANCELED;
383  }
384 
385  // Frame generation setup
386 
387  int frameStart = mDesc.startFrame;
388  int frameEnd = mDesc.endFrame;
389  const QSize exportSize = mDesc.exportSize;
390  bool transparency = false;
391  QString strCameraName = mDesc.strCameraName;
392  bool loop = mDesc.loop;
393 
394  auto cameraLayer = (LayerCamera*)obj->findLayerByName(strCameraName, Layer::CAMERA);
395  if (cameraLayer == nullptr)
396  {
397  cameraLayer = obj->getLayersByType< LayerCamera >().front();
398  }
399  int currentFrame = frameStart;
400 
401  /* We create an image with the correct dimensions and background
402  * color here and then copy this and draw over top of it to
403  * generate each frame. This is faster than having to generate
404  * a new background image for each frame.
405  */
406  QImage imageToExportBase(exportSize, QImage::Format_ARGB32_Premultiplied);
407  QColor bgColor = Qt::white;
408  if (transparency)
409  {
410  bgColor.setAlpha(0);
411  }
412  imageToExportBase.fill(bgColor);
413 
414  QSize camSize = cameraLayer->getViewSize();
415  QTransform centralizeCamera;
416  centralizeCamera.translate(camSize.width() / 2, camSize.height() / 2);
417 
418  int failCounter = 0;
419  /* Movie export uses a "sliding window" to reduce memory usage
420  * while having a relatively small impact on speed. This basically
421  * means that there is a maximum number of frames that can be waiting
422  * to be encoded by ffmpeg at any one time. The limit is set by the
423  * frameWindow variable which is designed to take up a maximum of
424  * about 1GB of memory
425  */
426  int frameWindow = (int) (1e9 / (camSize.width() * camSize.height() * 4.0));
427 
428  // Build FFmpeg command
429 
430  //int exportFps = mDesc.videoFps;
431  const QString tempAudioPath = mTempWorkDir + "/tmpaudio.wav";
432 
433  QString strCmd = QString("\"%1\"").arg(ffmpegPath);
434  strCmd += QString(" -f rawvideo -pixel_format bgra");
435  strCmd += QString(" -video_size %1x%2").arg(exportSize.width()).arg(exportSize.height());
436  strCmd += QString(" -framerate %1").arg(mDesc.fps);
437 
438  //strCmd += QString( " -r %1").arg( exportFps );
439  strCmd += QString(" -i -");
440  strCmd += QString(" -threads %1").arg(QThread::idealThreadCount() == 1 ? 0 : QThread::idealThreadCount());
441 
442  if (QFile::exists(tempAudioPath))
443  {
444  strCmd += QString(" -i \"%1\" ").arg(tempAudioPath);
445  }
446 
447  if(strOutputFile.endsWith(".apng"))
448  {
449  strCmd += QString(" -plays %1").arg(loop ? "0" : "1");
450  }
451 
452  if (strOutputFile.endsWith("mp4", Qt::CaseInsensitive))
453  {
454  strCmd += QString(" -pix_fmt yuv420p");
455  }
456  strCmd += " -y";
457  strCmd += QString(" \"%1\"").arg(strOutputFile);
458 
459  // Run FFmpeg command
460 
461  STATUS_CHECK(executeFFMpegPipe(strCmd, progress, [&](QProcess& ffmpeg, int framesProcessed)
462  {
463  if(framesProcessed < 0)
464  {
465  failCounter++;
466  }
467 
468  if(currentFrame > frameEnd)
469  {
470  ffmpeg.closeWriteChannel();
471  return false;
472  }
473 
474  if((currentFrame - frameStart <= framesProcessed + frameWindow || failCounter > 10) && currentFrame <= frameEnd)
475  {
476  QImage imageToExport = imageToExportBase.copy();
477  QPainter painter(&imageToExport);
478 
479  QTransform view = cameraLayer->getViewAtFrame(currentFrame);
480  painter.setWorldTransform(view * centralizeCamera);
481  painter.setWindow(QRect(0, 0, camSize.width(), camSize.height()));
482 
483  obj->paintImage(painter, currentFrame, false, true);
484  painter.end();
485 
486  // Should use sizeInBytes instead of byteCount to support large images,
487  // but this is only supported in QT 5.10+
488  int bytesWritten = ffmpeg.write(reinterpret_cast<const char*>(imageToExport.constBits()), imageToExport.byteCount());
489  Q_ASSERT(bytesWritten == imageToExport.byteCount());
490 
491  currentFrame++;
492  failCounter = 0;
493  return true;
494  }
495 
496  return false;
497  }));
498 
499  return Status::OK;
500 }
