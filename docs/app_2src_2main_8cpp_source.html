35 void installTranslator(PencilApplication& app)
36 {
37  QSettings setting(PENCIL2D, PENCIL2D);
38  QString strUserLocale = setting.value(SETTING_LANGUAGE).toString();
39  if (strUserLocale.isEmpty())
40  {
41  strUserLocale = QLocale::system().name();
42  }
43  QLocale::setDefault(QLocale(strUserLocale));
44 
45  strUserLocale.replace("-", "_");
46  QTranslator* qtTranslator = new QTranslator(&app);
47  qtTranslator->load("qt_" + strUserLocale, QLibraryInfo::location(QLibraryInfo::TranslationsPath));
48  app.installTranslator(qtTranslator);
49 
50  qDebug() << "Detect locale = " << strUserLocale;
51 
52  QTranslator* pencil2DTranslator = new QTranslator(&app);
53  bool b = pencil2DTranslator->load(":/qm/pencil_" + strUserLocale);
54 
55  qDebug() << "Load translation = " << b;
56 
57  b = app.installTranslator(pencil2DTranslator);
58 
59  qDebug() << "Install translation = " << b;
60 }

62 int handleArguments(PencilApplication& app)
63 {
64  QTextStream out(stdout);
65  QTextStream err(stderr);
66  QStringList args = PencilApplication::arguments();

72 
73  QCommandLineParser parser;
74  args.removeOne("-NSDocumentRevisionsDebugMode");
75 
76  parser.setApplicationDescription(QObject::tr("Pencil2D is an animation/drawing software for Mac OS X, Windows, and Linux. It lets you create traditional hand-drawn animation (cartoon) using both bitmap and vector graphics."));
77  parser.addHelpOption();
78  parser.addVersionOption();
79  parser.addPositionalArgument("input", QObject::tr("Path to the input pencil file."));
80 
81  QCommandLineOption exportOutOption(QStringList() << "o" << "export",
82  QObject::tr("Render the file to <output_path>"),
83  QObject::tr("output_path"));
84  parser.addOption(exportOutOption);
85 
86  // for backwards compatibility
87  QCommandLineOption exportSeqOption(QStringList() << "export-sequence",
88  QObject::tr("Render the file to <output_path>"),
89  QObject::tr("output_path"));
90 #if QT_VERSION >= QT_VERSION_CHECK(5, 8, 0)
91  exportSeqOption.setFlags(QCommandLineOption::HiddenFromHelp);
92 #endif
93  parser.addOption(exportSeqOption);
94 
95  QCommandLineOption cameraOption(QStringList() << "camera",
96  QObject::tr("Name of the camera layer to use"),
97  QObject::tr("layer_name"));
98  parser.addOption(cameraOption);
99 
100  QCommandLineOption widthOption(QStringList() << "width",
101  QObject::tr("Width of the output frames"),
102  QObject::tr("integer"));
103  parser.addOption(widthOption);
104 
105  QCommandLineOption heightOption(QStringList() << "height",
106  QObject::tr("Height of the output frames"),
107  QObject::tr("integer"));
108  parser.addOption(heightOption);
109 
110  QCommandLineOption startOption(QStringList() << "start",
111  QObject::tr("The first frame you want to include in the exported movie"),
112  QObject::tr("frame"));
113  parser.addOption(startOption);
114 
115  QCommandLineOption endOption(QStringList() << "end",
116  QObject::tr("The last frame you want to include in the exported movie. "
117  "Can also be last or last-sound to automatically use the last "
118  "frame containing animation or sound, respectively"),
119  QObject::tr("frame"));
120  parser.addOption(endOption);
121 
122  QCommandLineOption transparencyOption(QStringList() << "transparency",
123  QObject::tr("Render transparency when possible"));
124  parser.addOption(transparencyOption);
125 
126  parser.process(args);
127 
128  QStringList posArgs = parser.positionalArguments();
129  if (!posArgs.isEmpty())
130  {
131  inputPath = posArgs.at(0);
132  }
133 
134  outputPaths = parser.values(exportOutOption) << parser.values(exportSeqOption);
135 
136  if (!parser.value(widthOption).isEmpty())
137  {
138  bool ok = false;
139  width = parser.value(widthOption).toInt(&ok);
140  if (!ok)
141  {
142  err << QObject::tr("Warning: width value %1 is not an integer, ignoring.").arg(parser.value(widthOption)) << endl;
143  width = -1;
144  }
145  }
146  if (!parser.value(heightOption).isEmpty())
147  {
148  bool ok = false;
149  height = parser.value(heightOption).toInt(&ok);
150  if (!ok)
151  {
152  err << QObject::tr("Warning: height value %1 is not an integer, ignoring.").arg(parser.value(heightOption)) << endl;
153  height = -1;
154  }
155  }
156  if (!parser.value(startOption).isEmpty())
157  {
158  bool ok = false;
159  startFrame = parser.value(startOption).toInt(&ok);
160  if (!ok)
161  {
162  err << QObject::tr("Warning: start value %1 is not an integer, ignoring.").arg(parser.value(startOption)) << endl;
163  startFrame = 1;
164  }
165  if (startFrame < 1)
166  {
167  err << QObject::tr("Warning: start value must be at least 1, ignoring.") << endl;
168  startFrame = 1;
169  }
170  }
171  if (!parser.value(endOption).isEmpty())
172  {
173  if (parser.value(endOption) == "last")
174  {
175  endFrame = -1;
176  }
177  else if (parser.value(endOption) == "last-sound")
178  {
179  endFrame = -2;
180  }
181  else
182  {
183  bool ok = false;
184  endFrame = parser.value(endOption).toInt(&ok);
185  if (!ok)
186  {
187  err << QObject::tr("Warning: end value %1 is not an integer, last or last-sound, ignoring.").arg(parser.value(endOption)) << endl;
188  endFrame = -1;
189  }
190  }
191  if (endFrame > -1 && endFrame < startFrame)
192  {
193  err << QObject::tr("Warning: end value %1 is smaller than start value %2, ignoring.").arg(endFrame).arg(startFrame) << endl;
194  endFrame = startFrame;
195  }
196  }
197  transparency = parser.isSet(transparencyOption);
198 
199  if (!outputPaths.isEmpty())
200  {
201  if (inputPath.isEmpty())
202  {
203  // Error if there are output paths without an input path
204  err << QObject::tr("Error: No input file specified.") << endl;
205  return 1;
206  }
207 
208  QFileInfo inputFileInfo(inputPath);
209  if (!inputFileInfo.exists())
210  {
211  err << QObject::tr("Error: the input file at '%1' does not exist", "Command line error").arg(inputPath) << endl;
212  return 1;
213  }
214  if (!inputFileInfo.isFile())
215  {
216  err << QObject::tr("Error: the input path '%1' is not a file", "Command line error").arg(inputPath) << endl;
217  return 1;
218  }
219  }
220 
221  // Now that (almost) all possible user errors are handled, the actual program can be initialized
222  MainWindow2 mainWindow;
223  QObject::connect(&app, &PencilApplication::lostFocus, &mainWindow, &MainWindow2::appLostFocus);
224  QObject::connect(&app, &PencilApplication::openFileRequested, &mainWindow, &MainWindow2::openFile);
225  app.emitOpenFileRequest();
226 
227  // If there are no output paths, open up the GUI (to the input path if there is one)
228  if (outputPaths.isEmpty())
229  {
230  PlatformHandler::configurePlatformSpecificSettings();
231  mainWindow.show();
232  if (!inputPath.isEmpty())
233  {
234  mainWindow.openFile(inputPath);
235  }
236  return PencilApplication::exec();
237  }
238 
239  mainWindow.openFile(inputPath);
240 
241  if (!parser.value(cameraOption).isEmpty())
242  {
243  cameraLayer = dynamic_cast<LayerCamera*>(mainWindow.mEditor->layers()->findLayerByName(parser.value(cameraOption), Layer::CAMERA));
244  if (cameraLayer == nullptr)
245  {
246  err << QObject::tr("Warning: the specified camera layer %1 was not found, ignoring.").arg(parser.value(cameraOption)) << endl;
247  }
248  }
249  if (cameraLayer == nullptr)
250  {
251  cameraLayer = dynamic_cast<LayerCamera*>(mainWindow.mEditor->layers()->getLastCameraLayer());
252  }
253 
254  for (int i = 0; i < outputPaths.length(); i++)
255  {
256  // Detect format
257  QString format;
258  QMap<QString, QString> extensionMapping
259  {
260  { "png", "PNG" },
261  { "jpg" , "JPG" },
262  { "jpeg", "JPG" },
263  { "tif", "TIF" },
264  { "tiff", "TIF" },
265  { "bmp", "BMP" },
266  { "mp4", "MP4" },
267  { "avi", "AVI" },
268  { "gif", "GIF" },
269  { "webm", "WEBM" },
270  { "apng", "APNG" },
271  };
272  QString extension = outputPaths[i].mid(outputPaths[i].lastIndexOf(".") + 1).toLower();
273  if (inputPath.contains(".") && extensionMapping.contains(extension))
274  {
275  format = extensionMapping[extension];
276  }
277  else
278  {
279  err << QObject::tr("Warning: Output format is not specified or unsupported. Using PNG.", "Command line warning") << endl;
280  format = "PNG";
281  }
282 
283  QMap<QString, bool> formatMapping
284  {
285  { "PNG", false },
286  { "JPG", false },
287  { "TIF", false },
288  { "BMP", false },
289  { "MP4", true },
290  { "AVI", true },
291  { "GIF", true },
292  { "WEBM", true },
293  { "APNG", true },
294  };
295  bool asMovie = formatMapping[format];
296 
297  if (asMovie)
298  {
299  if (transparency)
300  {
301  err << QObject::tr("Warning: Transparency is not currently supported in movie files", "Command line warning") << endl;
302  }
303  out << QObject::tr("Exporting movie...", "Command line task progress") << endl;
304  mainWindow.mEditor->exportMovieCLI(outputPaths[i], cameraLayer, width, height, startFrame, endFrame);
305  out << QObject::tr("Done.", "Command line task done") << endl;
306  }
307  else {
308  out << QObject::tr("Exporting image sequence...", "Command line task progress") << endl;
309  mainWindow.mEditor->exportSeqCLI(outputPaths[i], cameraLayer, format, width, height, startFrame, endFrame, transparency);
310  out << QObject::tr("Done.", "Command line task done") << endl;
311  }
312  }
313 
314  return 0;
315 }
