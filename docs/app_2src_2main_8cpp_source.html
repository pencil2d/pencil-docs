35 void installTranslator(PencilApplication& app)
36 {
37  QSettings setting(PENCIL2D, PENCIL2D);
38  QString strUserLocale = setting.value(SETTING_LANGUAGE).toString();
39  if (strUserLocale.isEmpty())
40  {
41  strUserLocale = QLocale::system().name();
42  }
43 
44  strUserLocale.replace("-", "_");
45  QTranslator* qtTranslator = new QTranslator(&app);
46  qtTranslator->load("qt_" + strUserLocale, QLibraryInfo::location(QLibraryInfo::TranslationsPath));
47  app.installTranslator(qtTranslator);
48 
49  qDebug() << "Detect locale = " << strUserLocale;
50 
51  QTranslator* pencil2DTranslator = new QTranslator(&app);
52  bool b = pencil2DTranslator->load(":/qm/pencil_" + strUserLocale);
53 
54  qDebug() << "Load translation = " << b;
55 
56  b = app.installTranslator(pencil2DTranslator);
57 
58  qDebug() << "Install translation = " << b;
59 }

88  QObject::tr("output_path"));

105  QObject::tr("Height of the output frames"),
106  QObject::tr("integer"));
107  parser.addOption(heightOption);
108 
109  QCommandLineOption startOption(QStringList() << "start",

113 
114  QCommandLineOption endOption(QStringList() << "end",
115  QObject::tr("The last frame you want to include in the exported movie. "

123  parser.addOption(transparencyOption);

127  QStringList posArgs = parser.positionalArguments();
128  if (!posArgs.isEmpty())
129  {
130  inputPath = posArgs.at(0);
131  }
132 
133  outputPaths = parser.values(exportOutOption) << parser.values(exportSeqOption);
134 
135  if (!parser.value(widthOption).isEmpty())
136  {
137  bool ok = false;
138  width = parser.value(widthOption).toInt(&ok);
139  if (!ok)
140  {
141  err << QObject::tr("Warning: width value %1 is not an integer, ignoring.").arg(parser.value(widthOption)) << endl;
142  width = -1;

144  }
145  if (!parser.value(heightOption).isEmpty())
146  {
147  bool ok = false;

149  if (!ok)
150  {

152  height = -1;
153  }
154  }
155  if (!parser.value(startOption).isEmpty())
156  {
157  bool ok = false;
158  startFrame = parser.value(startOption).toInt(&ok);
159  if (!ok)
160  {
161  err << QObject::tr("Warning: start value %1 is not an integer, ignoring.").arg(parser.value(startOption)) << endl;
162  startFrame = 1;
163  }
164  if (startFrame < 1)
165  {
166  err << QObject::tr("Warning: start value must be at least 1, ignoring.") << endl;
167  startFrame = 1;
168  }
169  }
170  if (!parser.value(endOption).isEmpty())
171  {
172  if (parser.value(endOption) == "last")
173  {
174  endFrame = -1;
175  }
176  else if (parser.value(endOption) == "last-sound")
177  {
178  endFrame = -2;
179  }
180  else
181  {
182  bool ok = false;
183  endFrame = parser.value(endOption).toInt(&ok);
184  if (!ok)
185  {
186  err << QObject::tr("Warning: end value %1 is not an integer, last or last-sound, ignoring.").arg(parser.value(endOption)) << endl;
187  endFrame = -1;
188  }
189  }
190  if (endFrame > -1 && endFrame < startFrame)
191  {
192  err << QObject::tr("Warning: end value %1 is smaller than start value %2, ignoring.").arg(endFrame).arg(startFrame) << endl;
193  endFrame = startFrame;
194  }
195  }
196  transparency = parser.isSet(transparencyOption);
197 
198  if (!outputPaths.isEmpty())
199  {
200  if (inputPath.isEmpty())
201  {
202  // Error if there are output paths without an input path
203  err << QObject::tr("Error: No input file specified.") << endl;
204  return 1;
205  }

211  return 1;
212  }
213  if (!inputFileInfo.isFile())
214  {
215  err << QObject::tr("Error: the input path '%1' is not a file", "Command line error").arg(inputPath) << endl;
216  return 1;
217  }
218  }

223  app.emitOpenFileRequest();

233  }

241  cameraLayer = dynamic_cast<LayerCamera*>(mainWindow.mEditor->layers()->findLayerByName(parser.value(cameraOption), Layer::CAMERA));
242  if (cameraLayer == nullptr)

257  {

271  if (inputPath.contains(".") && extensionMapping.contains(extension))
272  {
273  format = extensionMapping[extension];
274  }
275  else
276  {
277  err << QObject::tr("Warning: Output format is not specified or unsupported. Using PNG.", "Command line warning") << endl;
278  format = "PNG";
279  }

282  {
283  { "PNG", false },
284  { "JPG", false },
285  { "TIF", false },
286  { "BMP", false },
287  { "MP4", true },
288  { "AVI", true },
289  { "GIF", true },
290  { "WEBM", true },
291  { "APNG", true },
292  };
293  bool asMovie = formatMapping[format];
294 
295  if (asMovie)
296  {
297  if (transparency)
298  {
299  err << QObject::tr("Warning: Transparency is not currently supported in movie files", "Command line warning") << endl;
300  }
301  out << QObject::tr("Exporting movie...", "Command line task progress") << endl;
302  mainWindow.mEditor->exportMovieCLI(outputPaths[i], cameraLayer, width, height, startFrame, endFrame);
303  out << QObject::tr("Done.", "Command line task done") << endl;
304  }
305  else {
306  out << QObject::tr("Exporting image sequence...", "Command line task progress") << endl;
307  mainWindow.mEditor->exportSeqCLI(outputPaths[i], cameraLayer, format, width, height, startFrame, endFrame, transparency);
308  out << QObject::tr("Done.", "Command line task done") << endl;
309  }
310  }
311 
312  return 0;
313 }
