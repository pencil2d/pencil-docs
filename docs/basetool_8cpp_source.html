125 QPixmap BaseTool::canvasCursor(float width, float feather, bool useFeather, float scalingFac, int windowWidth)
126 {
127  float propWidth = width * scalingFac;
128  float propFeather = feather * scalingFac;
129 
130  float cursorWidth = 0.0f;
131  float xyA = 0.0f;
132  float xyB = 0.0f;
133  float whA = 0.0f;
134  float whB = 0.0f;
135 
136  if (useFeather)
137  {
138  cursorWidth = propWidth + 0.5 * propFeather;
139  xyA = 1 + propFeather / 2;
140  xyB = 1 + propFeather / 8;
141  whA = qMax<float>(0, propWidth - xyA - 1);
142  whB = qMax<float>(0, cursorWidth - propFeather / 4 - 2);
143  }
144  else
145  {
146  cursorWidth = (propWidth + 0.5);
147  whA = qMax<float>(0, propWidth - 1);
148  whB = qMax<float>(0, cursorWidth / 4 - 2);
149  }
150 
151  float radius = cursorWidth / 2;
152 
153  // deallocate when cursor width gets some value larger than the widget
154  if (cursorWidth > windowWidth * 2)
155  {
156  return QPixmap(0, 0);
157  }
158 
159  if (cursorWidth < 1) { cursorWidth = 1; }
160 
161  QPixmap cursorPixmap = QPixmap(cursorWidth, cursorWidth);
162  if (!cursorPixmap.isNull())
163  {
164  cursorPixmap.fill(QColor(255, 255, 255, 0));
165  QPainter cursorPainter(&cursorPixmap);
166  QPen cursorPen = cursorPainter.pen();
167  cursorPainter.setRenderHint(QPainter::HighQualityAntialiasing);
168 
169  // Draw cross in center
170  cursorPen.setStyle(Qt::SolidLine);
171  cursorPen.setColor(QColor(0, 0, 0, 127));
172  cursorPainter.setPen(cursorPen);
173  cursorPainter.drawLine(QPointF(radius - 2, radius), QPointF(radius + 2, radius));
174  cursorPainter.drawLine(QPointF(radius, radius - 2), QPointF(radius, radius + 2));
175 
176  // Draw outer circle
177  if (useFeather)
178  {
179  cursorPen.setStyle(Qt::DotLine);
180  cursorPen.setColor(QColor(0, 0, 0, 255));
181  cursorPainter.setPen(cursorPen);
182  cursorPainter.drawEllipse(QRectF(xyB, xyB, whB, whB));
183  cursorPen.setDashOffset(4);
184  cursorPen.setColor(QColor(255, 255, 255, 255));
185  cursorPainter.setPen(cursorPen);
186  cursorPainter.drawEllipse(QRectF(xyB, xyB, whB, whB));
187  }
188 
189  // Draw inner circle
190  cursorPen.setStyle(Qt::DotLine);
191  cursorPen.setColor(QColor(0, 0, 0, 255));
192  cursorPainter.setPen(cursorPen);
193  cursorPainter.drawEllipse(QRectF(xyA, xyA, whA, whA));
194  cursorPen.setDashOffset(4);
195  cursorPen.setColor(QColor(255, 255, 255, 255));
196  cursorPainter.setPen(cursorPen);
197  cursorPainter.drawEllipse(QRectF(xyA, xyA, whA, whA));
198 
199  cursorPainter.end();
200  }
201  return cursorPixmap;
202 }

213 QPixmap BaseTool::quickSizeCursor(qreal scalingFac)
214 {
215  qreal propSize = qMax(0., properties.width) * scalingFac;
216  qreal propFeather = qMax(0., properties.feather) * scalingFac;
217  QRectF cursorRect(0, 0, propSize+2, propSize+2);
218 
219  QRectF sizeRect = cursorRect.adjusted(1, 1, -1, -1);
220  qreal featherRadius = (1 - propFeather / 100) * propSize / 2.;
221 
222  QPixmap cursorPixmap = QPixmap(cursorRect.size().toSize());
223  if (!cursorPixmap.isNull())
224  {
225  cursorPixmap.fill(QColor(255, 255, 255, 0));
226  QPainter cursorPainter(&cursorPixmap);
227  cursorPainter.setRenderHints(QPainter::Antialiasing, true);
228 
229  // Draw width (outside circle)
230  cursorPainter.setPen(QColor(255, 127, 127, 127));
231  cursorPainter.setBrush(QColor(0, 255, 127, 127));
232  cursorPainter.drawEllipse(sizeRect);
233 
234  // Draw feather (inside circle)
235  cursorPainter.setCompositionMode(QPainter::CompositionMode_Darken);
236  cursorPainter.setPen(QColor(0, 0, 0, 0));
237  cursorPainter.setBrush(QColor(0, 191, 95, 127));
238  cursorPainter.drawEllipse(cursorRect.center(), featherRadius, featherRadius);
239 
240  // Draw cursor in center
241  cursorPainter.setRenderHints(QPainter::Antialiasing, false);
242  cursorPainter.setPen(QColor(0, 0, 0, 255));
243  cursorPainter.drawLine(cursorRect.center() - QPoint(2, 0), cursorRect.center() + QPoint(2, 0));
244  cursorPainter.drawLine(cursorRect.center() - QPoint(0, 2), cursorRect.center() + QPoint(0, 2));
245 
246  cursorPainter.end();
247  }
248  return cursorPixmap;
249 }

251 bool BaseTool::startAdjusting(Qt::KeyboardModifiers modifiers, qreal step)
252 {
253  if (mQuickSizingProperties.contains(modifiers))
254  {
255  switch (mQuickSizingProperties.value(modifiers)) {
256  case WIDTH:
257  msOriginalPropertyValue = properties.width;
258  break;
259  case FEATHER:
260  msOriginalPropertyValue = properties.feather;
261  break;
262  case TOLERANCE:
263  msOriginalPropertyValue = properties.tolerance;
264  break;
265  default:
266  qDebug() << "Unhandled quick sizing property for tool" << typeName();
267  Q_ASSERT(false);
268  return false;
269  }
270 
271  msIsAdjusting = true;
272  mAdjustmentStep = step;
273  mScribbleArea->updateCanvasCursor();
274  return true;
275  }
276  return false;
277 }

287 void BaseTool::adjustCursor(Qt::KeyboardModifiers modifiers)
288 {
289  qreal inc = qPow(msOriginalPropertyValue * 100, 0.5);
290  qreal newValue = inc + getCurrentPoint().x();
291 
292  if (newValue < 0)
293  {
294  newValue = 0;
295  }
296 
297  newValue = qPow(newValue, 2) / 100;
298  if (mAdjustmentStep > 0)
299  {
300  int tempValue = static_cast<int>(newValue / mAdjustmentStep); // + 0.5 ?
301  newValue = tempValue * mAdjustmentStep;
302  }
303 
304  switch (mQuickSizingProperties.value(modifiers))
305  {
306  case WIDTH:
307  mEditor->tools()->setWidth(qBound(1., newValue, 200.));
308  break;
309  case FEATHER:
310  mEditor->tools()->setFeather(qBound(2., newValue, 200.));
311  break;
312  case TOLERANCE:
313  mEditor->tools()->setTolerance(qBound(0., newValue, 100.));
314  break;
315  default:
316  qDebug() << "Unhandled quick sizing property for tool" << typeName();
317  Q_ASSERT(false);
318  break;
319  }
320 }
