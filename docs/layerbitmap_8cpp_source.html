101 void LayerBitmap::loadDomElement(QDomElement element, QString dataDirPath)
102 {
103  if (!element.attribute("id").isNull())
104  {
105  int id = element.attribute("id").toInt();
106  setId(id);
107  }
108  setName(element.attribute("name"));
109  setVisible(element.attribute("visibility").toInt() == 1);
110 
111  QDomNode imageTag = element.firstChild();
112  while (!imageTag.isNull())
113  {
114  QDomElement imageElement = imageTag.toElement();
115  if (!imageElement.isNull())
116  {
117  if (imageElement.tagName() == "image")
118  {
119  QString path = dataDirPath + "/" + imageElement.attribute("src"); // the file is supposed to be in the data directory
120  //qDebug() << "LAY_BITMAP dataDirPath=" << dataDirPath << " ;path=" << path; //added for debugging puproses
121  QFileInfo fi(path);
122  if (!fi.exists()) path = imageElement.attribute("src");
123  int position = imageElement.attribute("frame").toInt();
124  int x = imageElement.attribute("topLeftX").toInt();
125  int y = imageElement.attribute("topLeftY").toInt();
126  loadImageAtFrame(path, QPoint(x, y), position);
127  }
128  }
129  imageTag = imageTag.nextSibling();
130  }
