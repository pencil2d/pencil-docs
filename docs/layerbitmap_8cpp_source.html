70Status LayerBitmap::saveKeyFrameFile(KeyFrame* keyframe, QString path)
71{
72 QString strFilePath = filePath(keyframe, QDir(path));
73
74 BitmapImage* bitmapImage = static_cast<BitmapImage*>(keyframe);
75
76 bool needSave = needSaveFrame(keyframe, strFilePath);
77 if (!needSave)
78 {
79 return Status::SAFE;
80 }
81
82 bitmapImage->setFileName(strFilePath);
83
84 Status st = bitmapImage->writeFile(strFilePath);
85 if (!st.ok())
86 {
87 bitmapImage->setFileName("");
88
89 DebugDetails dd;
90 dd << "LayerBitmap::saveKeyFrame";
91 dd << QString(" KeyFrame.pos() = %1").arg(keyframe->pos());
92 dd << QString(" strFilePath = %1").arg(strFilePath);
93 dd << QString("BitmapImage could not be saved");
94 dd.collect(st.details());
95 return Status(Status::FAIL, dd);
96 }
97
98 bitmapImage->setModified(false);
99 return Status::OK;
100}

110Status LayerBitmap::presave(const QString& sDataFolder)
111{
112 QDir dataFolder(sDataFolder);
113 // Handles keys that have been moved but not modified
114 std::vector<BitmapImage*> movedOnlyBitmaps;
115 foreachKeyFrame([&movedOnlyBitmaps,&dataFolder,this](KeyFrame* key)
116 {
117 auto bitmap = static_cast<BitmapImage*>(key);
118 // (b->fileName() != fileName(b) && !modified => the keyframe has been moved, but users didn't draw on it.
119 if (!bitmap->fileName().isEmpty()
120 && !bitmap->isModified()
121 && bitmap->fileName() != filePath(bitmap, dataFolder))
122 {
123 movedOnlyBitmaps.push_back(bitmap);
124 }
125 });
126
127 for (BitmapImage* b : movedOnlyBitmaps)
128 {
129 // Move to temporary locations first to avoid overwritting anything we shouldn't be
130 // Ex: Frame A moves from 1 -> 2, Frame B moves from 2 -> 3. Make sure A does not overwrite B
131 QString tmpPath = dataFolder.filePath(QString::asprintf("t_%03d.%03d.png", id(), b->pos()));
132 if (QFileInfo(b->fileName()).dir() != dataFolder) {
133 // Copy instead of move if the data folder itself has changed
134 QFile::copy(b->fileName(), tmpPath);
135 }
136 else {
137 QFile::rename(b->fileName(), tmpPath);
138 }
139 b->setFileName(tmpPath);
140 }
141
142 for (BitmapImage* b : movedOnlyBitmaps)
143 {
144 QString dest = filePath(b, dataFolder);
145 QFile::remove(dest);
146
147 QFile::rename(b->fileName(), dest);
148 b->setFileName(dest);
149 }
150
151 return Status::OK;
152}

164bool LayerBitmap::needSaveFrame(KeyFrame* key, const QString& savePath)
165{
166 if (key->isModified()) // keyframe was modified
167 return true;
168 if (QFile::exists(savePath) == false) // hasn't been saved before
169 return true;
170 if (key->fileName().isEmpty())
171 return true;
172 return false;
173}

175QDomElement LayerBitmap::createDomElement(QDomDocument& doc) const
176{
177 QDomElement layerElem = createBaseDomElement(doc);
178
179 foreachKeyFrame([&](KeyFrame* pKeyFrame)
180 {
181 BitmapImage* pImg = static_cast<BitmapImage*>(pKeyFrame);
182
183 QDomElement imageTag = doc.createElement("image");
184 imageTag.setAttribute("frame", pKeyFrame->pos());
185 imageTag.setAttribute("src", fileName(pKeyFrame));
186 imageTag.setAttribute("topLeftX", pImg->topLeft().x());
187 imageTag.setAttribute("topLeftY", pImg->topLeft().y());
188 imageTag.setAttribute("opacity", pImg->getOpacity());
189 layerElem.appendChild(imageTag);
190
191 Q_ASSERT(QFileInfo(pKeyFrame->fileName()).fileName() == fileName(pKeyFrame));
192 });
193
194 return layerElem;
195}

197void LayerBitmap::loadDomElement(const QDomElement& element, QString dataDirPath, ProgressCallback progressStep)
198{
199 this->loadBaseDomElement(element);
200
201 QDomNode imageTag = element.firstChild();
202 while (!imageTag.isNull())
203 {
204 QDomElement imageElement = imageTag.toElement();
205 if (!imageElement.isNull() && imageElement.tagName() == "image")
206 {
207 QString path = validateDataPath(imageElement.attribute("src"), dataDirPath);
208 if (!path.isEmpty())
209 {
210 int position = imageElement.attribute("frame").toInt();
211 int x = imageElement.attribute("topLeftX").toInt();
212 int y = imageElement.attribute("topLeftY").toInt();
213 qreal opacity = imageElement.attribute("opacity", "1.0").toDouble();
214 loadImageAtFrame(path, QPoint(x, y), position, opacity);
215 }
216
217 progressStep();
218 }
219 imageTag = imageTag.nextSibling();
220 }
221}
