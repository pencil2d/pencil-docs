69 QTransform LayerCamera::getViewAtFrame(int frameNumber) const
70 {
71  if (keyFrameCount() == 0)
72  {
73  return QTransform();
74  }
75 
76  // IMO: There should always be a keyframe on frame 1 on the Camera layer! (David)
77  if (frameNumber < firstKeyFramePosition())
78  {
79  frameNumber = firstKeyFramePosition();
80  }
81 
82  Camera* camera1 = static_cast<Camera*>(getLastKeyFrameAtPosition(frameNumber));
83 
84  int nextFrame = getNextKeyFramePosition(frameNumber);
85  Camera* camera2 = static_cast<Camera*>(getLastKeyFrameAtPosition(nextFrame));
86 
87  if (camera1 == nullptr && camera2 == nullptr)
88  {
89  return QTransform();
90  }
91  else if (camera1 == nullptr && camera2 != nullptr)
92  {
93  return camera2->getView();
94  }
95  else if (camera2 == nullptr && camera1 != nullptr)
96  {
97  return camera1->getView();
98  }
99 
100  if (camera1->compare(*camera2))
101  {
102  return camera1->getView();
103  }
104 
105  double frame1 = camera1->pos();
106  double frame2 = camera2->pos();
107 
108  // interpolation
109  qreal percent = getInterpolationPercent(camera1->getEasingType(), (frameNumber - frame1) / (frame2 - frame1));
110  auto lerp = [](double f1, double f2, double percent) -> double
111  {
112  return f1 * (1.0 - percent) + f2 * percent;
113  };
114 
115  QPointF controlPoint = camera1->getPathControlPoint();
116  if (!camera1->pathControlPointMoved()) {
117  controlPoint = getCenteredPathPoint(frame1);
118  }
119 
120  const QPointF& point = getBezierPoint(camera1->translation(), camera2->translation(),
121  -controlPoint, percent);
122 
123  double dx = point.x();
124  double dy = point.y();
125  double r = lerp(camera1->rotation(), camera2->rotation(), percent);
126  double s = lerp(camera1->scaling(), camera2->scaling(), percent);
127 
128  QTransform camTransform;
129  camTransform.scale(s, s);
130  camTransform.rotate(r);
131  camTransform.translate(dx, dy);
132 
133  return camTransform;
134 }

254 void LayerCamera::splitControlPointIfNeeded(int frame) const
255 {
256  int next = getNextKeyFramePosition(frame);
257  int prev = getPreviousKeyFramePosition(frame);
258 
259  // if inbetween frames
260  if (frame > prev && (frame > 1) && frame < next)
261  {
262  Camera* camFrame = getLastCameraAtFrame(frame, 0);
263  Camera* camPrev = getCameraAtFrame(prev);
264  Camera* camNext = getCameraAtFrame(next);
265  Q_ASSERT(camPrev && camFrame && camNext);
266 
267  if (camPrev->pathControlPointMoved()) {
268  qreal t = static_cast<qreal>(frame - prev) / (next - prev);
269  QPointF previousControlPoint = camPrev->getPathControlPoint();
270 
271  // Line from the current control point to the next frame
272  QLineF interpolatedLineCN = QLineF(previousControlPoint, -camNext->translation());
273 
274  // Line from the previous frame to the current control point
275  QLineF interpolatedLinePC = QLineF(-camPrev->translation(), previousControlPoint);
276  // Place the new control points, so the curve is kept
277  camPrev->setPathControlPoint(interpolatedLinePC.pointAt(t));
278  camFrame->setPathControlPoint(interpolatedLineCN.pointAt(t));
279  camFrame->setPathControlPointMoved(true);
280  }
281  } else {
282  Camera* camPrev = getCameraAtFrame(prev);
283  camPrev->setPathControlPointMoved(false);
284  }
285 }

287 void LayerCamera::mergeControlPointIfNeeded(int frame) const
288 {
289  int next = getNextKeyFramePosition(frame);
290  int prev = getPreviousKeyFramePosition(frame);
291 
292  // if inbetween frames
293  if (frame > prev && (frame > 1) && frame < next)
294  {
295  Camera* camPrev = getCameraAtFrame(prev);
296  Camera* camFrame = getLastCameraAtFrame(frame, 0);
297  Camera* camNext = getCameraAtFrame(next);
298  Q_ASSERT(camPrev && camFrame && camNext);
299 
300  if (camPrev->pathControlPointMoved()) {
301 
302  // Line from the current control point to the next frame
303  const QLineF& interpolatedLineCN = QLineF(camFrame->getPathControlPoint(), -camNext->translation());
304 
305  // Line from the previous frame translation to previous control point
306  const QLineF& interpolatedLinePC = QLineF(-camPrev->translation(), camPrev->getPathControlPoint());
307 
308  QPointF mergedCPoint;
309  auto intersection = interpolatedLinePC.intersect(interpolatedLineCN, &mergedCPoint);
310  // Try to recover the control point if the distance is within the threshold, otherwise do nothing
311  if (intersection == QLineF::IntersectType::UnboundedIntersection &&
312  QLineF(camFrame->getPathControlPoint(), mergedCPoint).length() < mControlPointMergeThreshold)
313  {
314  camPrev->setPathControlPoint(mergedCPoint);
315  camPrev->setPathControlPointMoved(true);
316  } else if (intersection == QLineF::IntersectType::NoIntersection) {
317  camPrev->setPathControlPointMoved(false);
318  }
319  }
320  }
321 }

345 void LayerCamera::resetCameraAtFrame(CameraFieldOption type, int frame) const
346 {
347  int frameToModify = frame;
348  if (!keyExists(frame)) {
349  frameToModify = getPreviousKeyFramePosition(frame);
350  }
351  Camera* camera = getLastCameraAtFrame(frameToModify, 0);
352 
353  switch (type)
354  {
355  case CameraFieldOption::RESET_FIELD:
356  camera->resetTransform();
357  break;
358  case CameraFieldOption::RESET_TRANSLATION:
359  camera->translate(QPoint(0,0));
360  break;
361  case CameraFieldOption::RESET_SCALING:
362  camera->scale(1.0);
363  break;
364  case CameraFieldOption::RESET_ROTATION:
365  camera->rotate(0.0);
366  break;
367  case CameraFieldOption::ALIGN_HORIZONTAL: {
368  qreal otherYCoord = camera->translation().y();
369  Camera* nextCam = getCameraAtFrame(getNextKeyFramePosition(frameToModify));
370  nextCam->translate(nextCam->translation().x(), otherYCoord);
371  camera->setPathControlPointMoved(false);
372  break;
373  }
374  case CameraFieldOption::ALIGN_VERTICAL: {
375  qreal otherXCoord = camera->translation().x();
376  Camera* nextCam = getCameraAtFrame(getNextKeyFramePosition(frameToModify));
377  nextCam->translate(otherXCoord, nextCam->translation().y());
378  camera->setPathControlPointMoved(false);
379  break;
380  }
381  case CameraFieldOption::HOLD_FRAME: {
382 
383  if (getMaxKeyFramePosition() == camera->pos()) { return; }
384 
385  QPointF translation = camera->translation();
386  qreal rotation = camera->rotation();
387  qreal scaling = camera->scaling();
388  camera->setPathControlPointMoved(false);
389  Camera* nextCamera = getLastCameraAtFrame(getNextKeyFramePosition(frame), 0);
390  nextCamera->translate(translation);
391  nextCamera->scale(scaling);
392  nextCamera->rotate(rotation);
393  nextCamera->setPathControlPointMoved(false);
394  break;
395  }
396  default:
397  Q_UNREACHABLE();
398  break;
399  }
400 
401  if (type != CameraFieldOption::RESET_SCALING && type != CameraFieldOption::RESET_ROTATION) {
402  setPathMovedAtFrame(frame - 1, false);
403  }
404 
405  camera->updateViewTransform();
406 }

462 QList<QPointF> LayerCamera::getBezierPointsAtFrame(int frame) const
463 {
464  QList<QPointF> points;
465  int prevFrame = getPreviousKeyFramePosition(frame);
466  int nextFrame = getNextKeyFramePosition(frame);
467  if (prevFrame < nextFrame)
468  {
469  Camera* prevCam = getCameraAtFrame(prevFrame);
470  Camera* nextCam = getCameraAtFrame(nextFrame);
471  points.append(QPointF(-prevCam->translation()));
472  if (prevCam->pathControlPointMoved()) {
473  points.append(QPointF(prevCam->getPathControlPoint()));
474  } else {
475  points.append(QPointF(getCenteredPathPoint(prevFrame)));
476  }
477  points.append(QPointF(-nextCam->translation()));
478  }
479  return points;

482 QPointF LayerCamera::getCenteredPathPoint(int frame) const
483 {

489

494  } else if (cam1 == nullptr && cam2 == nullptr) {
495  return QPointF();
496  }
497  return QLineF(-cam1->translation(), -cam2->translation()).pointAt(0.5);

586 void LayerCamera::loadDomElement(const QDomElement& element, QString dataDirPath, ProgressCallback progressStep)
587 {
588  Q_UNUSED(dataDirPath)
589  Q_UNUSED(progressStep)
590 
591  this->loadBaseDomElement(element);
592 
593  int width = element.attribute("width").toInt();
594  int height = element.attribute("height").toInt();
595  mShowPath = element.attribute("showPath").toInt();
596  updateDotColor(static_cast<DotColorType>(element.attribute("pathColorType").toInt()));
597  viewRect = QRect(-width / 2, -height / 2, width, height);
598 
599  QDomNode imageTag = element.firstChild();
600  while (!imageTag.isNull())
601  {
602  QDomElement imageElement = imageTag.toElement();
603  if (!imageElement.isNull())
604  {
605  if (imageElement.tagName() == "camera")
606  {
607  int frame = imageElement.attribute("frame").toInt();
608 
609  qreal rotate = imageElement.attribute("r", "0").toDouble();
610  qreal scale = imageElement.attribute("s", "1").toDouble();
611  qreal dx = imageElement.attribute("dx", "0").toDouble();
612  qreal dy = imageElement.attribute("dy", "0").toDouble();
613  CameraEasingType easing = static_cast<CameraEasingType>(imageElement.attribute("easing", "0").toInt());
614  qreal pathX = imageElement.attribute("pathCPX", "0").toDouble();
615  qreal pathY = imageElement.attribute("pathCPY", "0").toDouble();
616 
617  bool pathMoved = pathX != 0 || pathY != 0;
618 
619  loadImageAtFrame(frame, dx, dy, rotate, scale, easing, QPointF(pathX, pathY), pathMoved);
620  }
621  }
622  imageTag = imageTag.nextSibling();
623  }
