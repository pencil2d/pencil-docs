56 QTransform LayerCamera::getViewAtFrame(int frameNumber) const
57 {
58  if (keyFrameCount() == 0)
59  {
60  return QTransform();
61  }
62 
63  Camera* camera1 = static_cast<Camera*>(getLastKeyFrameAtPosition(frameNumber));
64  camera1->setEasingType(camera1->getEasingType());
65 
66  int nextFrame = getNextKeyFramePosition(frameNumber);
67  Camera* camera2 = static_cast<Camera*>(getLastKeyFrameAtPosition(nextFrame));
68  camera2->setEasingType(camera2->getEasingType());
69 
70  if (camera1 == nullptr && camera2 == nullptr)
71  {
72  return QTransform();
73  }
74  else if (camera1 == nullptr && camera2 != nullptr)
75  {
76  return camera2->view;
77  }
78  else if (camera2 == nullptr && camera1 != nullptr)
79  {
80  return camera1->view;
81  }
82 
83  if (camera1 == camera2)
84  {
85  return camera1->view;
86  }
87 
88  double frame1 = camera1->pos();
89  double frame2 = camera2->pos();
90 
91  // interpolation
92  qreal percent = getInterpolationPercent(camera1->getEasingType(), (frameNumber - frame1)/ (frame2 - frame1));
93 
94  auto interpolation = [=](double f1, double f2) -> double
95  {
96  return f1 * (1.0 - percent) + f2 * percent;
97  };
98 
99  return QTransform(interpolation(camera1->view.m11(), camera2->view.m11()),
100  interpolation(camera1->view.m12(), camera2->view.m12()),
101  interpolation(camera1->view.m21(), camera2->view.m21()),
102  interpolation(camera1->view.m22(), camera2->view.m22()),
103  interpolation(camera1->view.m31(), camera2->view.m31()),
104  interpolation(camera1->view.m32(), camera2->view.m32()));
105 }

107 void LayerCamera::linearInterpolateTransform(Camera* cam)
108 {
109  if (keyFrameCount() == 0)
110  return;
111 
112  int frameNumber = cam->pos();
113  Camera* camera1 = static_cast<Camera*>(getLastKeyFrameAtPosition(frameNumber - 1));
114  camera1->setEasingType(camera1->getEasingType());
115 
116  int nextFrame = getNextKeyFramePosition(frameNumber);
117  Camera* camera2 = static_cast<Camera*>(getLastKeyFrameAtPosition(nextFrame));
118  camera2->setEasingType(camera2->getEasingType());
119 
120  if (camera1 == nullptr && camera2 == nullptr)
121  {
122  return; // do nothing
123  }
124  else if (camera1 == nullptr && camera2 != nullptr)
125  {
126  return cam->assign(*camera2);
127  }
128  else if (camera2 == nullptr && camera1 != nullptr)
129  {
130  return cam->assign(*camera1);
131  }
132 
133  if (camera1 == camera2)
134  {
135  return cam->assign(*camera1);
136  }
137 
138  double frame1 = camera1->pos();
139  double frame2 = camera2->pos();
140 
141  // interpolation
142  qreal percent = getInterpolationPercent(camera1->getEasingType(), (frameNumber - frame1)/ (frame2 - frame1));
143 
144  auto lerp = [](double f1, double f2, double percent) -> double
145  {
146  return f1 * (1.0 - percent) + f2 * percent;
147  };
148 
149  double dx = lerp(camera1->translation().x(), camera2->translation().x(), percent);
150  double dy = lerp(camera1->translation().y(), camera2->translation().y(), percent);
151  double r = lerp(camera1->rotation(), camera2->rotation(), percent);
152  double s = lerp(camera1->scaling(), camera2->scaling(), percent);
153 
154  cam->translate(dx, dy);
155  cam->rotate(r);
156  cam->scale(s);
157 }

159 qreal LayerCamera::getInterpolationPercent(CameraEasingType type, qreal percent) const

193  case CameraEasingType::OUTINCIRC: easing.setType(QEasingCurve::OutInCirc); break;
194  default: easing.setType(QEasingCurve::Linear); break;

261 void LayerCamera::loadDomElement(const QDomElement& element, QString dataDirPath, ProgressCallback progressStep)
262 {
263  Q_UNUSED(dataDirPath);
264  Q_UNUSED(progressStep);
265 
266  this->loadBaseDomElement(element);
267 
268  int width = element.attribute("width").toInt();
269  int height = element.attribute("height").toInt();
270  viewRect = QRect(-width / 2, -height / 2, width, height);
271 
272  QDomNode imageTag = element.firstChild();

277  {

279  {
280  int frame = imageElement.attribute("frame").toInt();
281 
282  qreal rotate = imageElement.attribute("r", "0").toDouble();
283  qreal scale = imageElement.attribute("s", "1").toDouble();
284  qreal dx = imageElement.attribute("dx", "0").toDouble();
285  qreal dy = imageElement.attribute("dy", "0").toDouble();

289  }
