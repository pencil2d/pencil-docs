86void CanvasPainter::paintCached(const QRect& blitRect)
87{
88 if (!mPreLayersPixmapCacheValid)
89 {
90 QPainter preLayerPainter;
91 initializePainter(preLayerPainter, mPreLayersPixmap, blitRect);
92 renderPreLayers(preLayerPainter, blitRect);
93 preLayerPainter.end();
94 mPreLayersPixmapCacheValid = true;
95 }
96
97 QPainter mainPainter;
98 initializePainter(mainPainter, mCanvas, blitRect);
99 mainPainter.setWorldMatrixEnabled(false);
100 mainPainter.drawPixmap(mPointZero, mPreLayersPixmap);
101 mainPainter.setWorldMatrixEnabled(true);
102
103 paintCurrentFrame(mainPainter, blitRect, mCurrentLayerIndex, mCurrentLayerIndex);
104
105 if (!mPostLayersPixmapCacheValid)
106 {
107 QPainter postLayerPainter;
108 initializePainter(postLayerPainter, mPostLayersPixmap, blitRect);
109 renderPostLayers(postLayerPainter, blitRect);
110 postLayerPainter.end();
111 mPostLayersPixmapCacheValid = true;
112 }
113
114 mainPainter.setWorldMatrixEnabled(false);
115 mainPainter.drawPixmap(mPointZero, mPostLayersPixmap);
116 mainPainter.setWorldMatrixEnabled(true);
117}

125void CanvasPainter::initializePainter(QPainter& painter, QPaintDevice& device, const QRect& blitRect)
126{
127 painter.begin(&device);
128
129 // Only draw inside the clipped rectangle
130 painter.setClipRect(blitRect);
131
132 // Clear the area that's about to be painted again, to avoid painting on top of existing pixels
133 // causing artifacts.
134 painter.setCompositionMode(QPainter::CompositionMode_Clear);
135 painter.fillRect(blitRect, Qt::transparent);
136
137 // Surface has been cleared and is ready to be painted on
138 painter.setCompositionMode(QPainter::CompositionMode_SourceOver);
139 painter.setWorldMatrixEnabled(true);
140 painter.setWorldTransform(mViewTransform);
141}

143void CanvasPainter::renderPreLayers(QPainter& painter, const QRect& blitRect)
144{
145 if (mOptions.eLayerVisibility != LayerVisibility::CURRENTONLY || mObject->getLayer(mCurrentLayerIndex)->type() == Layer::CAMERA)
146 {
147 paintCurrentFrame(painter, blitRect, 0, mCurrentLayerIndex - 1);
148 }
149
150 paintOnionSkin(painter, blitRect);
151 painter.setOpacity(1.0);
152}

173void CanvasPainter::paint(const QRect& blitRect)
174{
175 QPainter preLayerPainter;
176 QPainter mainPainter;
177 QPainter postLayerPainter;
178
179 initializePainter(mainPainter, mCanvas, blitRect);
180
181 initializePainter(preLayerPainter, mPreLayersPixmap, blitRect);
182 renderPreLayers(preLayerPainter, blitRect);
183 preLayerPainter.end();
184
185 mainPainter.setWorldMatrixEnabled(false);
186 mainPainter.drawPixmap(mPointZero, mPreLayersPixmap);
187 mainPainter.setWorldMatrixEnabled(true);
188
189 paintCurrentFrame(mainPainter, blitRect, mCurrentLayerIndex, mCurrentLayerIndex);
190
191 initializePainter(postLayerPainter, mPostLayersPixmap, blitRect);
192 renderPostLayers(postLayerPainter, blitRect);
193 postLayerPainter.end();
194
195 mainPainter.setWorldMatrixEnabled(false);
196 mainPainter.drawPixmap(mPointZero, mPostLayersPixmap);
197 mainPainter.setWorldMatrixEnabled(true);
198
199 mPreLayersPixmapCacheValid = true;
200 mPostLayersPixmapCacheValid = true;
201}

203void CanvasPainter::paintOnionSkin(QPainter& painter, const QRect& blitRect)
204{
205 Layer* layer = mObject->getLayer(mCurrentLayerIndex);
206
207 mOnionSkinSubPainter.paint(painter, layer, mOnionSkinPainterOptions, mFrameNumber, [&] (OnionSkinPaintState state, int onionFrameNumber) {
208 if (state == OnionSkinPaintState::PREV) {
209 switch (layer->type())
210 {
211 case Layer::BITMAP: { paintBitmapOnionSkinFrame(painter, blitRect, layer, onionFrameNumber, mOnionSkinPainterOptions.colorizePrevFrames); break; }
212 case Layer::VECTOR: { paintVectorOnionSkinFrame(painter, blitRect, layer, onionFrameNumber, mOnionSkinPainterOptions.colorizePrevFrames); break; }
213 default: break;
214 }
215 }
216 if (state == OnionSkinPaintState::NEXT) {
217 switch (layer->type())
218 {
219 case Layer::BITMAP: { paintBitmapOnionSkinFrame(painter, blitRect, layer, onionFrameNumber, mOnionSkinPainterOptions.colorizeNextFrames); break; }
220 case Layer::VECTOR: { paintVectorOnionSkinFrame(painter, blitRect, layer, onionFrameNumber, mOnionSkinPainterOptions.colorizeNextFrames); break; }
221 default: break;
222 }
223 }
224 });
225}

227void CanvasPainter::paintBitmapOnionSkinFrame(QPainter& painter, const QRect& blitRect, Layer* layer, int nFrame, bool colorize)
228{
229 LayerBitmap* bitmapLayer = static_cast<LayerBitmap*>(layer);
230
231 BitmapImage* bitmapImage = bitmapLayer->getBitmapImageAtFrame(nFrame);
232
233 if (bitmapImage == nullptr) { return; }
234 bitmapImage->loadFile(); // Critical! force the BitmapImage to load the image
235
236 QPainter onionSkinPainter;
237 initializePainter(onionSkinPainter, mOnionSkinPixmap, blitRect);
238
239 onionSkinPainter.drawImage(bitmapImage->topLeft(), *bitmapImage->image());
240 paintOnionSkinFrame(painter, onionSkinPainter, nFrame, colorize, bitmapImage->getOpacity());
241}

258void CanvasPainter::paintOnionSkinFrame(QPainter& painter, QPainter& onionSkinPainter, int nFrame, bool colorize, qreal frameOpacity)
259{
260 // Don't transform the image here as we used the viewTransform in the image output
261 painter.setWorldMatrixEnabled(false);
262 // Remember to adjust overall opacity based on opacity value from image
263 painter.setOpacity(frameOpacity - (1.0-painter.opacity()));
264 if (colorize)
265 {
266 QColor colorBrush = Qt::transparent; //no color for the current frame
267
268 if (nFrame < mFrameNumber)
269 {
270 colorBrush = Qt::red;
271 }
272 else if (nFrame > mFrameNumber)
273 {
274 colorBrush = Qt::blue;
275 }
276 onionSkinPainter.setWorldMatrixEnabled(false);
277
278 onionSkinPainter.setCompositionMode(QPainter::CompositionMode_SourceIn);
279 onionSkinPainter.setBrush(colorBrush);
280 onionSkinPainter.drawRect(painter.viewport());
281 }
282 painter.drawPixmap(mPointZero, mOnionSkinPixmap);
283}

285void CanvasPainter::paintCurrentBitmapFrame(QPainter& painter, const QRect& blitRect, Layer* layer, bool isCurrentLayer)
286{
287 LayerBitmap* bitmapLayer = static_cast<LayerBitmap*>(layer);
288 BitmapImage* paintedImage = bitmapLayer->getLastBitmapImageAtFrame(mFrameNumber);
289
290 if (paintedImage == nullptr) { return; }
291 paintedImage->loadFile(); // Critical! force the BitmapImage to load the image
292
293 const bool isDrawing = mTiledBuffer && !mTiledBuffer->bounds().isEmpty();
294
295 QPainter currentBitmapPainter;
296 initializePainter(currentBitmapPainter, mCurrentLayerPixmap, blitRect);
297
298 painter.setOpacity(paintedImage->getOpacity() - (1.0-painter.opacity()));
299 painter.setWorldMatrixEnabled(false);
300
301 currentBitmapPainter.drawImage(paintedImage->topLeft(), *paintedImage->image());
302
303 if (isCurrentLayer && isDrawing)
304 {
305 currentBitmapPainter.setCompositionMode(mOptions.cmBufferBlendMode);
306 const auto tiles = mTiledBuffer->tiles();
307 for (const Tile* tile : tiles) {
308 currentBitmapPainter.drawPixmap(tile->posF(), tile->pixmap());
309 }
310 }
311
312 // We do not wish to draw selection transformations on anything but the current layer
313 Q_ASSERT(!isDrawing || mSelectionTransform.isIdentity());
314 if (isCurrentLayer && mRenderTransform && !isDrawing) {
315 paintTransformedSelection(currentBitmapPainter, paintedImage, mSelection);
316 }
317
318 painter.drawPixmap(mPointZero, mCurrentLayerPixmap);
319}

321void CanvasPainter::paintCurrentVectorFrame(QPainter& painter, const QRect& blitRect, Layer* layer, bool isCurrentLayer)
322{
323 LayerVector* vectorLayer = static_cast<LayerVector*>(layer);
324 VectorImage* vectorImage = vectorLayer->getLastVectorImageAtFrame(mFrameNumber, 0);
325 if (vectorImage == nullptr)
326 {
327 return;
328 }
329
330 QPainter currentVectorPainter;
331 initializePainter(currentVectorPainter, mCurrentLayerPixmap, blitRect);
332
333 const bool isDrawing = mTiledBuffer->isValid();
334
335 if (mRenderTransform) {
336 vectorImage->setSelectionTransformation(mSelectionTransform);
337 }
338
339 // Paint existing vector image to the painter
340 vectorImage->paintImage(currentVectorPainter, *mObject, mOptions.bOutlines, mOptions.bThinLines, mOptions.bAntiAlias);
341
342 if (isCurrentLayer && isDrawing) {
343 currentVectorPainter.setCompositionMode(mOptions.cmBufferBlendMode);
344
345 const auto tiles = mTiledBuffer->tiles();
346 for (const Tile* tile : tiles) {
347 currentVectorPainter.drawPixmap(tile->posF(), tile->pixmap());
348 }
349 }
350
351 // Don't transform the image here as we used the viewTransform in the image output
352 painter.setWorldMatrixEnabled(false);
353 painter.setTransform(QTransform());
354
355 // Remember to adjust opacity based on additional opacity value from the keyframe
356 painter.setOpacity(vectorImage->getOpacity() - (1.0-painter.opacity()));
357 painter.drawPixmap(mPointZero, mCurrentLayerPixmap);
358}

360void CanvasPainter::paintTransformedSelection(QPainter& painter, BitmapImage* bitmapImage, const QRect& selection) const
361{
362 // Make sure there is something selected
363 if (selection.width() == 0 && selection.height() == 0)
364 return;
365
366 QPixmap transformedPixmap = QPixmap(mSelection.size());
367 transformedPixmap.fill(Qt::transparent);
368
369 QPainter imagePainter(&transformedPixmap);
370 imagePainter.translate(-selection.topLeft());
371 imagePainter.drawImage(bitmapImage->topLeft(), *bitmapImage->image());
372 imagePainter.end();
373
374 painter.save();
375
376 painter.setTransform(mViewTransform);
377
378 // Clear the painted area to make it look like the content has been erased
379 painter.save();
380 painter.setCompositionMode(QPainter::CompositionMode_Clear);
381 painter.fillRect(selection, QColor(255,255,255,255));
382 painter.restore();
383
384 // Multiply the selection and view matrix to get proper rotation and scale values
385 // Now the image origin will be topleft
386 painter.setTransform(mSelectionTransform*mViewTransform);
387
388 // Draw the selection image separately and on top
389 painter.drawPixmap(selection, transformedPixmap);
390 painter.restore();
391}

399void CanvasPainter::paintCurrentFrame(QPainter& painter, const QRect& blitRect, int startLayer, int endLayer)
400{
401 painter.setOpacity(1.0);
402
403 bool isCameraLayer = mObject->getLayer(mCurrentLayerIndex)->type() == Layer::CAMERA;
404
405 for (int i = startLayer; i <= endLayer; ++i)
406 {
407 Layer* layer = mObject->getLayer(i);
408
409 if (!layer->visible())
410 continue;
411
412 if (mOptions.eLayerVisibility == LayerVisibility::RELATED && !isCameraLayer)
413 {
414 painter.setOpacity(calculateRelativeOpacityForLayer(mCurrentLayerIndex, i, mOptions.fLayerVisibilityThreshold));
415 }
416 bool isCurrentLayer = mCurrentLayerIndex == i;
417
418 CANVASPAINTER_LOG(" Render Layer[%d] %s", i, layer->name());
419 switch (layer->type())
420 {
421 case Layer::BITMAP: { paintCurrentBitmapFrame(painter, blitRect, layer, isCurrentLayer); break; }
422 case Layer::VECTOR: { paintCurrentVectorFrame(painter, blitRect, layer, isCurrentLayer); break; }
423 default: break;
424 }
425 }
