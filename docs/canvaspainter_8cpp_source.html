74 void CanvasPainter::paintCached()
75 {
76  QPixmap tempPixmap(mCanvas->size());
77  tempPixmap.fill(Qt::transparent);
78  mCanvas->fill(Qt::transparent);
79  QPainter tempPainter;
80  QPainter painter;
81  initializePainter(tempPainter, tempPixmap);
82  initializePainter(painter, *mCanvas);
83 
84  if (!mPreLayersCache)
85  {
86  renderPreLayers(painter);
87  mPreLayersCache.reset(new QPixmap(*mCanvas));
88  }
89  else
90  {
91  painter.setWorldMatrixEnabled(false);
92  painter.drawPixmap(0, 0, *(mPreLayersCache.get()));
93  painter.setWorldMatrixEnabled(true);
94  }
95 
96  renderCurLayer(painter);
97 
98  if (!mPostLayersCache)
99  {
100  renderPostLayers(tempPainter);
101  mPostLayersCache.reset(new QPixmap(tempPixmap));
102  painter.setWorldMatrixEnabled(false);
103  painter.drawPixmap(0, 0, tempPixmap);
104  painter.setWorldMatrixEnabled(true);
105  }
106  else
107  {
108  painter.setWorldMatrixEnabled(false);
109  painter.drawPixmap(0, 0, *(mPostLayersCache.get()));
110  painter.setWorldMatrixEnabled(true);
111  }
112 }

199 void CanvasPainter::paintOnionSkin(QPainter& painter)
200 {
201  Layer* layer = mObject->getLayer(mCurrentLayerIndex);
202 
203  mOnionSkinSubPainter.paint(painter, layer, mOnionSkinPainterOptions, mFrameNumber, [&] (OnionSkinPaintState state, int onionFrameNumber) {
204  if (state == OnionSkinPaintState::PREV) {
205  switch (layer->type())
206  {
207  case Layer::BITMAP: { paintBitmapFrame(painter, layer, onionFrameNumber, mOnionSkinPainterOptions.colorizePrevFrames, false, false); break; }
208  case Layer::VECTOR: { paintVectorFrame(painter, layer, onionFrameNumber, mOnionSkinPainterOptions.colorizePrevFrames, false, false); break; }
209  default: break;
210  }
211  }
212  if (state == OnionSkinPaintState::NEXT) {
213  switch (layer->type())
214  {
215  case Layer::BITMAP: { paintBitmapFrame(painter, layer, onionFrameNumber, mOnionSkinPainterOptions.colorizeNextFrames, false, false); break; }
216  case Layer::VECTOR: { paintVectorFrame(painter, layer, onionFrameNumber, mOnionSkinPainterOptions.colorizeNextFrames, false, false); break; }
217  default: break;
218  }
219  }
220  });
221 }

223 void CanvasPainter::paintBitmapFrame(QPainter& painter,
224  Layer* layer,
225  int nFrame,
226  bool colorize,
227  bool useLastKeyFrame,
228  bool isCurrentFrame)
229 {
230 #ifdef _DEBUG
231  LayerBitmap* bitmapLayer = dynamic_cast<LayerBitmap*>(layer);
232  Q_ASSERT(bitmapLayer);
233 #else
234  LayerBitmap* bitmapLayer = static_cast<LayerBitmap*>(layer);
235 #endif
236 
237  CANVASPAINTER_LOG(" Paint Bitmap Frame = %d, UseLastKeyFrame = %d", nFrame, useLastKeyFrame);
238  BitmapImage* paintedImage = nullptr;
239  if (useLastKeyFrame)
240  {
241  paintedImage = bitmapLayer->getLastBitmapImageAtFrame(nFrame, 0);
242  CANVASPAINTER_LOG(" Actual frame = %d", paintedImage->pos());
243  }
244  else
245  {
246  paintedImage = bitmapLayer->getBitmapImageAtFrame(nFrame);
247  }
248 
249  if (paintedImage == nullptr) { return; }
250  paintedImage->loadFile(); // Critical! force the BitmapImage to load the image
251  CANVASPAINTER_LOG(" Paint Image Size: %dx%d", paintedImage->image()->width(), paintedImage->image()->height());
252 
253  const bool frameIsEmpty = (paintedImage == nullptr || paintedImage->bounds().isEmpty());
254  const bool isDrawing = isCurrentFrame && mBuffer && !mBuffer->bounds().isEmpty();
255  if (frameIsEmpty && !isDrawing)
256  {
257  CANVASPAINTER_LOG(" Early return frame %d, %d", frameIsEmpty, isDrawing);
258  return;
259  }
260 
261  BitmapImage paintToImage = BitmapImage(paintedImage->bounds(), Qt::transparent);
262  paintToImage.paste(paintedImage);
263 
264  painter.setOpacity(paintedImage->getOpacity() - (1.0-painter.opacity()));
265  if (isCurrentFrame)
266  {
267  paintToImage.paste(mBuffer, mOptions.cmBufferBlendMode);
268  }
269 
270  if (colorize)
271  {
272  QBrush colorBrush = QBrush(Qt::transparent); //no color for the current frame
273 
274  if (nFrame < mFrameNumber)
275  {
276  colorBrush = QBrush(Qt::red);
277  }
278  else if (nFrame > mFrameNumber)
279  {
280  colorBrush = QBrush(Qt::blue);
281  }
282 
283  paintToImage.drawRect(paintedImage->bounds(),
284  Qt::NoPen,
285  colorBrush,
286  QPainter::CompositionMode_SourceIn,
287  false);
288  }
289 
290  // If the current frame on the current layer has a transformation, we apply it.
291  bool shouldPaintTransform = mRenderTransform && nFrame == mFrameNumber && layer == mObject->getLayer(mCurrentLayerIndex);
292  if (shouldPaintTransform)
293  {
294  paintToImage.clear(mSelection);
295  }
296 
297  painter.setWorldMatrixEnabled(true);
298  prescale(&paintToImage);
299  paintToImage.paintImage(painter, mScaledBitmap, mScaledBitmap.rect(), paintToImage.bounds());
300 
301  if (shouldPaintTransform)
302  {
303  paintTransformedSelection(painter);
304  }
305 // static int cc = 0;
306 // QString path = QString("C:/Temp/pencil2d/canvas-%1-%2-%3.png")
307 // .arg(cc++, 3, 10, QChar('0'))
308 // .arg(layer->name())
309 // .arg(mFrameNumber);
310 // Q_ASSERT(mCanvas->save(path));
311

336 void CanvasPainter::paintVectorFrame(QPainter& painter,
337  Layer* layer,
338  int nFrame,
339  bool colorize,
340  bool useLastKeyFrame,
341  bool isCurrentFrame)
342 {
343 #ifdef _DEBUG
344  LayerVector* vectorLayer = dynamic_cast<LayerVector*>(layer);
345  Q_ASSERT(vectorLayer);
346 #else
347  LayerVector* vectorLayer = static_cast<LayerVector*>(layer);
348 #endif
349 
350  CANVASPAINTER_LOG("Paint Onion skin vector, Frame = %d", nFrame);
351  VectorImage* vectorImage = nullptr;
352  if (useLastKeyFrame)
353  {
354  vectorImage = vectorLayer->getLastVectorImageAtFrame(nFrame, 0);
355  }
356  else
357  {
358  vectorImage = vectorLayer->getVectorImageAtFrame(nFrame);
359  }
360  if (vectorImage == nullptr)
361  {
362  return;
363  }
364 
365  QImage* strokeImage = new QImage(mCanvas->size(), QImage::Format_ARGB32_Premultiplied);
366 
367  if (mRenderTransform) {
368  vectorImage->setSelectionTransformation(mSelectionTransform);
369  }
370 
371  vectorImage->outputImage(strokeImage, mViewTransform, mOptions.bOutlines, mOptions.bThinLines, mOptions.bAntiAlias);
372 
373  // Go through a Bitmap image to paint the onion skin colour
374  BitmapImage rasterizedVectorImage;
375  rasterizedVectorImage.setImage(strokeImage);
376 
377  if (colorize)
378  {
379  QBrush colorBrush = QBrush(Qt::transparent); //no color for the current frame
380 
381  if (nFrame < mFrameNumber)
382  {
383  colorBrush = QBrush(Qt::red);
384  }
385  else if (nFrame > mFrameNumber)
386  {
387  colorBrush = QBrush(Qt::blue);
388  }
389  rasterizedVectorImage.drawRect(strokeImage->rect(),
390  Qt::NoPen, colorBrush,
391  QPainter::CompositionMode_SourceIn, false);
392  }
393 
394  // Don't transform the image here as we used the viewTransform in the image output
395  painter.setWorldMatrixEnabled(false);
396 
397  painter.setOpacity(vectorImage->getOpacity() - (1.0-painter.opacity()));
398  if (isCurrentFrame)
399  {
400  // Paste buffer onto image to see stroke in realtime
401  rasterizedVectorImage.paste(mBuffer, mOptions.cmBufferBlendMode);
402  }
403 
404  // Paint buffer pasted on top of vector image:
405  // fixes polyline not being rendered properly
406  rasterizedVectorImage.paintImage(painter);
407 }

409 void CanvasPainter::paintTransformedSelection(QPainter& painter) const
410 {
411  // Make sure there is something selected
412  if (mSelection.width() == 0 || mSelection.height() == 0)
413  return;
414 
415  Layer* layer = mObject->getLayer(mCurrentLayerIndex);
416 
417  if (layer->type() == Layer::BITMAP)
418  {
419  // Get the transformed image
420  BitmapImage* bitmapImage = static_cast<LayerBitmap*>(layer)->getLastBitmapImageAtFrame(mFrameNumber, 0);
421  if (bitmapImage == nullptr) { return; };
422  BitmapImage transformedImage = bitmapImage->transformed(mSelection, mSelectionTransform, mOptions.bAntiAlias);
423 
424  // Paint the transformation output
425  painter.setWorldMatrixEnabled(true);
426  transformedImage.paintImage(painter);

436 void CanvasPainter::paintCurrentFrame(QPainter& painter, int startLayer, int endLayer)
437 {
438  painter.setOpacity(1.0);
439 
440 
441  bool isCameraLayer = mObject->getLayer(mCurrentLayerIndex)->type() == Layer::CAMERA;
442 
443  for (int i = startLayer; i <= endLayer; ++i)
444  {
445  Layer* layer = mObject->getLayer(i);
446 
447  if (layer->visible() == false)
448  continue;
449 
450  if (mOptions.eLayerVisibility == LayerVisibility::RELATED && !isCameraLayer)
451  {
452  painter.setOpacity(calculateRelativeOpacityForLayer(mCurrentLayerIndex, i, mOptions.fLayerVisibilityThreshold));
453  }
454 
455  CANVASPAINTER_LOG(" Render Layer[%d] %s", i, layer->name());
456  switch (layer->type())
457  {
458  case Layer::BITMAP: { paintBitmapFrame(painter, layer, mFrameNumber, false, true, i == mCurrentLayerIndex); break; }
459  case Layer::VECTOR: { paintVectorFrame(painter, layer, mFrameNumber, false, true, i == mCurrentLayerIndex); break; }
460  default: break;
461  }
462  }
463 }
