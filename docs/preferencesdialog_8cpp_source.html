406 void TimelinePage::updateValues()
407 {
408  QSignalBlocker b1(ui->scrubBox);
409  ui->scrubBox->setChecked(mManager->isOn(SETTING::SHORT_SCRUB));
410 
411  QSignalBlocker b3(ui->timelineLength);
412  ui->timelineLength->setValue(mManager->getInt(SETTING::TIMELINE_SIZE));
413  if (mManager->getString(SETTING::TIMELINE_SIZE).toInt() <= 0)
414  ui->timelineLength->setValue(240);
415 
416  QSignalBlocker b4(ui->radioButtonAddNewKey);
417  QSignalBlocker b5(ui->radioButtonDuplicate);
418  QSignalBlocker b6(ui->radioButtonDrawOnPrev);
419  int action = mManager->getInt(SETTING::DRAW_ON_EMPTY_FRAME_ACTION);
420  switch (action)
421  {
422  case CREATE_NEW_KEY:
423  ui->radioButtonAddNewKey->setChecked(true);
424  break;
425  case DUPLICATE_PREVIOUS_KEY:
426  ui->radioButtonDuplicate->setChecked(true);
427  break;
428  case KEEP_DRAWING_ON_PREVIOUS_KEY:
429  ui->radioButtonDrawOnPrev->setChecked(true);
430  break;
431  default:
432  break;
433  }
434 
435  // to secure that you have a relevant minimum setting for sound scrub
436  int fps = mManager->getInt(SETTING::FPS);
437  int minMsec = 1000 / fps;
438  if (minMsec > 100) { minMsec = 100; }
439  ui->soundScrubSpinBox->setMinimum(minMsec);
440  ui->soundScrubSlider->setMinimum(minMsec);
441 
442  ui->flipRollMsecsSlider->setValue(mManager->getInt(SETTING::FLIP_ROLL_MSEC));
443  ui->flipRollNumDrawingsSlider->setValue(mManager->getInt(SETTING::FLIP_ROLL_DRAWINGS));
444  ui->flipInBtwnMsecSlider->setValue(mManager->getInt(SETTING::FLIP_INBETWEEN_MSEC));
445  ui->flipRollMsecsSpinBox->setValue(mManager->getInt(SETTING::FLIP_ROLL_MSEC));
446  ui->flipRollNumDrawingsSpinBox->setValue(mManager->getInt(SETTING::FLIP_ROLL_DRAWINGS));
447  ui->flipInBtwnMsecSpinBox->setValue(mManager->getInt(SETTING::FLIP_INBETWEEN_MSEC));
448  ui->soundScrubSpinBox->setValue(mManager->getInt(SETTING::SOUND_SCRUB_MSEC));
449  ui->soundScrubSlider->setValue(mManager->getInt(SETTING::SOUND_SCRUB_MSEC));
450 
451  int convertedVisibilityThreshold = static_cast<int>(mManager->getFloat(SETTING::LAYER_VISIBILITY_THRESHOLD)*100);
452 
453  ui->visibilitySlider->setValue(convertedVisibilityThreshold);
454  ui->visibilitySpinbox->setValue(convertedVisibilityThreshold);
455 
456  int visibilityType = mManager->getInt(SETTING::LAYER_VISIBILITY);
457  ui->layerVisibilityComboBox->setCurrentIndex(visibilityType);
458  layerVisibilityChanged(visibilityType);
459 }

659 void FilesPage::removePreset()
660 {
661  if (ui->presetListWidget->count() <= 1) { return; }
662  if (ui->presetListWidget->selectedItems().empty()) { return; }
663 
664  // 1. Remove the items from list widget
665  QList<QListWidgetItem*> itemsToRemove = ui->presetListWidget->selectedItems();
666  for (QListWidgetItem* item : itemsToRemove)
667  {
668  ui->presetListWidget->removeItemWidget(item);
669  }
670 
671  // 2. Delete preset pclx files
672  for (QListWidgetItem* item : itemsToRemove)
673  {
674  int index = item->data(Qt::UserRole).toInt();
675  QFile presetFile(PresetDialog::getPresetPath(index));
676  presetFile.remove();
677  }
678 
679  // 3. Delete items from the ini settings
680  for (QListWidgetItem* item : itemsToRemove)
681  {
682  int index = item->data(Qt::UserRole).toInt();
683  mPresetSettings->remove(QString::number(index));
684  }
685 
686  // 4. check if the default preset has been deleted
687  int prevDefaultIndex = mManager->getInt(SETTING::DEFAULT_PRESET);
688  for (QListWidgetItem* item : itemsToRemove)
689  {
690  int index = item->data(Qt::UserRole).toInt();
691  if (index == prevDefaultIndex)
692  {
693  mManager->set(SETTING::DEFAULT_PRESET, 0);
694  }
695  }
696 
697  // 5. delete items
698  for (QListWidgetItem* item : itemsToRemove)
699  {
700  delete item;
701  }
702  updateValues();
703 }
