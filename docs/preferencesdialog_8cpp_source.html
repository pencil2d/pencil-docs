521 void ToolsPage::rotationIncrementChange(int value)
522 {
523  // Use log scale
524  int angle = qPow(value / 359.0, 2) * 359 + 1;
525  // Basically round up to the nearest number that is a divisor of 360
526  while (360 % angle != 0) {
527  angle++;
528  }
529  ui->rotationIncrementDisplay->setText(tr("%1 degree(s)", "", angle).arg(angle));
530  mManager->set(SETTING::ROTATION_INCREMENT, angle);
