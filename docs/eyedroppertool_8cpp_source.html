83void EyedropperTool::pointerMoveEvent(PointerEvent* event)
84{
85 Layer* layer = mEditor->layers()->currentLayer();
86 if (layer == nullptr) { return; }
87
88 if (layer->type() == Layer::BITMAP)
89 {
90 QColor pickedColor = getBitmapColor(static_cast<LayerBitmap*>(layer), event->canvasPos());
91 if (pickedColor.isValid())
92 {
93 mScribbleArea->setCursor(cursor(pickedColor));
94 }
95 else
96 {
97 mScribbleArea->setCursor(cursor());
98 }
99 }
100 if (layer->type() == Layer::VECTOR)
101 {
102 int pickedColor = getVectorColor(static_cast<LayerVector*>(layer), event->canvasPos());
103 if (pickedColor >= 0)
104 {
105 mScribbleArea->setCursor(cursor(mEditor->object()->getColor(pickedColor).color));
106 }
107 else
108 {
109 mScribbleArea->setCursor(cursor());
110 }
111 }
112}

125void EyedropperTool::updateFrontColor(const QPointF& pos)
126{
127 Layer* layer = mEditor->layers()->currentLayer();
128 if (layer == nullptr) { return; }
129
130 if (layer->type() == Layer::BITMAP)
131 {
132 QColor pickedColor = getBitmapColor(static_cast<LayerBitmap*>(layer), pos);
133 if (pickedColor.isValid())
134 {
135 mEditor->color()->setFrontColor(pickedColor);
136 }
137 }
138 else if (layer->type() == Layer::VECTOR)
139 {
140 int pickedColor = getVectorColor(static_cast<LayerVector*>(layer), pos);
141 if (pickedColor >= 0)
142 {
143 mEditor->color()->setColorNumber(pickedColor);
144 }
145 }
146}

148QColor EyedropperTool::getBitmapColor(LayerBitmap* layer, const QPointF& pos)
149{
150 BitmapImage* targetImage = layer->getLastBitmapImageAtFrame(mEditor->currentFrame(), 0);
151 if (targetImage == nullptr || !targetImage->contains(pos)) return QColor();
152
153 QColor pickedColour;
154 const QRgb pixelColor = targetImage->constScanLine(qFloor(pos.x()),
155 qFloor(pos.y()));
156 pickedColour.setRgba(pixelColor);
157
158 if (pickedColour.alpha() <= 0) pickedColour = QColor();
159 return pickedColour;
160}

162int EyedropperTool::getVectorColor(LayerVector* layer, const QPointF& pos)
163{
164 auto vectorImage = static_cast<VectorImage*>(layer->getLastKeyFrameAtPosition(mEditor->currentFrame()));
165 if (vectorImage == nullptr) return -1;
166
167 // Check curves
168 const qreal toleranceDistance = 10.0;
169 const QList<int> closestCurves = vectorImage->getCurvesCloseTo(pos, toleranceDistance);
170 const QList<int> visibleClosestCurves = filter(closestCurves, [vectorImage](int i) { return vectorImage->isCurveVisible(i); });
171
172 if (!visibleClosestCurves.isEmpty())
173 {
174 return vectorImage->getCurvesColor(visibleClosestCurves.last());
175 }
176
177 // Check fills
178 int colorNumber = vectorImage->getColorNumber(pos);
179 return colorNumber;
180}
