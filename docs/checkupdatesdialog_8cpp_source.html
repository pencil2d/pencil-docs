154 QString CheckUpdatesDialog::getVersionNumberFromXml(QString xml)
155 {
156  // XML source: http://github.com/pencil2d/pencil/releases.atom
157 
158  QXmlStreamReader xmlReader(xml);
159 
160  while (!xmlReader.atEnd() && !xmlReader.hasError())
161  {
162  QXmlStreamReader::TokenType tokenType = xmlReader.readNext();
163  if (tokenType == QXmlStreamReader::StartElement && xmlReader.name() == "entry")
164  {
165  while (!xmlReader.atEnd() && !xmlReader.hasError())
166  {
167  xmlReader.readNext();
168  if (xmlReader.name() == "title")
169  {
170  QString titleTag = xmlReader.readElementText();
171  return titleTag.remove(QRegExp("^v")); // remove the leading 'v'
172  }
173  }
174  }
175  }
176  if (xmlReader.error() != QXmlStreamReader::NoError)
177  {
178  qDebug() << xmlReader.errorString();
179  return "0.0.1";
180  }
181  return "0.0.1";
182 }
