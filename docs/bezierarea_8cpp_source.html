52 Status BezierArea::createDomElement( QXmlStreamWriter& xmlStream )
53 {
54  xmlStream.writeStartElement( "area" );
55  xmlStream.writeAttribute( "colourNumber", QString::number( mColourNumber ) );
56  xmlStream.writeAttribute("filled", QString::number( mIsFilled ) );
57 
58  int errorLocation = -1;
59  for ( int i = 0; i < mVertex.size(); i++ )
60  {
61  xmlStream.writeEmptyElement( "vertex" );
62  xmlStream.writeAttribute( "curve", QString::number( mVertex.at( i ).curveNumber ) );
63  xmlStream.writeAttribute( "vertex", QString::number( mVertex.at( i ).vertexNumber ) );
64 
65  if ( errorLocation < 0 && xmlStream.hasError() )
66  {
67  errorLocation = i;
68  }
69  }
70 
71  xmlStream.writeEndElement(); // Close area element
72 
73  if ( xmlStream.hasError() && errorLocation >= 0 )
74  {
75  QStringList debugInfo = QStringList() << "BezierArea::createDomElement"
76  << QString( "colourNumber = %1" ).arg( mColourNumber )
77  << QString( "- mVertex[%1] has failed to write" ).arg( errorLocation )
78  << QString( "&nbsp;&nbsp;curve = %1" ).arg( mVertex.at( errorLocation ).curveNumber )
79  << QString( "&nbsp;&nbsp;vertex = %1 " ).arg( mVertex.at( errorLocation ).vertexNumber );
80 
81  return Status( Status::FAIL, debugInfo );
82  }
83 
84  return Status::OK;
85 }
