36void OverlayPainter::paint(QPainter &painter)
37{
38 if (mCameraLayer == nullptr) { return; }
39
40 painter.save();
41 initializePainter(painter);
42
43 QTransform camTransform = mCameraLayer->getViewAtFrame(mOptions.nFrameIndex);
44 QRect cameraRect = mCameraLayer->getViewRect();
45 Camera* camera = mCameraLayer->getLastCameraAtFrame(mOptions.nFrameIndex, 0);
46
47 if (camera == nullptr) { return; }
48
49
50 painter.setWorldTransform(mViewTransform);
51 if (mOptions.bCenter)
52 {
53 paintOverlayCenter(painter, camTransform, cameraRect);
54 }
55 if (mOptions.bThirds)
56 {
57 paintOverlayThirds(painter, camTransform, cameraRect);
58 }
59 if (mOptions.bGoldenRatio)
60 {
61 paintOverlayGolden(painter, camTransform, cameraRect);
62 }
63 if (mOptions.bSafeArea)
64 {
65 paintOverlaySafeAreas(painter, *camera, camTransform, cameraRect);
66 }
67
68 if (mOptions.bPerspective1)
69 {
70 paintOverlayPerspectiveOnePoint(painter, camTransform, cameraRect);
71 }
72 if (mOptions.bPerspective2)
73 {
74 paintOverlayPerspectiveTwoPoints(painter, *camera, camTransform, cameraRect);
75 }
76 if (mOptions.bPerspective3)
77 {
78 paintOverlayPerspectiveThreePoints(painter, *camera, camTransform, cameraRect);
79 }
80
81 if (mOptions.bGrid)
82 {
83 paintGrid(painter);
84 }
85
86 painter.restore();
87}

89void OverlayPainter::paintOverlayCenter(QPainter &painter, const QTransform& camTransform, const QRect& camRect) const
90{
91 painter.save();
92 painter.setCompositionMode(QPainter::RasterOp_NotSourceAndNotDestination);
93
94 QPen pen(Qt::DashLine);
95 qreal space = 10;
96 QVector<qreal> dashes;
97 dashes << 10 << space << 10 << space << 10 << space;
98 pen.setDashPattern(dashes);
99 pen.setCosmetic(true);
100 painter.setPen(pen);
101 painter.setBrush(Qt::NoBrush);
102 painter.setRenderHint(QPainter::Antialiasing, false);
103
104 QPolygon poly = camTransform.inverted().mapToPolygon(camRect);
105 QPoint centerTop = QLineF(poly.at(0), poly.at(1)).pointAt(0.5).toPoint();
106 QPoint centerBottom = QLineF(poly.at(2), poly.at(3)).pointAt(0.5).toPoint();
107 QPoint centerLeft = QLineF(poly.at(0), poly.at(3)).pointAt(0.5).toPoint();
108 QPoint centerRight = QLineF(poly.at(1), poly.at(2)).pointAt(0.5).toPoint();
109 painter.drawLine(QLineF(centerTop, centerBottom).pointAt(0.4).toPoint(),
110 QLineF(centerTop, centerBottom).pointAt(0.6).toPoint());
111 painter.drawLine(QLineF(centerLeft, centerRight).pointAt(0.4).toPoint(),
112 QLineF(centerLeft, centerRight).pointAt(0.6).toPoint());
113
114 painter.restore();

173void OverlayPainter::paintOverlaySafeAreas(QPainter &painter, const Camera& camera, const QTransform& camTransform, const QRect& camRect) const
174{
175 painter.save();
176 painter.setCompositionMode(QPainter::RasterOp_NotSourceAndNotDestination);
177 QPen pen(Qt::DashLine);
178 qreal space = 10;
179 QVector<qreal> dashes;
180 dashes << 10 << space << 10 << space << 10 << space;
181 pen.setDashPattern(dashes);
182 pen.setCosmetic(true);
183 painter.setPen(pen);
184 painter.setBrush(Qt::NoBrush);
185 painter.setRenderHint(QPainter::TextAntialiasing, true);
186
187 QPolygon poly = camTransform.inverted().mapToPolygon(camRect);
188 QLineF topLeftCrossLine(poly.at(0), poly.at(2));
189 QLineF bottomLeftCrossLine(poly.at(3), poly.at(1));
190
191 if (mOptions.bActionSafe)
192 {
193 int action = mOptions.nActionSafe;
194 painter.drawLine(topLeftCrossLine.pointAt((action / 2.0) / 100.0).toPoint(),
195 bottomLeftCrossLine.pointAt((100 - (action / 2.0)) / 100).toPoint());
196 painter.drawLine(topLeftCrossLine.pointAt((action / 2.0) / 100.0).toPoint(),
197 bottomLeftCrossLine.pointAt((action / 2.0) / 100).toPoint());
198 painter.drawLine(topLeftCrossLine.pointAt((100 - (action / 2.0)) / 100.0).toPoint(),
199 bottomLeftCrossLine.pointAt((100 - (action / 2.0)) / 100).toPoint());
200 painter.drawLine(topLeftCrossLine.pointAt((100 - (action / 2.0)) / 100.0).toPoint(),
201 bottomLeftCrossLine.pointAt((action / 2.0) / 100).toPoint());
202
203 if (mOptions.bShowSafeAreaHelperText)
204 {
205 painter.save();
206 QPointF topLeft = topLeftCrossLine.pointAt((action / 2.0) / 100 ).toPoint();
207
208 QTransform trans = QTransform::fromTranslate(topLeft.x(), topLeft.y());
209 QTransform rot = QTransform().rotate(-camera.rotation());
210 QTransform scale = QTransform::fromScale(camera.scaling(), camera.scaling());
211
212 QTransform t = scale.inverted() * rot * trans;
213 painter.setTransform(t, true);
214 painter.drawText(QPoint(), QObject::tr("Safe Action area %1 %").arg(action));
215 painter.restore();
216 }
217 }
218 if (mOptions.bTitleSafe)
219 {
220 int title = mOptions.nTitleSafe;
221 painter.drawLine(topLeftCrossLine.pointAt((title / 2.0) / 100.0).toPoint(),
222 bottomLeftCrossLine.pointAt((100 - (title / 2.0)) / 100).toPoint());
223 painter.drawLine(topLeftCrossLine.pointAt((title / 2.0) / 100.0).toPoint(),
224 bottomLeftCrossLine.pointAt((title / 2.0) / 100).toPoint());
225 painter.drawLine(topLeftCrossLine.pointAt((100 - (title / 2.0)) / 100.0).toPoint(),
226 bottomLeftCrossLine.pointAt((100 - (title / 2.0)) / 100).toPoint());
227 painter.drawLine(topLeftCrossLine.pointAt((100 - (title / 2.0)) / 100.0).toPoint(),
228 bottomLeftCrossLine.pointAt((title / 2.0) / 100).toPoint());
229
230 if (mOptions.bShowSafeAreaHelperText)
231 {
232 QPointF bottomLeft = bottomLeftCrossLine.pointAt((title / 2.0) / 100);
233 painter.save();
234
235 QTransform trans = QTransform::fromTranslate(bottomLeft.x(), bottomLeft.y());
236 QTransform rot = QTransform().rotate(-camera.rotation());
237 QTransform scale = QTransform::fromScale(camera.scaling(), camera.scaling());
238
239 QTransform t = scale.inverted() * rot * trans;
240 painter.setTransform(t, true);
241 painter.drawText(QPoint(), QObject::tr("Safe Title area %1 %").arg(title));
242 painter.restore();
243 }
244 }

249{
250 painter.save();
251 painter.setRenderHint(QPainter::Antialiasing, true);
252
253 qreal degrees = static_cast<qreal>(mOptions.nOverlayAngle);
254 if (degrees == 7.0) { degrees = 7.5; }
255 int repeats = static_cast<int>(360 / degrees);
256 QLineF angleLine;
257
258 QPointF singlePoint = camTransform.inverted().map(mOptions.mSinglePerspPoint);
259 if (singlePoint == QPointF(0, 0))

262 singlePoint = QPointF(0.1, 0.1);

290void OverlayPainter::paintOverlayPerspectiveTwoPoints(QPainter& painter, const Camera& camera, const QTransform& camTransform, const QRect& camRect) const
291{
292 painter.save();
293 painter.setRenderHint(QPainter::Antialiasing, true);
294
295 qreal degrees = static_cast<qreal>(mOptions.nOverlayAngle);
296 if (degrees == 7.0) { degrees = 7.5; }
297 int repeats = static_cast<int>(180 / degrees);
298
299 QPointF leftPoint = camTransform.inverted().map(mOptions.mLeftPerspPoint);
300 QPointF rightPoint = camTransform.inverted().map(mOptions.mRightPerspPoint);
301 if (leftPoint == QPointF(0.0, 0.0))
302 {
303 // TODO: bug in Qt prevents points from being (0,0)...
304 leftPoint = QPointF(0.1, 0.1);
305 }
306
307 if (rightPoint == QPointF(0.0, 0.0))
308 {
309 // TODO: bug in Qt prevents points from being (0,0)...
310 rightPoint = QPointF(0.1, 0.1);
311 }
312
313 QLineF angleLineLeft;
314 QLineF angleLineRight;
315 angleLineLeft.setAngle(LEFTANGLEOFFSET);
316 angleLineLeft.setP1(leftPoint);
317 angleLineLeft.setLength(camRect.width() * LINELENGTHFACTOR);
318 angleLineRight.setAngle(RIGHTANGLEOFFSET);
319 angleLineRight.setP1(rightPoint);
320 angleLineRight.setLength(camRect.width() * LINELENGTHFACTOR);
321 QVector<QLineF> lines;
322 for (int i = 0; i <= repeats; i++)
323 {
324 angleLineLeft.setAngle((LEFTANGLEOFFSET - i * degrees) + camera.rotation());
325 angleLineRight.setAngle((RIGHTANGLEOFFSET - i * degrees) + camera.rotation());
326 lines.append(angleLineRight);
327 lines.append(angleLineLeft);
328 }
329
330 painter.drawLines(lines);
331
332 painter.setWorldMatrixEnabled(false);
333
334 leftPoint = mViewTransform.map(leftPoint);
335 rightPoint = mViewTransform.map(rightPoint);
336
337 if (mOptions.bShowHandle) {
338 painter.setCompositionMode(QPainter::CompositionMode_SourceOver);
339 painter.setPen(mPalette.color(QPalette::HighlightedText));
340 painter.setBrush(mPalette.color(QPalette::Highlight));
341 painter.drawEllipse(QRectF(leftPoint.x()-HANDLE_WIDTH*.5, leftPoint.y()-HANDLE_WIDTH*.5, HANDLE_WIDTH,HANDLE_WIDTH));
342 painter.drawEllipse(QRectF(rightPoint.x()-HANDLE_WIDTH*.5, rightPoint.y()-HANDLE_WIDTH*.5, HANDLE_WIDTH, HANDLE_WIDTH));
343 }
344
345 painter.restore();

395void OverlayPainter::paintGrid(QPainter& painter) const
396{
397 painter.save();
398 int gridSizeW = mOptions.nGridSizeW;
399 int gridSizeH = mOptions.nGridSizeH;
400
401 QRectF rect = painter.viewport();
402 QRectF boundingRect = mViewTransform.inverted().mapRect(rect);
403
404 int left = round100(boundingRect.left(), gridSizeW) - gridSizeW;
405 int right = round100(boundingRect.right(), gridSizeW) + gridSizeW;
406 int top = round100(boundingRect.top(), gridSizeH) - gridSizeH;
407 int bottom = round100(boundingRect.bottom(), gridSizeH) + gridSizeH;
408
409 QPen pen(Qt::lightGray);
410 pen.setCosmetic(true);
411 painter.setPen(pen);
412 painter.setBrush(Qt::NoBrush);
413 painter.setRenderHint(QPainter::Antialiasing, false);
414 // draw vertical grid lines
415 for (int x = left; x < right; x += gridSizeW)
416 {
417 painter.drawLine(x, top, x, bottom);
418 }
419
420 // draw horizontal grid lines
421 for (int y = top; y < bottom; y += gridSizeH)
422 {
423 painter.drawLine(left, y, right, y);
424 }
425
426 painter.restore();
427}
