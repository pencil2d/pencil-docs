46 void SelectTool::beginSelection()
47 {
48  // Store original click position for help with selection rectangle.
49  mAnchorOriginPoint = getLastPoint();
50 
51  mScribbleArea->calculateSelectionTransformation();
52 
53  // paint and apply the transformation
54  mScribbleArea->paintTransformedSelection();
55  mScribbleArea->applyTransformedSelection();
56 
57  if (mScribbleArea->isSomethingSelected()) // there is something selected
58  {
59  if (mCurrentLayer->type() == Layer::VECTOR)
60  {
61  static_cast<LayerVector*>(mCurrentLayer)->getLastVectorImageAtFrame(mEditor->currentFrame(), 0)->deselectAll();
62  }
63 
64  mScribbleArea->findMoveModeOfCornerInRange();
65  mAnchorOriginPoint = whichAnchorPoint();
66 
67  // the user did not click on one of the corners
68  if (mScribbleArea->getMoveMode() == MoveMode::NONE)
69  {
70  mScribbleArea->mySelection.setTopLeft(getLastPoint());
71  mScribbleArea->mySelection.setBottomRight(getLastPoint());
72  }
73  }
74  else
75  {
76  mScribbleArea->setSelection(QRectF(getCurrentPoint().x(),
77  getCurrentPoint().y(),1,1));
78  }
79  mScribbleArea->update();

121 void SelectTool::pointerReleaseEvent(PointerEvent* event)
122 {
123  mCurrentLayer = mEditor->layers()->currentLayer();
124  if (mCurrentLayer == nullptr) return;
125  if (event->button() != Qt::LeftButton) return;
126 
127  // if there's a small very small distance between current and last point
128  // discard the selection...
129  // TODO: improve by adding a timer to check if the user is deliberately selecting
130  if (QLineF(mAnchorOriginPoint, getCurrentPoint()).length() < 5.0)
131  {
132  mScribbleArea->deselectAll();
133  }
134  if (maybeDeselect())
135  {
136  mScribbleArea->deselectAll();
137  }
138  else
139  {
140  keepSelection();
141  }
142 
143  mScribbleArea->updateToolCursor();
144 
145  mScribbleArea->updateCurrentFrame();
146  mScribbleArea->setAllDirty();
147 }
