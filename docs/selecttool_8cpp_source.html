43QCursor SelectTool::cursor()
44{
45 // Don't update cursor while we're moving the selection
46 if (mScribbleArea->isPointerInUse()) { return QCursor(mCursorPixmap); }
47
48 MoveMode mode = mEditor->select()->getMoveMode();
49
50 mCursorPixmap.fill(QColor(255, 255, 255, 0));
51 QPainter cursorPainter(&mCursorPixmap);
52 cursorPainter.setRenderHint(QPainter::Antialiasing);
53
54 switch(mode)
55 {
56 case MoveMode::TOPLEFT:
57 case MoveMode::BOTTOMRIGHT:
58 {
59 cursorPainter.drawPixmap(QPoint(6,6),QPixmap("://icons/general/cursor-diagonal-left.svg"));
60 break;
61 }
62 case MoveMode::TOPRIGHT:
63 case MoveMode::BOTTOMLEFT:
64 {
65 cursorPainter.drawPixmap(QPoint(6,6),QPixmap("://icons/general/cursor-diagonal-right.svg"));
66 break;
67 }
68 case MoveMode::MIDDLE:
69 {
70 cursorPainter.drawPixmap(QPoint(6,6),QPixmap("://icons/general/cursor-move.svg"));
71 break;
72 }
73 case MoveMode::NONE:
74 cursorPainter.drawPixmap(QPoint(3,3), QPixmap(":icons/general/cross.png"));
75 break;
76 default:
77 Q_UNREACHABLE();
78 break;
79 }
80 return QCursor(mCursorPixmap);
81}

96void SelectTool::beginSelection(Layer* currentLayer, const QPointF& pos)
97{
98 auto selectMan = mEditor->select();
99
100 if (selectMan->somethingSelected() && mMoveMode != MoveMode::NONE) // there is something selected
101 {
102 if (currentLayer->type() == Layer::VECTOR)
103 {
104 VectorImage* vectorImage = static_cast<LayerVector*>(currentLayer)->getLastVectorImageAtFrame(mEditor->currentFrame(), 0);
105 if (vectorImage != nullptr) {
106 vectorImage->deselectAll();
107 }
108 }
109 mSelectionRect = mEditor->select()->mapToSelection(mEditor->select()->mySelectionRect()).boundingRect();
110 }
111 else
112 {
113 selectMan->setSelection(QRectF(pos.x(), pos.y(), 1, 1), mEditor->layers()->currentLayer()->type() == Layer::BITMAP);
114 mAnchorOriginPoint = pos;
115 }
116
117 mScribbleArea->updateFrame();
118}

120void SelectTool::pointerPressEvent(PointerEvent* event)
121{
122 Layer* currentLayer = mEditor->layers()->currentLayer();
123 if (currentLayer == nullptr) return;
124 if (!currentLayer->isPaintable()) { return; }
125 if (event->button() != Qt::LeftButton) { return; }
126 auto selectMan = mEditor->select();
127
128 mPressPoint = event->canvasPos();
129 selectMan->setMoveModeForAnchorInRange(mPressPoint);
130 mMoveMode = selectMan->getMoveMode();
131 mStartMoveMode = mMoveMode;
132
133 beginSelection(currentLayer, mPressPoint);
134}

136void SelectTool::pointerMoveEvent(PointerEvent* event)
137{
138 Layer* currentLayer = mEditor->layers()->currentLayer();
139 if (currentLayer == nullptr) { return; }
140 if (!currentLayer->isPaintable()) { return; }
141 auto selectMan = mEditor->select();
142
143 if (!selectMan->somethingSelected()) { return; }
144
145 selectMan->setMoveModeForAnchorInRange(event->canvasPos());
146 mMoveMode = selectMan->getMoveMode();
147 mScribbleArea->updateToolCursor();
148
149 if (mScribbleArea->isPointerInUse())
150 {
151 controlOffsetOrigin(event->canvasPos(), mAnchorOriginPoint);
152
153 if (currentLayer->type() == Layer::VECTOR)
154 {
155 VectorImage* vectorImage = static_cast<LayerVector*>(currentLayer)->getLastVectorImageAtFrame(mEditor->currentFrame(), 0);
156 if (vectorImage != nullptr) {
157 vectorImage->select(selectMan->mapToSelection(QPolygonF(selectMan->mySelectionRect())).boundingRect());
158 }
159 }
160 }
161
162 mScribbleArea->updateFrame();
163}

165void SelectTool::pointerReleaseEvent(PointerEvent* event)
166{
167 Layer* currentLayer = mEditor->layers()->currentLayer();
168 if (currentLayer == nullptr) return;
169 if (event->button() != Qt::LeftButton) return;
170
171 // if there's a small very small distance between current and last point
172 // discard the selection...
173 // TODO: improve by adding a timer to check if the user is deliberately selecting
174 if (QLineF(mAnchorOriginPoint, event->canvasPos()).length() < 5.0)
175 {
176 mEditor->deselectAll();
177 }
178 if (maybeDeselect(event->canvasPos()))
179 {
180 mEditor->deselectAll();
181 }
182 else
183 {
184 keepSelection(currentLayer);
185 }
186
187 mStartMoveMode = MoveMode::NONE;
188 mSelectionRect = mEditor->select()->mapToSelection(mEditor->select()->mySelectionRect()).boundingRect();
189
190 mScribbleArea->updateToolCursor();
191 mScribbleArea->updateFrame();
192}

214void SelectTool::controlOffsetOrigin(QPointF currentPoint, QPointF anchorPoint)
215{
216 // when the selection is none, manage the selection Origin
217 if (mStartMoveMode != MoveMode::NONE) {
218 QRectF rect = mSelectionRect;
219
220 QPointF offset = offsetFromPressPos(currentPoint);
221 if (mStartMoveMode == MoveMode::TOPLEFT) {
222 rect.adjust(offset.x(), offset.y(), 0, 0);
223 } else if (mStartMoveMode == MoveMode::TOPRIGHT) {
224 rect.adjust(0, offset.y(), offset.x(), 0);
225 } else if (mStartMoveMode == MoveMode::BOTTOMRIGHT) {
226 rect.adjust(0, 0, offset.x(), offset.y());
227 } else if (mStartMoveMode == MoveMode::BOTTOMLEFT) {
228 rect.adjust(offset.x(), 0, 0, offset.y());
229 } else {
230 rect.translate(offset.x(), offset.y());
231 }
232
233 rect = rect.normalized();
234 if (rect.isValid()) {
235 mEditor->select()->setSelection(rect, true);
236 }
237 } else {
238 manageSelectionOrigin(currentPoint, anchorPoint);
239 }
240}

246void SelectTool::manageSelectionOrigin(QPointF currentPoint, QPointF originPoint)
247{
248 qreal mouseX = currentPoint.x();
249 qreal mouseY = currentPoint.y();
250
251 QRectF selectRect = mSelectionRect;
252
253 if (mouseX <= originPoint.x())
254 {
255 selectRect.setLeft(mouseX);
256 selectRect.setRight(originPoint.x());
257 }
258 else
259 {
260 selectRect.setLeft(originPoint.x());
261 selectRect.setRight(mouseX);
262 }
263
264 if (mouseY <= originPoint.y())
265 {
266 selectRect.setTop(mouseY);
267 selectRect.setBottom(originPoint.y());
268 }
269 else
270 {
271 selectRect.setTop(originPoint.y());
272 selectRect.setBottom(mouseY);
273 }
274
275 if (selectRect.width() <= 0) {
276 selectRect.setWidth(1);
277 }
278 if (selectRect.height() <= 0) {
279 selectRect.setHeight(1);
280 }
281
282 editor()->select()->setSelection(selectRect);
283}
