94 void SelectTool::beginSelection()
95 {
96  auto selectMan = mEditor->select();
97 
98  if (selectMan->somethingSelected() && mMoveMode != MoveMode::NONE) // there is something selected
99  {
100  if (mCurrentLayer->type() == Layer::VECTOR)
101  {
102  VectorImage* vectorImage = static_cast<LayerVector*>(mCurrentLayer)->getLastVectorImageAtFrame(mEditor->currentFrame(), 0);

133 void SelectTool::pointerMoveEvent(PointerEvent*)
134 {
135  mCurrentLayer = mEditor->layers()->currentLayer();
136  if (mCurrentLayer == nullptr) { return; }
137  if (!mCurrentLayer->isPaintable()) { return; }
138  auto selectMan = mEditor->select();
139 
140  if (!selectMan->somethingSelected()) { return; }
141 
142  selectMan->setMoveModeForAnchorInRange(getCurrentPoint());
143  mMoveMode = selectMan->getMoveMode();
144  mScribbleArea->updateToolCursor();
145 
146  if (mScribbleArea->isPointerInUse())

148  controlOffsetOrigin(getCurrentPoint(), mAnchorOriginPoint);
149 
150  if (mCurrentLayer->type() == Layer::VECTOR)

152  VectorImage* vectorImage = static_cast<LayerVector*>(mCurrentLayer)->getLastVectorImageAtFrame(mEditor->currentFrame(), 0);

162 void SelectTool::pointerReleaseEvent(PointerEvent* event)
163 {
164  mCurrentLayer = mEditor->layers()->currentLayer();
165  if (mCurrentLayer == nullptr) return;
166  if (event->button() != Qt::LeftButton) return;
167 
168  // if there's a small very small distance between current and last point
169  // discard the selection...
170  // TODO: improve by adding a timer to check if the user is deliberately selecting
171  if (QLineF(mAnchorOriginPoint, getCurrentPoint()).length() < 5.0)
172  {
173  mEditor->deselectAll();
174  }
175  if (maybeDeselect())
176  {
177  mEditor->deselectAll();
178  }
179  else
180  {
181  keepSelection();
182  }
183 
184  mStartMoveMode = MoveMode::NONE;
185  mSelectionRect = mEditor->select()->mapToSelection(mEditor->select()->mySelectionRect()).boundingRect();
186 
187  mScribbleArea->updateToolCursor();
188  mScribbleArea->updateCurrentFrame();
189 }

211 void SelectTool::controlOffsetOrigin(QPointF currentPoint, QPointF anchorPoint)
212 {
213  QPointF offset = offsetFromPressPos();
214 
215  if (editor()->layers()->currentLayer()->type() == Layer::BITMAP) {
216  offset = QPointF(offset).toPoint();
217  }
218 
219  // when the selection is none, manage the selection Origin
220  if (mStartMoveMode != MoveMode::NONE) {
221  QRectF rect = mSelectionRect;
222 
223  QPointF offset = offsetFromPressPos();
224  if (mStartMoveMode == MoveMode::TOPLEFT) {
225  rect.adjust(offset.x(), offset.y(), 0, 0);
226  } else if (mStartMoveMode == MoveMode::TOPRIGHT) {
227  rect.adjust(0, offset.y(), offset.x(), 0);
228  } else if (mStartMoveMode == MoveMode::BOTTOMRIGHT) {
229  rect.adjust(0, 0, offset.x(), offset.y());
230  } else if (mStartMoveMode == MoveMode::BOTTOMLEFT) {
231  rect.adjust(offset.x(), 0, 0, offset.y());
232  } else {
233  rect.translate(offset.x(), offset.y());
234  }
235 
236  rect = rect.normalized();
237  if (rect.isValid()) {
238  mEditor->select()->setSelection(rect, true);
239  }
240  } else {
241  manageSelectionOrigin(currentPoint, anchorPoint);
242  }
243 }

249 void SelectTool::manageSelectionOrigin(QPointF currentPoint, QPointF originPoint)
250 {
251  qreal mouseX = currentPoint.x();
252  qreal mouseY = currentPoint.y();
253 
254  QRectF selectRect = mSelectionRect;
255 
256  if (mouseX <= originPoint.x())
257  {
258  selectRect.setLeft(mouseX);
259  selectRect.setRight(originPoint.x());
260  }
261  else
262  {
263  selectRect.setLeft(originPoint.x());
264  selectRect.setRight(mouseX);
265  }
266 
267  if (mouseY <= originPoint.y())
268  {
269  selectRect.setTop(mouseY);
270  selectRect.setBottom(originPoint.y());
271  }
272  else
273  {
274  selectRect.setTop(originPoint.y());
275  selectRect.setBottom(mouseY);
276  }
277 
278  if (selectRect.width() <= 0) {
279  selectRect.setWidth(1);
280  }
281  if (selectRect.height() <= 0) {
282  selectRect.setHeight(1);
283  }
284 
285  editor()->select()->setSelection(selectRect);
286 }
