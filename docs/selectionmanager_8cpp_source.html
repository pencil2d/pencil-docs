113 {
114  if (mSelectionPolygon.count() < 3) { return; }
115 
116  QPolygonF projectedPolygon = mapToSelection(mSelectionPolygon);
117 
118  const double calculatedSelectionTol = selectionTolerance();
119 
120  MoveMode mode;
121  if (QLineF(point, projectedPolygon[0]).length() < calculatedSelectionTol)
122  {

125  else if (QLineF(point, projectedPolygon[1]).length() < calculatedSelectionTol)
126  {
127  mode = MoveMode::TOPRIGHT;
128  }
129  else if (QLineF(point, projectedPolygon[2]).length() < calculatedSelectionTol)
130  {
131  mode = MoveMode::BOTTOMRIGHT;
132  }
133  else if (QLineF(point, projectedPolygon[3]).length() < calculatedSelectionTol)
134  {
135  mode = MoveMode::BOTTOMLEFT;
136  }
137  else if (projectedPolygon.containsPoint(point, Qt::WindingFill))
138  {

148 void SelectionManager::adjustSelection(const QPointF& currentPoint, const QPointF& offset, qreal rotationOffset, int rotationIncrement)
149 {
150  switch (mMoveMode)
151  {
152  case MoveMode::MIDDLE: {
153  translate(currentPoint - offset);
154  break;
155  }
156  case MoveMode::TOPLEFT:
157  case MoveMode::TOPRIGHT:
158  case MoveMode::BOTTOMRIGHT:
159  case MoveMode::BOTTOMLEFT: {
160 
161  QPolygonF projectedPolygon = mapToSelection(mSelectionPolygon);
162  QVector2D currentPVec = QVector2D(currentPoint);
163 
164  qreal originWidth = mSelectionPolygon[1].x() - mSelectionPolygon[0].x();
165  qreal originHeight = mSelectionPolygon[3].y() - mSelectionPolygon[0].y();
166 
167  QVector2D staticXAnchor;
168  QVector2D staticYAnchor;
169  QVector2D movingAnchor;
170  if (mMoveMode == MoveMode::TOPLEFT) {
171  movingAnchor = QVector2D(projectedPolygon[0]);
172  staticXAnchor = QVector2D(projectedPolygon[1]);
173  staticYAnchor = QVector2D(projectedPolygon[3]);
174  } else if (mMoveMode == MoveMode::TOPRIGHT) {
175  movingAnchor = QVector2D(projectedPolygon[1]);
176  staticXAnchor = QVector2D(projectedPolygon[0]);
177  staticYAnchor = QVector2D(projectedPolygon[2]);
178  } else if (mMoveMode == MoveMode::BOTTOMRIGHT) {
179  movingAnchor = QVector2D(projectedPolygon[2]);
180  staticXAnchor = QVector2D(projectedPolygon[3]);
181  staticYAnchor = QVector2D(projectedPolygon[1]);
182  } else {
183  movingAnchor = QVector2D(projectedPolygon[3]);
184  staticXAnchor = QVector2D(projectedPolygon[2]);
185  staticYAnchor = QVector2D(projectedPolygon[0]);
186  }
187 
188  QVector2D directionVecX = staticXAnchor - currentPVec;
189  QVector2D directionVecY = staticYAnchor - currentPVec;
190 
191  // Calculates the signed distance
192  qreal distanceX = QVector2D::dotProduct(directionVecX, (staticXAnchor - movingAnchor).normalized());
193  qreal distanceY = QVector2D::dotProduct(directionVecY, (staticYAnchor - movingAnchor).normalized());
194 
195  qreal scaleX = distanceX / originWidth;
196  qreal scaleY = distanceY / originHeight;
197  if (mAspectRatioFixed) {
198  scaleY = scaleX;
199  }
200 
201  scale(scaleX, scaleY);
202 
203  break;
204  }
205  case MoveMode::ROTATION: {
206  rotate(rotationOffset, rotationIncrement);
207  break;
208  }
209  default:
210  break;
211  }
212  calculateSelectionTransformation();
213 }

291 void SelectionManager::calculateSelectionTransformation()
292 {
293  QTransform t;
294  t.translate(-mAnchorPoint.x(), -mAnchorPoint.y());
295  QTransform t2;
296  t2.translate(mTranslation.x(), mTranslation.y());
297 
298  QTransform r;
299  r.rotate(mRotatedAngle);
300  QTransform s;
301  s.scale(mScaleX, mScaleY);
302  mSelectionTransform = t * s * r * t2;
303 }

305 QPointF SelectionManager::offsetFromAspectRatio(qreal offsetX, qreal offsetY) const
306 {
307  QPolygonF projectedPolygon = mapToSelection(mSelectionPolygon);
308  qreal width = QLineF(projectedPolygon[0], projectedPolygon[1]).dx();
309  qreal height = QLineF(projectedPolygon[0], projectedPolygon[3]).dy();
310  qreal factor = width / height;
311 
312  if (mMoveMode == MoveMode::TOPLEFT || mMoveMode == MoveMode::BOTTOMRIGHT)
313  {
314  offsetY = offsetX / factor;
315  }
316  else if (mMoveMode == MoveMode::TOPRIGHT || mMoveMode == MoveMode::BOTTOMLEFT)
317  {
318  offsetY = -(offsetX / factor);
319  }
320  else if (mMoveMode == MoveMode::MIDDLE)
321  {
322  qreal absX = offsetX;
323  if (absX < 0) { absX = -absX; }
324 
325  qreal absY = offsetY;
326  if (absY < 0) { absY = -absY; }
327 
328  if (absX > absY) { offsetY = 0; }
329  if (absY > absX) { offsetX = 0; }
330  }
331  return QPointF(offsetX, offsetY);
332 }
