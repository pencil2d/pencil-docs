108void LayerManager::setCurrentLayer(int layerIndex)
109{
110 Q_ASSERT(layerIndex >= 0);
111 Q_ASSERT(layerIndex < object()->getLayerCount());
112
113 // Deselect frames of previous layer.
114 Layer* previousLayer = currentLayer();
115 previousLayer->deselectAll();
116
117 emit currentLayerWillChange(layerIndex);
118
119 // Do not check if layer index has changed
120 // because the current layer may have changed either way
121 editor()->setCurrentLayerIndex(layerIndex);
122 emit currentLayerChanged(layerIndex);
123
124 if (object()->getLayer(layerIndex)->type() == Layer::CAMERA)
125 {
126 mLastCameraLayerIdx = layerIndex;
127 }
128}

155QString LayerManager::nameSuggestLayer(const QString& name)
156{
157 // if no layers: return name
158 if (count() == 0)
159 {
160 return name;
161 }
162 QVector<QString> sLayers;
163 // fill Vector with layer names
164 for (int i = 0; i < count(); i++)
165 {
166 sLayers.append(getLayer(i)->name());
167 }
168 // if name is not in list, return name
169 if (!sLayers.contains(name))
170 {
171 return name;
172 }
173 int newIndex = 2;
174 QString newName = name;
175 do {
176 newName = QStringLiteral("%1 %2")
177 .arg(name).arg(QString::number(newIndex++));
178 } while (sLayers.contains(newName));
179 return newName;
180}

182Layer* LayerManager::createLayer(Layer::LAYER_TYPE type, const QString& strLayerName)
183{
184 Layer* layer = nullptr;
185 switch (type) {
186 case Layer::BITMAP:
187 layer = object()->addNewBitmapLayer();
188 break;
189 case Layer::VECTOR:
190 layer = object()->addNewVectorLayer();
191 break;
192 case Layer::SOUND:
193 layer = object()->addNewSoundLayer();
194 break;
195 case Layer::CAMERA:
196 layer = object()->addNewCameraLayer();
197 break;
198 default:
199 Q_ASSERT(true);
200 return nullptr;
201 }
202
203 layer->setName(strLayerName);
204 emit layerCountChanged(count());
205 setCurrentLayer(getLastLayerIndex());
206
207 return layer;
208}

316Status LayerManager::deleteLayer(int index)
317{
318 Layer* layer = object()->getLayer(index);
319 if (layer->type() == Layer::CAMERA)
320 {
321 std::vector<LayerCamera*> camLayers = object()->getLayersByType<LayerCamera>();
322 if (camLayers.size() == 1)
323 return Status::ERROR_NEED_AT_LEAST_ONE_CAMERA_LAYER;
324 }
325 Q_ASSERT(object()->getLayerCount() >= 2);
326
327 // current layer is the last layer && we are deleting it
328 if (index == object()->getLayerCount() - 1 &&
329 index == currentLayerIndex())
330 {
331 setCurrentLayer(currentLayerIndex() - 1);
332 }
333 object()->deleteLayer(layer);
334 if (index >= currentLayerIndex())
335 {
336 // current layer has changed, so trigger updates
337 setCurrentLayer(currentLayerIndex());
338 }
339
340 emit layerDeleted(index);
341 emit layerCountChanged(count());
342
343 return Status::OK;
344}

364int LayerManager::animationLength(bool includeSounds)
365{
366 int maxFrame = -1;
367
368 Object* o = object();
369 for (int i = 0; i < o->getLayerCount(); i++)
370 {
371 if (o->getLayer(i)->type() == Layer::SOUND)
372 {
373 if (!includeSounds)
374 continue;
375
376 Layer* soundLayer = o->getLayer(i);
377 soundLayer->foreachKeyFrame([&maxFrame](KeyFrame* keyFrame)
378 {
379 int endPosition = keyFrame->pos() + (keyFrame->length() - 1);
380 if (endPosition > maxFrame)
381 {
382 maxFrame = endPosition;
383 }
384 });
385 }
386 else
387 {
388 int lastFramePos = o->getLayer(i)->getMaxKeyFramePosition();
389 if (lastFramePos > maxFrame)
390 {
391 maxFrame = lastFramePos;
392 }
393 }
394 }
395 return maxFrame;
396}
