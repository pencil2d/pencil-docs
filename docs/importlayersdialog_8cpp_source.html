80 void ImportLayersDialog::importLayers()
81 {
82  Object* object = mEditor->object();
83  int currentFrame = mEditor->currentFrame();
84  Q_ASSERT(ui->lwLayers->count() == mImportObject->getLayerCount());
85 
86  for (int i = 0; i < ui->lwLayers->count(); i++ )
87  {
88  QListWidgetItem* item = ui->lwLayers->item(i);
89  if (item->isSelected())
90  {
91  int layerId = item->data(Qt::UserRole).toInt();
92 
93  mImportLayer = mImportObject->takeLayer(layerId);
94  mImportLayer->setName(mEditor->layers()->nameSuggestLayer(item->text()));
95  loadKeyFrames(mImportLayer); // all keyframes of this layer must be in memory
96 
97  object->addLayer(mImportLayer);
98 
99  if (mImportLayer->type() == Layer::SOUND)
100  {
101  LayerSound* layerSound = static_cast<LayerSound*>(mImportLayer);
102  layerSound->foreachKeyFrame([this](KeyFrame* key)
103  {
104  SoundClip* clip = dynamic_cast<SoundClip*>(key);
105  Status st = mEditor->sound()->loadSound(clip, clip->fileName());
106  Q_ASSERT(st.ok());
107  });
108  }
109  }
110  }
111  mEditor->object()->modification();
112 
113  mImportObject.reset();
114  getLayers();
115  mEditor->scrubTo(currentFrame);
116 }

123 void ImportLayersDialog::getLayers()
124 {
125  QProgressDialog progress(tr("Opening document..."), tr("Abort"), 0, 100, this);
126 
127  // Don't show progress bar if running without a GUI (aka. when rendering from command line)
128  if (isVisible())
129  {
130  hideQuestionMark(progress);
131  progress.setWindowModality(Qt::WindowModal);
132  progress.show();
133  }
134 
135  FileManager fm;
136  connect(&fm, &FileManager::progressChanged, [&progress](int p)
137  {
138  progress.setValue(p);
139  QApplication::processEvents(QEventLoop::ExcludeUserInputEvents);
140  });
141  connect(&fm, &FileManager::progressRangeChanged, [&progress](int max)
142  {
143  progress.setRange(0, max + 3);
144  });
145  mImportObject.reset(fm.load(mFileName));
146 
147  ui->lwLayers->clear();
148  for (int i = 0; i < mImportObject->getLayerCount(); i++)
149  {
150  const QString layerName = mImportObject->getLayer(i)->name();
151  const int layerId = mImportObject->getLayer(i)->id();
152 
153  // Store the layer name as well as layer ID cuz two layers could have the same name
154  QListWidgetItem* item = new QListWidgetItem(layerName);
155  item->setData(Qt::UserRole, layerId);
156  ui->lwLayers->addItem(item);
157  }
158 }
