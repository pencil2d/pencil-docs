45void BucketTool::loadSettings()
46{
47 mPropertyEnabled[TOLERANCE] = true;
48 mPropertyEnabled[WIDTH] = true;
49 mPropertyEnabled[FILL_MODE] = true;
50
51 QSettings settings(PENCIL2D, PENCIL2D);
52
53 properties.width = settings.value("fillThickness", 4.0).toDouble();
54 properties.feather = 10;
55 properties.stabilizerLevel = StabilizationLevel::NONE;
56 properties.useAA = DISABLED;
57 properties.tolerance = settings.value(SETTING_BUCKET_TOLERANCE, 32.0).toDouble();
58 properties.toleranceEnabled = settings.value(SETTING_BUCKET_TOLERANCE_ON, false).toBool();
59
60 properties.bucketFillExpand = settings.value(SETTING_BUCKET_FILL_EXPAND, 2.0).toInt();
61 properties.bucketFillExpandEnabled = settings.value(SETTING_BUCKET_FILL_EXPAND_ON, true).toBool();
62 properties.bucketFillReferenceMode = settings.value(SETTING_BUCKET_FILL_REFERENCE_MODE, 0).toInt();
63 properties.fillMode = settings.value(SETTING_FILL_MODE, 0).toInt();
64}

167void BucketTool::pointerPressEvent(PointerEvent* event)
168{
169 mInterpolator.pointerPressEvent(event);
170 startStroke(event->inputType());
171
172 Layer* targetLayer = mEditor->layers()->currentLayer();
173
174 if (targetLayer->type() != Layer::BITMAP) { return; }
175
176 LayerCamera* layerCam = mEditor->layers()->getCameraLayerBelow(mEditor->currentLayerIndex());
177
178 mBitmapBucket = BitmapBucket(mEditor,
179 mEditor->color()->frontColor(),
180 layerCam ? layerCam->getViewAtFrame(mEditor->currentFrame()).inverted().mapRect(layerCam->getViewRect()) : QRect(),
181 getCurrentPoint(),
182 properties);
183
184 // Because we can change layer to on the fly, but we do not act reactively
185 // on it, it's necessary to invalidate layer cache on press event.
186 // Otherwise, the cache will be drawn until a move event has been initiated.
187 mScribbleArea->invalidatePainterCaches();
188}

190void BucketTool::pointerMoveEvent(PointerEvent* event)
191{
192 mInterpolator.pointerMoveEvent(event);
193 if (event->buttons() & Qt::LeftButton && event->inputType() == mCurrentInputType)
194 {
195 Layer* layer = mEditor->layers()->currentLayer();
196 if (layer->type() == Layer::VECTOR)
197 {
198 drawStroke();
199 }
200 else if (layer->type() == Layer::BITMAP)
201 {
202 paintBitmap();
203 mFilledOnMove = true;
204 }
205 }
206}

208void BucketTool::pointerReleaseEvent(PointerEvent* event)
209{
210 mInterpolator.pointerReleaseEvent(event);
211 if (event->inputType() != mCurrentInputType) return;
212
213 Layer* layer = editor()->layers()->currentLayer();
214 if (layer == nullptr) { return; }
215
216 if (event->button() == Qt::LeftButton)
217 {
218 // Backup of bitmap image is more complicated now and has therefore been moved to bitmap code
219 if (layer->type() == Layer::VECTOR) {
220 mEditor->backup(typeName());
221 paintVector(layer);
222 }
223 else if (layer->type() == Layer::BITMAP && !mFilledOnMove)
224 {
225 paintBitmap();
226 }
227 }
228 mFilledOnMove = false;
229
230 endStroke();
231}

233void BucketTool::paintBitmap()
234{
235 mBitmapBucket.paint(getCurrentPoint(), [this](BucketState progress, int layerIndex, int frameIndex)
236 {
237 if (progress == BucketState::WillFillTarget)
238 {
239 mEditor->backup(layerIndex, frameIndex, typeName());
240 }
241 else if (progress == BucketState::DidFillTarget)
242 {
243 mEditor->setModified(layerIndex, frameIndex);
244 }

248void BucketTool::paintVector(Layer* layer)
249{
250 mScribbleArea->clearDrawingBuffer();
251
252 VectorImage* vectorImage = static_cast<LayerVector*>(layer)->getLastVectorImageAtFrame(mEditor->currentFrame(), 0);
253 if (vectorImage == nullptr) { return; } // Can happen if the first frame is deleted while drawing
254
255 if (!vectorImage->isPathFilled())
256 {
257 vectorImage->fillSelectedPath(mEditor->color()->frontColorNumber());
258 }
259
260 vectorImage->applyWidthToSelection(properties.width);
261 vectorImage->applyColorToSelectedCurve(mEditor->color()->frontColorNumber());
262 vectorImage->applyColorToSelectedArea(mEditor->color()->frontColorNumber());
263
264 applyChanges();
265
266 mEditor->setModified(mEditor->layers()->currentLayerIndex(), mEditor->currentFrame());
267}

274void BucketTool::drawStroke()
275{
276 StrokeTool::drawStroke();
277
278 if (properties.stabilizerLevel != mInterpolator.getStabilizerLevel())
279 {
280 mInterpolator.setStabilizerLevel(properties.stabilizerLevel);
281 }
282
283 QList<QPointF> p = mInterpolator.interpolateStroke();
284
285 Layer* layer = mEditor->layers()->currentLayer();
286
287 if (layer->type() == Layer::VECTOR)
288 {
289 mCurrentWidth = 30;
290 QColor pathColor = qPremultiply(mEditor->color()->frontColor().rgba());
291
292 QPen pen(pathColor,
293 mCurrentWidth * mEditor->view()->scaling(),
294 Qt::NoPen,
295 Qt::RoundCap,
296 Qt::RoundJoin);
297
298 if (p.size() == 4)
299 {
300 QPainterPath path(p[0]);
301 path.cubicTo(p[1], p[2], p[3]);
302 mScribbleArea->drawPath(path, pen, Qt::NoBrush, QPainter::CompositionMode_Source);
303 }
304 }
