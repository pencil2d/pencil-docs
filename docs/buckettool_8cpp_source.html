186 void BucketTool::drawStroke()
187 {
188  StrokeTool::drawStroke();
189 
190  if (properties.stabilizerLevel != m_pStrokeManager->getStabilizerLevel()) {
191  m_pStrokeManager->setStabilizerLevel(properties.stabilizerLevel);
192  }
193 
194  QList<QPointF> p = m_pStrokeManager->interpolateStroke();
195 
196  Layer* layer = mEditor->layers()->currentLayer();
197 
198 
199  if( layer->type() == Layer::VECTOR ) {
200  mCurrentWidth = 30;
201  int rad = qRound( ( mCurrentWidth / 2 + 2 ) * mEditor->view()->scaling() );
202 
203  QColor pathColor = qPremultiply(mEditor->color()->frontColor().rgba());
204 // pathColor.setAlpha(255);
205 
206  QPen pen( pathColor,
207  mCurrentWidth * mEditor->view()->scaling(),
208  Qt::NoPen,
209  Qt::RoundCap,
210  Qt::RoundJoin );
211 
212  if( p.size() == 4 ) {
213  QPainterPath path( p[ 0 ] );
214  path.cubicTo( p[ 1 ],
215  p[ 2 ],
216  p[ 3 ] );
217  mScribbleArea->drawPath( path, pen, Qt::NoBrush, QPainter::CompositionMode_Source );
218  mScribbleArea->refreshVector( path.boundingRect().toRect(), rad );
219  }
220  }
221 }
