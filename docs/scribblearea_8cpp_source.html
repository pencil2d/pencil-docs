199 void ScribbleArea::invalidateOnionSkinsCacheAround(int frameNumber)
200 {
201  if (frameNumber < 0) { return; }
202 
203  bool isOnionAbsolute = mPrefs->getString(SETTING::ONION_TYPE) == "absolute";
204  Layer *layer = mEditor->layers()->currentLayer(0);
205 
206  // The current layer can be null if updateFrame is triggered when creating a new project
207  if (!layer) return;
208 
209  if (mPrefs->isOn(SETTING::PREV_ONION))
210  {
211  int onionFrameNumber = frameNumber;
212  if (isOnionAbsolute)
213  {
214  onionFrameNumber = layer->getPreviousFrameNumber(onionFrameNumber + 1, true);
215  }
216 
217  for(int i = 1; i <= mPrefs->getInt(SETTING::ONION_PREV_FRAMES_NUM); i++)
218  {
219  onionFrameNumber = layer->getPreviousFrameNumber(onionFrameNumber, isOnionAbsolute);
220  if (onionFrameNumber < 0) break;
221 
222  invalidateCacheForFrame(onionFrameNumber);
223  }
224  }
225 
226  if (mPrefs->isOn(SETTING::NEXT_ONION))
227  {
228  int onionFrameNumber = frameNumber;
229 
230  for(int i = 1; i <= mPrefs->getInt(SETTING::ONION_NEXT_FRAMES_NUM); i++)
231  {
232  onionFrameNumber = layer->getNextFrameNumber(onionFrameNumber, isOnionAbsolute);
233  if (onionFrameNumber < 0) break;
234 
235  invalidateCacheForFrame(onionFrameNumber);
236  }
237  }
238 }

354 {
355  // Don't handle this event on auto repeat
356  if (event->isAutoRepeat()) { return; }

361  if (mInstantTool) { return; } // prevents shortcuts calls while using instant tool

365  return; // has been handled by tool

370  {
371  setTemporaryTool(ERASER);
372  return;
373  }
374 
375  // ---- fixed normal keys ----
376 
377  auto selectMan = mEditor->select();
378  bool isSomethingSelected = selectMan->somethingSelected();
379  if (isSomethingSelected) {
380  keyEventForSelection(event);
381  } else {
382  keyEvent(event);

390  {
391  case Qt::Key_Right:
392  selectMan->translate(QPointF(1, 0));
393  paintTransformedSelection();

395  case Qt::Key_Left:
396  selectMan->translate(QPointF(-1, 0));
397  paintTransformedSelection();
398  break;
399  case Qt::Key_Up:
400  selectMan->translate(QPointF(0, -1));
401  paintTransformedSelection();
402  break;
403  case Qt::Key_Down:

405  paintTransformedSelection();
406  break;

410  break;

417  mEditor->deselectAll();

484 // mouse and tablet event handlers
485 void ScribbleArea::wheelEvent(QWheelEvent* event)
486 {
487  // Don't change view if tool is in use
488  if (isPointerInUse()) return;
489 
490  Layer* layer = mEditor->layers()->currentLayer();
491  if (layer->type() == Layer::CAMERA && !layer->visible())
492  {
493  showLayerNotVisibleWarning(); // FIXME: crash when using tablets
494  return;
495  }
496 
497  const QPoint pixels = event->pixelDelta();
498  const QPoint angle = event->angleDelta();
499  const QPointF offset = mEditor->view()->mapScreenToCanvas(event->posF());
500 
501  const qreal currentScale = mEditor->view()->scaling();
502  if (!pixels.isNull())
503  {
504  // XXX: This pixel-based zooming algorithm currently has some shortcomings compared to the angle-based one:
505  // Zooming in is faster than zooming out and scrolling twice with delta x yields different zoom than
506  // scrolling once with delta 2x. Someone with the ability to test this code might want to "upgrade" it.
507  const int delta = pixels.y();
508  const qreal newScale = currentScale * (1 + (delta * 0.01));
509  mEditor->view()->scaleWithOffset(newScale, offset);
510  }
511  else if (!angle.isNull())
512  {
513  const int delta = angle.y();
514  // 12 rotation steps at "standard" wheel resolution (120/step) result in 100x zoom
515  const qreal newScale = currentScale * std::pow(100, delta / (12.0 * 120));
516  mEditor->view()->scaleWithOffset(newScale, offset);
517  }
518  updateCanvasCursor();
519  event->accept();
520 }

522 void ScribbleArea::tabletEvent(QTabletEvent *e)
523 {
524  PointerEvent event(e);
525 
526  if (event.pointerType() == QTabletEvent::Eraser)
527  {
528  editor()->tools()->tabletSwitchToEraser();
529  }
530  else
531  {
532  editor()->tools()->tabletRestorePrevTool();
533  }
534 
535  if (event.eventType() == QTabletEvent::TabletPress)
536  {
537  event.accept();
538  mStrokeManager->pointerPressEvent(&event);
539  mStrokeManager->setTabletInUse(true);
540  if (mIsFirstClick)
541  {
542  mIsFirstClick = false;
543  mDoubleClickTimer->start();
544  pointerPressEvent(&event);
545  }
546  else
547  {
548  qreal distance = QLineF(currentTool()->getCurrentPressPoint(), currentTool()->getLastPressPoint()).length();
549 
550  if (mDoubleClickMillis <= DOUBLE_CLICK_THRESHOLD && distance < 5.0) {
551  currentTool()->pointerDoubleClickEvent(&event);
552  }
553  else
554  {
555  // in case we handled the event as double click but really should have handled it as single click.
556  pointerPressEvent(&event);
557  }
558  }
559  mTabletInUse = event.isAccepted();
560  }
561  else if (event.eventType() == QTabletEvent::TabletMove)
562  {
563  if (!(event.buttons() & (Qt::LeftButton | Qt::RightButton)) || mTabletInUse)
564  {
565  mStrokeManager->pointerMoveEvent(&event);
566  pointerMoveEvent(&event);
567  }
568  }
569  else if (event.eventType() == QTabletEvent::TabletRelease)
570  {
571  if (mTabletInUse)
572  {
573  mStrokeManager->pointerReleaseEvent(&event);
574  pointerReleaseEvent(&event);
575  mStrokeManager->setTabletInUse(false);
576  mTabletInUse = false;
577  }
578  }
579 
580 #if defined Q_OS_LINUX
581  // Generate mouse events on linux to work around bug where the
582  // widget will not receive mouseEnter/mouseLeave
583  // events and the cursor will not update correctly.
584  // See https://codereview.qt-project.org/c/qt/qtbase/+/255384
585  // Scribblearea should not do anything with the mouse event when mTabletInUse is true.
586  event.ignore();
587 #else
588  // Always accept so that mouse events are not generated (theoretically)
589  // Unfortunately Windows sometimes generates the events anyway
590  // As long as mTabletInUse is true, mouse events *should* be ignored even when
591  // the are generated
592  event.accept();
593 #endif
594 }

596 void ScribbleArea::pointerPressEvent(PointerEvent* event)
597 {
598  bool isCameraLayer = mEditor->layers()->currentLayer()->type() == Layer::CAMERA;
599  if ((currentTool()->type() != HAND || isCameraLayer) && (event->button() != Qt::RightButton) && (event->button() != Qt::MidButton || isCameraLayer))
600  {
601  Layer* layer = mEditor->layers()->currentLayer();
602  if (!layer->visible())
603  {
604  event->ignore();
605  // This needs to be async so that mTabletInUse is set to false before
606  // further events are created (modal dialogs do not currently block tablet events)
607  QTimer::singleShot(0, this, &ScribbleArea::showLayerNotVisibleWarning);
608  return;
609  }
610  }
611 
612  if (event->buttons() & (Qt::MidButton | Qt::RightButton))
613  {
614  setTemporaryTool(HAND);
615  getTool(HAND)->pointerPressEvent(event);
616  }
617 
618  const bool isPressed = event->buttons() & Qt::LeftButton;
619  if (isPressed && mQuickSizing)
620  {
621  //qDebug() << "Start Adjusting" << event->buttons();
622  if (currentTool()->startAdjusting(event->modifiers(), 1))
623  {
624  return;
625  }
626  }
627 
628  if (event->button() == Qt::LeftButton)
629  {
630  currentTool()->pointerPressEvent(event);
631  }
