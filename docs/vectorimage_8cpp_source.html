193 {
194  QDomNode atomTag = element.firstChild(); // an atom in a vector picture is a curve or an area
195  while (!atomTag.isNull())

198  if (!atomElement.isNull())

229 void VectorImage::addPoint(int curveNumber, int vertexNumber, qreal fraction)
230 {
231  mCurves[curveNumber].addPoint(vertexNumber, fraction);
232  // updates the bezierAreas
233  for (int j = 0; j < mArea.size(); j++)
234  {
235  // shift the references of all the points beyond the new point
236  for (int k = 0; k < mArea.at(j).mVertex.size(); k++)
237  {
238  if (mArea[j].getVertexRef(k).curveNumber == curveNumber)
239  {
240  if (mArea[j].getVertexRef(k).vertexNumber >= vertexNumber)
241  {
242  mArea[j].mVertex[k].vertexNumber++;
243  }
244  }
245  }
246  // insert the new point in the area if necessary
247  for (int k = 1; k < mArea.at(j).mVertex.size(); k++)
248  {
249  if (VertexRef(curveNumber, vertexNumber + 1) == mArea.at(j).mVertex.at(k)) // area[j].vertex[k] == VertexRef(curveNumber, vertexNumber+1)
250  {
251  if (VertexRef(curveNumber, vertexNumber - 1) == mArea.at(j).mVertex.at(k - 1))
252  {
253  mArea[j].mVertex.insert(k, VertexRef(curveNumber, vertexNumber));
254  }
255  }
256  if (VertexRef(curveNumber, vertexNumber - 1) == mArea.at(j).mVertex.at(k))
257  {
258  if (VertexRef(curveNumber, vertexNumber + 1) == mArea.at(j).mVertex.at(k - 1))
259  {
260  mArea[j].mVertex.insert(k, VertexRef(curveNumber, vertexNumber));
261  }
262  }
263  }
264  }
265  modification();

294 void VectorImage::insertCurve(int position, BezierCurve& newCurve, qreal factor, bool interacts)
295 {
296  if (newCurve.getVertexSize() < 1) // security - a new curve should have a least 2 vertices
297  return;
298 
299  // Does the curve interact with others or with itself?
300  if (interacts)
301  {
302  // tolerance for taking the intersection as an existing vertex on a curve
303  qreal tol = qMax(newCurve.getWidth() / factor, 3.0 / factor);
304  //qDebug() << "tolerance" << tol;
305 
306  checkCurveExtremity(newCurve, tol);
307  checkCurveIntersections(newCurve, tol);
308  }
309 
310 
311  // Append or insert the curve in the list
312  if (position < 0 || position > mCurves.size() - 1)
313  {
314  mCurves.append(newCurve);
315  }
316  else
317  {
318  // If it's an insert we have to shift the curve numbers in the areas
319  for (int i = 0; i < mArea.size(); i++)
320  {
321  for (int j = 0; j < mArea.at(i).mVertex.size(); j++)
322  {
323  if (mArea.at(i).mVertex[j].curveNumber >= position) {
324  mArea[i].mVertex[j].curveNumber++;
325  }
326  }
327  }
328  mCurves.insert(position, newCurve);
329  }
330  updateImageSize(newCurve);
331  modification();
332 }

350 void VectorImage::checkCurveExtremity(BezierCurve& newCurve, qreal tolerance)
351 {
352  // finds if the new curve is closed
353  QPointF P = newCurve.getVertex(-1);
354  QPointF Q = newCurve.getVertex(newCurve.getVertexSize() - 1);
355  if (BezierCurve::eLength(P - Q) < tolerance)
356  {
357  newCurve.setVertex(newCurve.getVertexSize() - 1, P);
358  }
359 
360  // finds if the first or last point of the new curve is close to other curves
361  for (int i = 0; i < mCurves.size(); i++) // for each other curve
362  {
363  for (int j = 0; j < mCurves.at(i).getVertexSize(); j++) // for each cubic section of the other curve
364  {
365  QPointF P = newCurve.getVertex(-1);
366  QPointF Q = newCurve.getVertex(newCurve.getVertexSize() - 1);
367  QPointF P1 = mCurves.at(i).getVertex(j - 1);
368  QPointF P2 = mCurves.at(i).getVertex(j);
369  qreal tol3 = 2.0*sqrt(0.25*((P1 - P2).x()*(P1 - P2).x() + (P1 - P2).y()*(P1 - P2).y()) + tolerance*tolerance);
370  qreal dist1 = BezierCurve::eLength(P - P1);
371  qreal dist2 = BezierCurve::eLength(P - P2);
372  if (dist1 <= 0.2*tolerance)
373  {
374  newCurve.setVertex(-1, P1); //qDebug() << "--b " << P1;
375  }
376  else
377  {
378  if (dist2 <= 0.2*tolerance)
379  {
380  newCurve.setVertex(-1, P2); //qDebug() << "--c " << P2;
381  }
382  else
383  {
384  if (dist1 + dist2 <= 3 * tol3) // preselection, to speed up
385  {
386  QPointF nearestPoint = P;
387  qreal t = -1.0;
388  qreal distance = BezierCurve::findDistance(mCurves[i], j, P, nearestPoint, t);
389  if (distance < tolerance)
390  {
391  newCurve.setOrigin(nearestPoint); //qDebug() << "--d " << nearestPoint;
392  addPoint(i, j, t);
393  }
394  }
395  }
396  }
397 
398  dist1 = BezierCurve::eLength(Q - P1);
399  dist2 = BezierCurve::eLength(Q - P2);
400  if (dist1 <= 0.2*tolerance)
401  {
402  newCurve.setLastVertex(P1); //qDebug() << "--e " << P1;
403  }
404  else
405  {
406  if (dist2 <= 0.2*tolerance)
407  {
408  newCurve.setLastVertex(P2); //qDebug() << "--f " << P2;
409  }
410  else
411  {
412  if (dist1 + dist2 <= 3 * tol3) // pre-selection, to speed up
413  {
414  QPointF nearestPoint = Q;
415  qreal t = -1.0;;
416  qreal distance = BezierCurve::findDistance(mCurves[i], j, Q, nearestPoint, t);
417  if (distance < tolerance)
418  {
419  newCurve.setLastVertex(nearestPoint); //qDebug() << "--g " << nearestPoint;
420  addPoint(i, j, t);
421  }
422  }
423  }
424  //qDebug() << "Modif last";
425  }
426  }
427  }
428  modification();
429 }
