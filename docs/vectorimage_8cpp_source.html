163 void VectorImage::loadDomElement(QDomElement element)
164 {
165  QDomNode atomTag = element.firstChild(); // an atom in a vector picture is a curve or an area
166  while (!atomTag.isNull())
167  {
168  QDomElement atomElement = atomTag.toElement();
169  if (!atomElement.isNull())
170  {
171  if (atomElement.tagName() == "curve")
172  {
173  BezierCurve newCurve;
174  newCurve.loadDomElement(atomElement);
175  mCurves.append(newCurve);

178  {
179  BezierArea newArea;
180  newArea.loadDomElement(atomElement);
181  addArea(newArea);
182  }
183  }
184  atomTag = atomTag.nextSibling();
185  }
186  clean();

200 void VectorImage::addPoint(int curveNumber, int vertexNumber, qreal fraction)
201 {
202  mCurves[curveNumber].addPoint(vertexNumber, fraction);
203  // updates the bezierAreas
204  for (int j = 0; j < mArea.size(); j++)
205  {
206  // shift the references of all the points beyond the new point
207  for (int k = 0; k < mArea.at(j).mVertex.size(); k++)
208  {
209  if (mArea[j].getVertexRef(k).curveNumber == curveNumber)
210  {
211  if (mArea[j].getVertexRef(k).vertexNumber >= vertexNumber)
212  {
213  mArea[j].mVertex[k].vertexNumber++;
214  }
215  }
216  }
217  // insert the new point in the area if necessary
218  for (int k = 1; k < mArea.at(j).mVertex.size(); k++)
219  {
220  if (VertexRef(curveNumber, vertexNumber + 1) == mArea.at(j).mVertex.at(k)) // area[j].vertex[k] == VertexRef(curveNumber, vertexNumber+1)
221  {
222  if (VertexRef(curveNumber, vertexNumber - 1) == mArea.at(j).mVertex.at(k - 1))
223  {
224  mArea[j].mVertex.insert(k, VertexRef(curveNumber, vertexNumber));
225  }
226  }
227  if (VertexRef(curveNumber, vertexNumber - 1) == mArea.at(j).mVertex.at(k))
228  {
229  if (VertexRef(curveNumber, vertexNumber + 1) == mArea.at(j).mVertex.at(k - 1))
230  {
231  mArea[j].mVertex.insert(k, VertexRef(curveNumber, vertexNumber));
232  }
233  }
234  }
235  }
236  modification();

265 void VectorImage::insertCurve(int position, BezierCurve& newCurve, qreal factor, bool interacts)
266 {
267  if (newCurve.getVertexSize() < 1) // security - a new curve should have a least 2 vertices
268  return;
269 
270  // Does the curve interact with others or with itself?
271  if (interacts)
272  {
273  // tolerance for taking the intersection as an existing vertex on a curve
274  qreal tol = qMax(newCurve.getWidth() / factor, 3.0 / factor);
275  //qDebug() << "tolerance" << tol;
276 
277  checkCurveExtremity(newCurve, tol);
278  checkCurveIntersections(newCurve, tol);
279  }
280 
281 
282  // Append or insert the curve in the list
283  if (position < 0 || position > mCurves.size() - 1)
284  {
285  mCurves.append(newCurve);
286  }
287  else
288  {
289  // If it's an insert we have to shift the curve numbers in the areas
290  for (int i = 0; i < mArea.size(); i++)
291  {
292  for (int j = 0; j < mArea.at(i).mVertex.size(); j++)
293  {
294  if (mArea.at(i).mVertex[j].curveNumber >= position) {
295  mArea[i].mVertex[j].curveNumber++;
296  }
297  }
298  }
299  mCurves.insert(position, newCurve);
300  }
301  updateImageSize(newCurve);
302  modification();
303 }

321 void VectorImage::checkCurveExtremity(BezierCurve& newCurve, qreal tolerance)
322 {
323  // finds if the new curve is closed
324  QPointF P = newCurve.getVertex(-1);
325  QPointF Q = newCurve.getVertex(newCurve.getVertexSize() - 1);
326  if (BezierCurve::eLength(P - Q) < tolerance)
327  {
328  newCurve.setVertex(newCurve.getVertexSize() - 1, P);
329  }
330 
331  // finds if the first or last point of the new curve is close to other curves
332  for (int i = 0; i < mCurves.size(); i++) // for each other curve
333  {
334  for (int j = 0; j < mCurves.at(i).getVertexSize(); j++) // for each cubic section of the other curve
335  {
336  QPointF P = newCurve.getVertex(-1);
337  QPointF Q = newCurve.getVertex(newCurve.getVertexSize() - 1);
338  QPointF P1 = mCurves.at(i).getVertex(j - 1);
339  QPointF P2 = mCurves.at(i).getVertex(j);
340  qreal tol3 = 2.0*sqrt(0.25*((P1 - P2).x()*(P1 - P2).x() + (P1 - P2).y()*(P1 - P2).y()) + tolerance*tolerance);
341  qreal dist1 = BezierCurve::eLength(P - P1);
342  qreal dist2 = BezierCurve::eLength(P - P2);
343  if (dist1 <= 0.2*tolerance)
344  {
345  newCurve.setVertex(-1, P1); //qDebug() << "--b " << P1;
346  }
347  else
348  {
349  if (dist2 <= 0.2*tolerance)
350  {
351  newCurve.setVertex(-1, P2); //qDebug() << "--c " << P2;
352  }
353  else
354  {
355  if (dist1 + dist2 <= 3 * tol3) // preselection, to speed up
356  {
357  QPointF nearestPoint = P;
358  qreal t = -1.0;
359  qreal distance = BezierCurve::findDistance(mCurves[i], j, P, nearestPoint, t);
360  if (distance < tolerance)
361  {
362  newCurve.setOrigin(nearestPoint); //qDebug() << "--d " << nearestPoint;
363  addPoint(i, j, t);
364  }
365  }
366  }
367  }
368 
369  dist1 = BezierCurve::eLength(Q - P1);
370  dist2 = BezierCurve::eLength(Q - P2);
371  if (dist1 <= 0.2*tolerance)
372  {
373  newCurve.setLastVertex(P1); //qDebug() << "--e " << P1;
374  }
375  else
376  {
377  if (dist2 <= 0.2*tolerance)
378  {
379  newCurve.setLastVertex(P2); //qDebug() << "--f " << P2;
380  }
381  else
382  {
383  if (dist1 + dist2 <= 3 * tol3) // pre-selection, to speed up
384  {
385  QPointF nearestPoint = Q;
386  qreal t = -1.0;;
387  qreal distance = BezierCurve::findDistance(mCurves[i], j, Q, nearestPoint, t);
388  if (distance < tolerance)
389  {
390  newCurve.setLastVertex(nearestPoint); //qDebug() << "--g " << nearestPoint;
391  addPoint(i, j, t);
392  }
393  }
394  }
395  //qDebug() << "Modif last";
396  }
397  }
398  }
399  modification();

938  }

940  }

970  removeCurveAt(curve);
971  curve--;
972  }
973  }
974  }
