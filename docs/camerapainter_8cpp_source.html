98 void CameraPainter::paintVisuals(QPainter& painter) const
99 {
100  LayerCamera* cameraLayerBelow = static_cast<LayerCamera*>(mObject->getLayerBelow(mCurrentLayerIndex, Layer::CAMERA));
101 
102  if (cameraLayerBelow == nullptr) { return; }
103 
104  const Layer* currentLayer = mObject->getLayer(mCurrentLayerIndex);
105 
106  if (mLayerVisibility == LayerVisibility::CURRENTONLY && currentLayer->type() != Layer::CAMERA) { return; }
107 
108  if (!mIsPlaying || mOnionSkinOptions.enabledWhilePlaying) {
109 
110  int startLayerI = 0;
111  int endLayerI = mObject->getLayerCount() - 1;
112  for (int i = startLayerI; i <= endLayerI; i++) {
113  Layer* layer = mObject->getLayer(i);
114  if (layer->type() != Layer::CAMERA) { continue; }
115 
116  LayerCamera* cameraLayer = static_cast<LayerCamera*>(layer);
117 
118  bool isCurrentLayer = cameraLayer == cameraLayerBelow;
119 
120  painter.save();
121  painter.setOpacity(1);
122  if (mLayerVisibility == LayerVisibility::RELATED && !isCurrentLayer) {
123  painter.setOpacity(calculateRelativeOpacityForLayer(mCurrentLayerIndex, i, mRelativeLayerOpacityThreshold));
124  }
125 
126  paintOnionSkinning(painter, cameraLayer);
127 
128  painter.restore();
129  }
130  }
131 
132  if (!cameraLayerBelow->visible()) { return; }
133 
134  QTransform camTransform = cameraLayerBelow->getViewAtFrame(mFrameIndex);
135  QRect cameraRect = cameraLayerBelow->getViewRect();
136  paintBorder(painter, camTransform, cameraRect);
137 }

139 void CameraPainter::paintBorder(QPainter& painter, const QTransform& camTransform, const QRect& camRect) const
140 {
141  painter.save();
142  QRect viewRect = painter.viewport();
143 
144  painter.setOpacity(1.0);
145  painter.setWorldMatrixEnabled(true);
146  painter.setPen(Qt::NoPen);
147  painter.setBrush(QColor(0, 0, 0, 80));
148  painter.setCompositionMode(QPainter::CompositionMode_SourceOver);
149 
150  QTransform viewInverse = mViewTransform.inverted();
151  QRect boundingRect = viewInverse.mapRect(viewRect);
152 
153  QPolygon camPoly = camTransform.inverted().map(QPolygon(camRect));
154  QPolygon boundingRectPoly = boundingRect;
155  QPolygon visibleCanvasPoly = boundingRectPoly.subtracted(camPoly);
156 
157  painter.drawPolygon(visibleCanvasPoly);
158 
159  painter.restore();
160 }

162 void CameraPainter::paintOnionSkinning(QPainter& painter, const LayerCamera* cameraLayer) const
163 {
164  QPen onionSkinPen;
165 
166  painter.save();
167  painter.setBrush(Qt::NoBrush);
168 
169  onionSkinPen.setStyle(Qt::PenStyle::DashLine);
170  mOnionSkinPainter.paint(painter, cameraLayer, mOnionSkinOptions, mFrameIndex, [&] (OnionSkinPaintState state, int onionSkinNumber) {
171 
172  const QTransform& cameraTransform = cameraLayer->getViewAtFrame(onionSkinNumber);
173  const QPolygonF& cameraPolygon = Transform::mapToWorldPolygon(cameraTransform, mViewTransform, cameraLayer->getViewRect());
174  if (state == OnionSkinPaintState::PREV) {
175 
176  if (mOnionSkinOptions.colorizePrevFrames) {
177  onionSkinPen.setColor(Qt::red);
178  }
179 
180  painter.setPen(onionSkinPen);
181  painter.drawPolygon(cameraPolygon);
182  } else if (state == OnionSkinPaintState::NEXT) {
183  if (mOnionSkinOptions.colorizeNextFrames) {
184  onionSkinPen.setColor(Qt::blue);
185  }
186 
187  painter.setPen(onionSkinPen);
188  painter.drawPolygon(cameraPolygon);
189  } else if (state == OnionSkinPaintState::CURRENT) {
190  painter.save();
191  painter.setPen(Qt::black);
192  painter.drawPolygon(cameraPolygon);
193  painter.restore();
194  }
195  });
196  painter.restore();
197 }
