129void Editor::settingUpdated(SETTING setting)
130{
131 switch (setting)
132 {
133 case SETTING::AUTO_SAVE:
134 mIsAutosave = mPreferenceManager->isOn(SETTING::AUTO_SAVE);
135 break;
136 case SETTING::AUTO_SAVE_NUMBER:
137 mAutosaveNumber = mPreferenceManager->getInt(SETTING::AUTO_SAVE_NUMBER);
138 break;
139 case SETTING::ONION_TYPE:
140 mScribbleArea->onOnionSkinTypeChanged();
141 emit updateTimeLineCached();
142 break;
143 case SETTING::FRAME_POOL_SIZE:
144 mObject->setActiveFramePoolSize(mPreferenceManager->getInt(SETTING::FRAME_POOL_SIZE));
145 break;
146 case SETTING::LAYER_VISIBILITY:
147 mScribbleArea->setLayerVisibility(static_cast<LayerVisibility>(mPreferenceManager->getInt(SETTING::LAYER_VISIBILITY)));
148 emit updateTimeLine();
149 break;
150 default:
151 break;
152 }
153}

155void Editor::onCurrentLayerWillChange(int index)
156{
157 Layer* newLayer = layers()->getLayer(index);
158 Layer* currentLayer = layers()->currentLayer();
159 Q_ASSERT(newLayer && currentLayer);
160 if (currentLayer->type() != newLayer->type()) {
161 // We apply transform changes upon leaving a layer and deselect all
162 mScribbleArea->applyTransformedSelection();
163
164 if (currentLayer->type() == Layer::VECTOR) {
165 auto keyFrame = static_cast<VectorImage*>(currentLayer->getLastKeyFrameAtPosition(mFrame));
166 if (keyFrame)
167 {
168 keyFrame->deselectAll();
169 }
170 }
171
172 select()->resetSelectionProperties();
173 }

194void Editor::copy()
195{
196 Layer* currentLayer = layers()->currentLayer();
197
198 Q_ASSERT(currentLayer != nullptr);
199
200 if (!canCopy()) { return; }
201
202 backup(tr("Copy"));
203
204 if (currentLayer->hasAnySelectedFrames() && !select()->somethingSelected()) {
205 clipboards()->copySelectedFrames(currentLayer);
206 } else if (currentLayer->type() == Layer::BITMAP) {
207 BitmapImage* bitmapImage = static_cast<BitmapImage*>(currentLayer->getLastKeyFrameAtPosition(currentFrame()));
208 clipboards()->copyBitmapImage(bitmapImage, select()->mySelectionRect());
209 } else if (currentLayer->type() == Layer::VECTOR) {
210 VectorImage* vectorImage = static_cast<VectorImage*>(currentLayer->getLastKeyFrameAtPosition(currentFrame()));
211 clipboards()->copyVectorImage(vectorImage);
212 }
213}

236void Editor::pasteFromPreviousFrame()
237{
238 Layer* currentLayer = layers()->currentLayer();
239 int prevFrame = currentLayer->getPreviousKeyFramePosition(mFrame);
240 if (!currentLayer->keyExists(mFrame) || prevFrame == mFrame)
241 {
242 return;
243 }
244
245 if (currentLayer->type() == Layer::BITMAP)
246 {
247 backup(tr("Paste from Previous Keyframe"));
248 BitmapImage* bitmapImage = static_cast<BitmapImage*>(currentLayer->getKeyFrameAt(prevFrame));
249 if (select()->somethingSelected())
250 {
251 BitmapImage copy = bitmapImage->copy(select()->mySelectionRect().toRect());
252 pasteToCanvas(&copy, mFrame);
253 }
254 else
255 {
256 pasteToCanvas(bitmapImage, mFrame);
257 }
258 }
259 else if (currentLayer->type() == Layer::VECTOR)
260 {
261 backup(tr("Paste from Previous Keyframe"));
262 VectorImage* vectorImage = static_cast<VectorImage*>(currentLayer->getKeyFrameAt(prevFrame));
263 pasteToCanvas(vectorImage, mFrame);
264 }
265}

267void Editor::pasteToCanvas(BitmapImage* bitmapImage, int frameNumber)
268{
269 Layer* currentLayer = layers()->currentLayer();
270
271 Q_ASSERT(currentLayer->type() == Layer::BITMAP);
272
273 if (select()->somethingSelected())
274 {
275 QRectF selection = select()->mySelectionRect();
276 if (bitmapImage->width() <= selection.width() && bitmapImage->height() <= selection.height())
277 {
278 bitmapImage->moveTopLeft(selection.topLeft());
279 }
280 else
281 {
282 bitmapImage->transform(selection, true);
283 }
284 }
285 mScribbleArea->handleDrawingOnEmptyFrame();
286 BitmapImage *canvasImage = static_cast<BitmapImage*>(currentLayer->getLastKeyFrameAtPosition(frameNumber));
287
288 // Paste clipboard onto current shown image
289 canvasImage->paste(bitmapImage);
290
291 // TODO: currently we don't support placing an image without already pasting it on an already existing
292 // image, this should be reworked such that a hovering selection could be shown, before applying it...
293 select()->setSelection(bitmapImage->bounds());
294 emit frameModified(frameNumber);
295}

311void Editor::pasteToFrames()
312{
313 auto clipboardFrames = clipboards()->getClipboardFrames();
314 Q_ASSERT(!clipboardFrames.empty());
315 Layer* currentLayer = layers()->currentLayer();
316
317 currentLayer->deselectAll();
318
319 int newPositionOffset = mFrame - clipboardFrames.cbegin()->first;
320 for (auto it = clipboardFrames.cbegin(); it != clipboardFrames.cend(); ++it)
321 {
322 int newPosition = it->first + newPositionOffset;
323
324 KeyFrame* keyFrameNewPos = currentLayer->getKeyFrameWhichCovers(newPosition);
325
326 if (keyFrameNewPos != nullptr) {
327
328 // Select and move any frames that may come into contact with the new position
329 currentLayer->newSelectionOfConnectedFrames(newPosition);
330 currentLayer->moveSelectedFrames(1);
331 }
332
333 // It's a bug if the keyframe is nullptr at this point...
334 Q_ASSERT(it->second != nullptr);
335
336 // TODO: undo/redo implementation
337 KeyFrame* keyClone = it->second->clone();
338 currentLayer->addKeyFrame(newPosition, keyClone);
339 if (currentLayer->type() == Layer::SOUND)
340 {
341 auto soundClip = static_cast<SoundClip*>(keyClone);
342 sound()->loadSound(soundClip, soundClip->fileName());
343 }
344
345 currentLayer->setFrameSelected(keyClone->pos(), true);
346 }
347}

349void Editor::paste()
350{
351 Layer* currentLayer = layers()->currentLayer();
352
353 Q_ASSERT(currentLayer != nullptr);
354
355 if (!canPaste()) { return; }
356
357 if (clipboards()->getClipboardFrames().empty()) {
358
359 backup(tr("Paste"));
360
361 clipboards()->setFromSystemClipboard(mScribbleArea->getCentralPoint(), currentLayer);
362
363 BitmapImage clipboardImage = clipboards()->getBitmapClipboard();
364 VectorImage clipboardVectorImage = clipboards()->getVectorClipboard();
365 if (currentLayer->type() == Layer::BITMAP && clipboardImage.isLoaded()) {
366 pasteToCanvas(&clipboardImage, mFrame);
367 } else if (currentLayer->type() == Layer::VECTOR && !clipboardVectorImage.isEmpty()) {
368 pasteToCanvas(&clipboardVectorImage, mFrame);
369 }
370 } else {
371 // TODO: implement undo/redo
372 pasteToFrames();
373 }
374
375 emit frameModified(mFrame);
376}

469Status Editor::openObject(const QString& strFilePath, const std::function<void(int)>& progressChanged, const std::function<void(int)>& progressRangeChanged)
470{
471 // Check for potential issues with the file
472 Q_ASSERT(!strFilePath.isEmpty());
473 QFileInfo fileInfo(strFilePath);
474 DebugDetails dd;
475 dd << QString("Raw file path: %1").arg(strFilePath);
476 dd << QString("Resolved file path: %1").arg(fileInfo.absoluteFilePath());
477 if (fileInfo.isDir())
478 {
479 return Status(Status::ERROR_FILE_CANNOT_OPEN,
480 dd,
481 tr("Could not open file"),
482 tr("The file you have selected is a directory, so we are unable to open it. "
483 "If you are are trying to open a project that uses the old structure, "
484 "please open the file ending with .pcl, not the data folder."));
485 }
486 if (!fileInfo.exists())
487 {
488 return Status(Status::FILE_NOT_FOUND,
489 dd,
490 tr("Could not open file"),
491 tr("The file you have selected does not exist, so we are unable to open it. "
492 "Please make sure that you've entered the correct path and that the file is accessible and try again."));
493 }
494 if (!fileInfo.isReadable())
495 {
496 dd << QString("Permissions: 0x%1").arg(fileInfo.permissions(), 0, 16);
497 return Status(Status::ERROR_FILE_CANNOT_OPEN,
498 dd,
499 tr("Could not open file"),
500 tr("This program does not have permission to read the file you have selected. "
501 "Please check that you have read permissions for this file and try again."));
502 }
503
504 int progress = 0;
505 FileManager fm(this);
506 connect(&fm, &FileManager::progressChanged, [&progress, &progressChanged](int p)
507 {
508 progressChanged(progress = p);
509 });
510 connect(&fm, &FileManager::progressRangeChanged, [&progressRangeChanged](int max)
511 {
512 progressRangeChanged(max + 3);
513 });
514
515 QString fullPath = fileInfo.absoluteFilePath();
516
517 Object* object = fm.load(fullPath);
518
519 Status fmStatus = fm.error();
520 if (!fmStatus.ok())
521 {
522 dd.collect(fmStatus.details());
523 fmStatus.setDetails(dd);
524 return fmStatus;
525 }
526
527 if (object == nullptr)
528 {
529 return Status(Status::ERROR_FILE_CANNOT_OPEN,
530 dd,
531 tr("Could not open file"),
532 tr("An unknown error occurred while trying to load the file and we are not able to load your file."));
533 }
534
535 setObject(object);
536
537 progressChanged(progress + 1);
538
539 layers()->notifyAnimationLengthChanged();
540 setFps(playback()->fps());
541
542 return Status::OK;
543}

545Status Editor::setObject(Object* newObject)
546{
547 Q_ASSERT(newObject);
548
549 if (newObject == mObject.get())
550 {
551 return Status::SAFE;
552 }

562 }
563 emit objectLoaded();
564
565 return Status::OK;

584Status Editor::importBitmapImage(const QString& filePath)
585{
586 QImageReader reader(filePath);
587
588 Q_ASSERT(layers()->currentLayer()->type() == Layer::BITMAP);
589 const auto layer = static_cast<LayerBitmap*>(layers()->currentLayer());
590
591 if (!layer->visible())
592 {
593 mScribbleArea->showLayerNotVisibleWarning();
594 return Status::SAFE;
595 }
596
597 Status status = Status::OK;
598 DebugDetails dd;
599 dd << QString("Raw file path: %1").arg(filePath);
600
601 QImage img(reader.size(), QImage::Format_ARGB32_Premultiplied);
602 if (!reader.read(&img)) {
603 QString format = reader.format();
604 if (!format.isEmpty())
605 {
606 dd << QString("QImageReader format: %1").arg(format);
607 }
608 dd << QString("QImageReader ImageReaderError type: %1").arg(reader.errorString());
609
610 QString errorDesc;
611 switch(reader.error())
612 {
613 case QImageReader::ImageReaderError::FileNotFoundError:
614 errorDesc = tr("File not found at path \"%1\". Please check the image is present at the specified location and try again.").arg(filePath);
615 break;
616 case QImageReader::UnsupportedFormatError:
617 errorDesc = tr("Image format is not supported. Please convert the image file to one of the following formats and try again:\n%1")
618 .arg(QString::fromUtf8(reader.supportedImageFormats().join(", ")));
619 break;
620 default:
621 errorDesc = tr("An error has occurred while reading the image. Please check that the file is a valid image and try again.");
622 }
623
624 status = Status(Status::FAIL, dd, tr("Import failed"), errorDesc);
625 }
626
627 const QPoint pos(view()->getImportView().dx() - (img.width() / 2),
628 view()->getImportView().dy() - (img.height() / 2));
629
630 if (!layer->keyExists(mFrame))
631 {
632 const bool ok = addNewKey();
633 Q_ASSERT(ok);
634 }
635 BitmapImage* bitmapImage = layer->getBitmapImageAtFrame(mFrame);
636 BitmapImage importedBitmapImage(pos, img);
637 bitmapImage->paste(&importedBitmapImage);
638 emit frameModified(bitmapImage->pos());
639
640 scrubTo(mFrame+1);
641
642 backup(tr("Import Image"));
643
644 return status;
645}

647Status Editor::importVectorImage(const QString& filePath)
648{
649 Q_ASSERT(layers()->currentLayer()->type() == Layer::VECTOR);
650
651 auto layer = static_cast<LayerVector*>(layers()->currentLayer());
652
653 Status status = Status::OK;
654 DebugDetails dd;
655 dd << QString("Raw file path: %1").arg(filePath);
656
657 VectorImage* vectorImage = layer->getVectorImageAtFrame(currentFrame());
658 if (vectorImage == nullptr)
659 {
660 addNewKey();
661 vectorImage = layer->getVectorImageAtFrame(currentFrame());
662 }
663
664 VectorImage importedVectorImage;
665 bool ok = importedVectorImage.read(filePath);
666 if (ok)
667 {
668 importedVectorImage.selectAll();
669 vectorImage->paste(importedVectorImage);
670 emit frameModified(importedVectorImage.pos());
671
672 backup(tr("Import Image"));
673 }
674 else {
675 status = Status(Status::FAIL, dd, tr("Import failed"), tr("You cannot import images into a vector layer."));
676 }
677
678 return status;

681Status Editor::importImage(const QString& filePath)
682{
683 Layer* layer = layers()->currentLayer();
684
685 DebugDetails dd;
686 dd << QString("Raw file path: %1").arg(filePath);
687
688 if (view()->getImportFollowsCamera())
689 {
690 LayerCamera* camera = static_cast<LayerCamera*>(layers()->getLastCameraLayer());
691 Q_ASSERT(camera);
692 QTransform transform = camera->getViewAtFrame(currentFrame());
693 view()->setImportView(transform);
694 }
695 switch (layer->type())
696 {
697 case Layer::BITMAP:
698 return importBitmapImage(filePath);
699
700 case Layer::VECTOR:
701 return importVectorImage(filePath);
702
703 default:
704 dd << QString("Current layer: %1").arg(layer->type());
705 return Status(Status::ERROR_INVALID_LAYER_TYPE, dd, tr("Import failed"), tr("You can only import images to a bitmap layer."));
706 }
707}

709Status Editor::importAnimatedImage(const QString& filePath, int frameSpacing, const std::function<void(int)>& progressChanged, const std::function<bool()>& wasCanceled)
710{
711 frameSpacing = qMax(1, frameSpacing);
712
713 DebugDetails dd;
714 dd << QString("Raw file path: %1").arg(filePath);
715
716 Layer* layer = layers()->currentLayer();
717 if (layer->type() != Layer::BITMAP)
718 {
719 dd << QString("Current layer: %1").arg(layer->type());
720 return Status(Status::ERROR_INVALID_LAYER_TYPE, dd, tr("Import failed"), tr("You can only import images to a bitmap layer."));
721 }
722 LayerBitmap* bitmapLayer = static_cast<LayerBitmap*>(layers()->currentLayer());
723
724 QImageReader reader(filePath);
725 dd << QString("QImageReader format: %1").arg(QString(reader.format()));
726 if (!reader.supportsAnimation()) {
727 return Status(Status::ERROR_INVALID_LAYER_TYPE, dd, tr("Import failed"), tr("The selected image has a format that does not support animation."));
728 }
729
730 QImage img(reader.size(), QImage::Format_ARGB32_Premultiplied);
731 const QPoint pos(view()->getImportView().dx() - (img.width() / 2),
732 view()->getImportView().dy() - (img.height() / 2));
733 int totalFrames = reader.imageCount();
734 while (reader.read(&img))
735 {
736 if (reader.error())
737 {
738 dd << QString("QImageReader ImageReaderError type: %1").arg(reader.errorString());
739
740 QString errorDesc;
741 switch(reader.error())
742 {
743 case QImageReader::ImageReaderError::FileNotFoundError:
744 errorDesc = tr("File not found at path \"%1\". Please check the image is present at the specified location and try again.").arg(filePath);
745 break;
746 case QImageReader::UnsupportedFormatError:
747 errorDesc = tr("Image format is not supported. Please convert the image file to one of the following formats and try again:\n%1")
748 .arg((QString)reader.supportedImageFormats().join(", "));
749 break;
750 default:
751 errorDesc = tr("An error has occurred while reading the image. Please check that the file is a valid image and try again.");
752 }
753
754 return Status(Status::FAIL, dd, tr("Import failed"), errorDesc);
755 }
756
757 if (!bitmapLayer->keyExists(mFrame))
758 {
759 addNewKey();
760 }
761 BitmapImage* bitmapImage = bitmapLayer->getBitmapImageAtFrame(mFrame);
762 BitmapImage importedBitmapImage(pos, img);
763 bitmapImage->paste(&importedBitmapImage);
764 emit frameModified(bitmapImage->pos());
765
766 if (wasCanceled())
767 {
768 break;
769 }
770
771 scrubTo(mFrame + frameSpacing);
772
773 backup(tr("Import Image"));
774
775 progressChanged(qFloor(qMin(static_cast<double>(reader.currentImageNumber()) / totalFrames, 1.0) * 100));
776 }
777
778 return Status::OK;
779}

781void Editor::selectAll() const
782{
783 Layer* layer = layers()->currentLayer();
784
785 QRectF rect;
786 if (layer->type() == Layer::BITMAP)
787 {
788 // Selects the drawn area (bigger or smaller than the screen). It may be more accurate to select all this way
789 // as the drawing area is not limited
790 BitmapImage *bitmapImage = static_cast<BitmapImage*>(layer->getLastKeyFrameAtPosition(mFrame));
791 if (bitmapImage == nullptr) { return; }
792
793 rect = bitmapImage->bounds();
794 }

888KeyFrame* Editor::addKeyFrame(const int layerNumber, int frameIndex)
889{
890 Layer* layer = mObject->getLayer(layerNumber);
891 Q_ASSERT(layer);
892
893 if (!layer->visible())
894 {
895 mScribbleArea->showLayerNotVisibleWarning();
896 return nullptr;
897 }
898
899 // Find next available space for a keyframe (where either no key exists or there is an empty sound key)
900 while (layer->keyExists(frameIndex))
901 {
902 if (layer->type() == Layer::SOUND
903 && layer->getKeyFrameAt(frameIndex)->fileName().isEmpty()
904 && layer->removeKeyFrame(frameIndex))
905 {
906 break;
907 }
908 else
909 {
910 frameIndex += 1;
911 }
912 }
913
914 const bool ok = layer->addNewKeyFrameAt(frameIndex);
915 Q_ASSERT(ok); // We already ensured that there is no keyframe at frameIndex, so this should always succeed
916 scrubTo(frameIndex); // currentFrameChanged() emit inside.
917 emit frameModified(frameIndex);
918 layers()->notifyAnimationLengthChanged();
919 return layer->getKeyFrameAt(frameIndex);
920}

923{

925 Q_ASSERT(layer != nullptr);
926
927 if (!layer->visible())
928 {
929 mScribbleArea->showLayerNotVisibleWarning();
930 return;
931 }
932
933 if (!layer->keyExistsWhichCovers(currentFrame()))
934 {
935 scrubBackward();

939 backup(tr("Remove frame"));
940

945}
