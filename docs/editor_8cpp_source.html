230 void Editor::backup(int backupLayer, int backupFrame, QString undoText)
231 {
232  while (mBackupList.size() - 1 > mBackupIndex && mBackupList.size() > 0)
233  {
234  delete mBackupList.takeLast();
235  }
236  while (mBackupList.size() > 19) // we authorize only 20 levels of cancellation
237  {
238  delete mBackupList.takeFirst();
239  mBackupIndex--;
240  }
241 
242  Layer* layer = mObject->getLayer(backupLayer);
243  if (layer != nullptr)
244  {
245  if (layer->type() == Layer::BITMAP)
246  {
247  BitmapImage* bitmapImage = static_cast<LayerBitmap*>(layer)->getLastBitmapImageAtFrame(backupFrame, 0);
248  if (currentFrame() == 1) {
249  int previous = layer->getPreviousKeyFramePosition(backupFrame);
250  bitmapImage = static_cast<LayerBitmap*>(layer)->getBitmapImageAtFrame(previous);
251  }
252  if (bitmapImage != nullptr)
253  {
254  BackupBitmapElement* element = new BackupBitmapElement(bitmapImage);
255  element->layer = backupLayer;
256  element->frame = bitmapImage->pos();
257  element->undoText = undoText;
258  element->somethingSelected = select()->somethingSelected();
259  element->mySelection = select()->mySelectionRect();
260  element->myTransformedSelection = select()->myTransformedSelectionRect();
261  element->myTempTransformedSelection = select()->myTempTransformedSelectionRect();
262  element->rotationAngle = select()->myRotation();
263  mBackupList.append(element);
264  mBackupIndex++;
265  }
266  }
267  else if (layer->type() == Layer::VECTOR)
268  {
269  VectorImage* vectorImage = static_cast<LayerVector*>(layer)->getLastVectorImageAtFrame(backupFrame, 0);
270  if (vectorImage != nullptr)
271  {
272  BackupVectorElement* element = new BackupVectorElement(vectorImage);
273  element->layer = backupLayer;
274  element->frame = vectorImage->pos();
275  element->undoText = undoText;
276  element->somethingSelected = select()->somethingSelected();
277  element->mySelection = select()->mySelectionRect();
278  element->myTransformedSelection = select()->myTransformedSelectionRect();
279  element->myTempTransformedSelection = select()->myTempTransformedSelectionRect();
280  element->rotationAngle = select()->myRotation();
281  mBackupList.append(element);
282  mBackupIndex++;
283  }
284  }
285  else if (layer->type() == Layer::SOUND)
286  {
287  int previous = layer->getPreviousKeyFramePosition(backupFrame);
288  KeyFrame* key = layer->getLastKeyFrameAtPosition(backupFrame);
289 
290  // in case tracks overlap, get previous frame
291  if (key == nullptr)
292  {
293  KeyFrame* previousKey = layer->getKeyFrameAt(previous);
294  key = previousKey;
295  }
296  if (key != nullptr) {
297  SoundClip* clip = static_cast<SoundClip*>(key);
298  if (clip)
299  {
300  BackupSoundElement* element = new BackupSoundElement(clip);
301  element->layer = backupLayer;
302  element->frame = backupFrame;
303  element->undoText = undoText;
304  element->fileName = clip->fileName();
305  element->originalName = clip->soundClipName();
306  mBackupList.append(element);
307  mBackupIndex++;
308  }
309  }
310  }
311  }
312 
313  updateAutoSaveCounter();
314 
315  emit updateBackup();
316 }

318 void Editor::sanitizeBackupElementsAfterLayerDeletion(int layerIndex)
319 {
320  for (int i = 0; i < mBackupList.size(); i++)
321  {
322  BackupElement *backupElement = mBackupList[i];
323  BackupBitmapElement *bitmapElement;
324  BackupVectorElement *vectorElement;
325  BackupSoundElement *soundElement;
326  switch (backupElement->type())
327  {
328  case BackupElement::BITMAP_MODIF:
329  bitmapElement = qobject_cast<BackupBitmapElement*>(backupElement);
330  Q_ASSERT(bitmapElement);
331  if (bitmapElement->layer > layerIndex)
332  {
333  bitmapElement->layer--;
334  continue;
335  }
336  else if (bitmapElement->layer != layerIndex)
337  {
338  continue;
339  }
340  break;
341  case BackupElement::VECTOR_MODIF:
342  vectorElement = qobject_cast<BackupVectorElement*>(backupElement);
343  Q_ASSERT(vectorElement);
344  if (vectorElement->layer > layerIndex)
345  {
346  vectorElement->layer--;
347  continue;
348  }
349  else if (vectorElement->layer != layerIndex)
350  {
351  continue;
352  }
353  break;
354  case BackupElement::SOUND_MODIF:
355  soundElement = qobject_cast<BackupSoundElement*>(backupElement);
356  Q_ASSERT(soundElement);
357  if (soundElement->layer > layerIndex)
358  {
359  soundElement->layer--;
360  continue;
361  }
362  else if (soundElement->layer != layerIndex)
363  {
364  continue;
365  }
366  break;
367  default:
368  Q_UNREACHABLE();
369  }
370  if (i <= mBackupIndex) {
371  mBackupIndex--;
372  }
373  mBackupList.removeAt(i);
374  i--;
375  }
376 }

378 void Editor::restoreKey()
379 {
380  BackupElement* lastBackupElement = mBackupList[mBackupIndex];
381 
382  Layer* layer = nullptr;
383  int frame = 0;
384  int layerIndex = 0;
385  if (lastBackupElement->type() == BackupElement::BITMAP_MODIF)
386  {
387  BackupBitmapElement* lastBackupBitmapElement = static_cast<BackupBitmapElement*>(lastBackupElement);
388  layerIndex = lastBackupBitmapElement->layer;
389  frame = lastBackupBitmapElement->frame;
390  layer = object()->getLayer(layerIndex);
391  addKeyFrame(layerIndex, frame);
392  dynamic_cast<LayerBitmap*>(layer)->getBitmapImageAtFrame(frame)->paste(&lastBackupBitmapElement->bitmapImage);
393  }
394  if (lastBackupElement->type() == BackupElement::VECTOR_MODIF)
395  {
396  BackupVectorElement* lastBackupVectorElement = static_cast<BackupVectorElement*>(lastBackupElement);
397  layerIndex = lastBackupVectorElement->layer;
398  frame = lastBackupVectorElement->frame;
399  layer = object()->getLayer(layerIndex);
400  addKeyFrame(layerIndex, frame);
401  dynamic_cast<LayerVector*>(layer)->getVectorImageAtFrame(frame)->paste(lastBackupVectorElement->vectorImage);
402  }
403  if (lastBackupElement->type() == BackupElement::SOUND_MODIF)
404  {
405  QString strSoundFile;
406  BackupSoundElement* lastBackupSoundElement = static_cast<BackupSoundElement*>(lastBackupElement);
407  layerIndex = lastBackupSoundElement->layer;
408  frame = lastBackupSoundElement->frame;
409 
410  strSoundFile = lastBackupSoundElement->fileName;
411  if (strSoundFile.isEmpty()) return;
412  KeyFrame* key = addKeyFrame(layerIndex, frame);
413  SoundClip* clip = dynamic_cast<SoundClip*>(key);
414  if (clip)
415  {
416  Status st = sound()->loadSound(clip, lastBackupSoundElement->fileName);
417  clip->setSoundClipName(lastBackupSoundElement->originalName);
418  if (!st.ok())
419  {
420  removeKey();
421  emit layers()->currentLayerChanged(layers()->currentLayerIndex()); // trigger timeline repaint.
422  }
423  }
424  }

427 void Editor::undo()
428 {
429  if (mBackupList.size() > 0 && mBackupIndex > -1)
430  {
431  if (mBackupIndex == mBackupList.size() - 1)
432  {
433  BackupElement* lastBackupElement = mBackupList[mBackupIndex];
434  if (lastBackupElement->type() == BackupElement::BITMAP_MODIF)
435  {
436  BackupBitmapElement* lastBackupBitmapElement = static_cast<BackupBitmapElement*>(lastBackupElement);
437  backup(lastBackupBitmapElement->layer, lastBackupBitmapElement->frame, "NoOp");
438  mBackupIndex--;
439  }
440  if (lastBackupElement->type() == BackupElement::VECTOR_MODIF)
441  {
442  BackupVectorElement* lastBackupVectorElement = static_cast<BackupVectorElement*>(lastBackupElement);
443  backup(lastBackupVectorElement->layer, lastBackupVectorElement->frame, "NoOp");
444  mBackupIndex--;
445  }
446  if (lastBackupElement->type() == BackupElement::SOUND_MODIF)
447  {
448  BackupSoundElement* lastBackupSoundElement = static_cast<BackupSoundElement*>(lastBackupElement);
449  backup(lastBackupSoundElement->layer, lastBackupSoundElement->frame, "NoOp");
450  mBackupIndex--;
451  }
452  }
453 
454  qDebug() << "Undo" << mBackupIndex;
455  mBackupList[mBackupIndex]->restore(this);
456  mBackupIndex--;
457  mScribbleArea->cancelTransformedSelection();
458 
459  Layer* layer = layers()->currentLayer();
460  if (layer == nullptr) { return; }
461 
462  select()->resetSelectionTransform();
463  if (layer->type() == Layer::VECTOR) {
464  VectorImage *vectorImage = static_cast<LayerVector*>(layer)->getVectorImageAtFrame(mFrame);
465  vectorImage->calculateSelectionRect();
466  select()->setSelection(vectorImage->getSelectionRect(), false);
467  }
468  emit updateBackup();
469  }

519 void Editor::copy()
520 {
521  Layer* layer = mObject->getLayer(layers()->currentLayerIndex());
522  if (layer == nullptr)
523  {
524  return;
525  }
526 
527  if (layer->type() == Layer::BITMAP)
528  {
529  LayerBitmap* layerBitmap = static_cast<LayerBitmap*>(layer);
530  BitmapImage* bitmapImage = layerBitmap->getLastBitmapImageAtFrame(currentFrame(), 0);
531  if (bitmapImage == nullptr) { return; }

533  {
534  g_clipboardBitmapImage = bitmapImage->copy(select()->mySelectionRect().toRect()); // copy part of the image
535  }

539  }
540  clipboardBitmapOk = true;
541  if (g_clipboardBitmapImage.image() != nullptr)
542  QApplication::clipboard()->setImage(*g_clipboardBitmapImage.image());

546  clipboardVectorOk = true;
547  VectorImage *vectorImage = static_cast<LayerVector*>(layer)->getLastVectorImageAtFrame(currentFrame(), 0);
548  if (vectorImage == nullptr) { return; }

553 void Editor::paste()
554 {
555  Layer* layer = mObject->getLayer(layers()->currentLayerIndex());
556  if (layer != nullptr)
557  {
558  if (layer->type() == Layer::BITMAP && g_clipboardBitmapImage.image() != nullptr)
559  {
560  backup(tr("Paste"));
561 
562  BitmapImage tobePasted = g_clipboardBitmapImage.copy();
563  qDebug() << "to be pasted --->" << tobePasted.image()->size();
564  if (select()->somethingSelected())
565  {
566  QRectF selection = select()->mySelectionRect();
567  if (g_clipboardBitmapImage.width() <= selection.width() && g_clipboardBitmapImage.height() <= selection.height())
568  {
569  tobePasted.moveTopLeft(selection.topLeft());
570  }
571  else
572  {
573  tobePasted.transform(selection, true);
574  }
575  }
576  auto pLayerBitmap = static_cast<LayerBitmap*>(layer);
577  mScribbleArea->handleDrawingOnEmptyFrame();
578  BitmapImage *bitmapImage = pLayerBitmap->getLastBitmapImageAtFrame(currentFrame(), 0);
579  Q_CHECK_PTR(bitmapImage);
580  bitmapImage->paste(&tobePasted); // paste the clipboard
581  }
582  else if (layer->type() == Layer::VECTOR && clipboardVectorOk)
583  {
584  backup(tr("Paste"));
585  deselectAll();
586  mScribbleArea->handleDrawingOnEmptyFrame();
587  VectorImage* vectorImage = static_cast<LayerVector*>(layer)->getLastVectorImageAtFrame(currentFrame(), 0);
588  Q_CHECK_PTR(vectorImage);
589  vectorImage->paste(g_clipboardVectorImage); // paste the clipboard
590  select()->setSelection(vectorImage->getSelectionRect(), false);
591  }
592  }
593  emit frameModified(mFrame);
594 }

725 bool Editor::importBitmapImage(QString filePath, int space)
726 {
727  QImageReader reader(filePath);
728 
729  Q_ASSERT(layers()->currentLayer()->type() == Layer::BITMAP);
730  auto layer = static_cast<LayerBitmap*>(layers()->currentLayer());
731 
732  QImage img(reader.size(), QImage::Format_ARGB32_Premultiplied);
733  if (img.isNull())
734  {
735  return false;
736  }
737 
738  const QPoint pos = QPoint(static_cast<int>(view()->getImportView().dx()),
739  static_cast<int>(view()->getImportView().dy())) - QPoint(img.width() / 2, img.height() / 2);
740  while (reader.read(&img))
741  {
742  if (!layer->keyExists(currentFrame()))
743  {
744  addNewKey();
745  }
