169 int FlowLayout::doLayout(const QRect &rect, bool testOnly) const
170 {
171  int left, top, right, bottom;
172  getContentsMargins(&left, &top, &right, &bottom);
173  QRect effectiveRect = rect.adjusted(+left, +top, -right, -bottom);
174  int x = effectiveRect.x();
175  int y = effectiveRect.y();
176  int lineHeight = 0;
178 
180  QLayoutItem *item;
181  foreach (item, itemList) {
182  QWidget *wid = item->widget();
183  int spaceX = horizontalSpacing();
184  if (spaceX == -1)
185  spaceX = wid->style()->layoutSpacing(
186  QSizePolicy::PushButton, QSizePolicy::PushButton, Qt::Horizontal);
187  int spaceY = verticalSpacing();
188  if (spaceY == -1)
189  spaceY = wid->style()->layoutSpacing(
190  QSizePolicy::PushButton, QSizePolicy::PushButton, Qt::Vertical);
193  int nextX = x + item->sizeHint().width() + spaceX;
194  if (nextX - spaceX > effectiveRect.right() && lineHeight > 0) {
195  x = effectiveRect.x();
196  y = y + lineHeight + spaceY;
197  nextX = x + item->sizeHint().width() + spaceX;
198  lineHeight = 0;
199  }
200 
201  if (!testOnly)
202  item->setGeometry(QRect(QPoint(x, y), item->sizeHint()));
203 
204  x = nextX;
205  lineHeight = qMax(lineHeight, item->sizeHint().height());
206  }
207  return y + lineHeight - rect.y() + bottom;
208 }
