124void GeneralPage::updateValues()
125{
126 int index = ui->languageCombo->findData(mManager->getString(SETTING::LANGUAGE));
127
128 if (index >= 0)
129 {
130 QSignalBlocker b(ui->languageCombo);
131 ui->languageCombo->setCurrentIndex(index);
132 }
133
134 QSignalBlocker b1(ui->curveSmoothingLevel);
135 ui->curveSmoothingLevel->setValue(mManager->getInt(SETTING::CURVE_SMOOTHING));
136 QSignalBlocker b2(ui->windowOpacityLevel);
137 ui->windowOpacityLevel->setValue(100 - mManager->getInt(SETTING::WINDOW_OPACITY));
138 QSignalBlocker b3(ui->shadowsBox);
139 ui->shadowsBox->setChecked(mManager->isOn(SETTING::SHADOW));
140 QSignalBlocker b4(ui->toolCursorsBox);
141 ui->toolCursorsBox->setChecked(mManager->isOn(SETTING::TOOL_CURSOR));
142 QSignalBlocker b5(ui->antialiasingBox);
143 ui->antialiasingBox->setChecked(mManager->isOn(SETTING::ANTIALIAS));
144 QSignalBlocker b6(ui->canvasCursorBox);
145 ui->canvasCursorBox->setChecked(mManager->isOn(SETTING::CANVAS_CURSOR));
146 QSignalBlocker b7(ui->gridSizeInputW);
147 ui->gridSizeInputW->setValue(mManager->getInt(SETTING::GRID_SIZE_W));
148 QSignalBlocker b11(ui->gridSizeInputH);
149 ui->gridSizeInputH->setValue(mManager->getInt(SETTING::GRID_SIZE_H));
150 QSignalBlocker b8(ui->gridCheckBox);
151 ui->gridCheckBox->setChecked(mManager->isOn(SETTING::GRID));
152 QSignalBlocker b16(ui->actionSafeCheckBox);
153
154 bool actionSafeOn = mManager->isOn(SETTING::ACTION_SAFE_ON);
155 ui->actionSafeCheckBox->setChecked(actionSafeOn);
156 QSignalBlocker b14(ui->actionSafeInput);
157 ui->actionSafeInput->setValue(mManager->getInt(SETTING::ACTION_SAFE));
158 QSignalBlocker b17(ui->titleSafeCheckBox);
159 bool titleSafeOn = mManager->isOn(SETTING::TITLE_SAFE_ON);
160 ui->titleSafeCheckBox->setChecked(titleSafeOn);
161 QSignalBlocker b15(ui->titleSafeInput);
162 ui->titleSafeInput->setValue(mManager->getInt(SETTING::TITLE_SAFE));
163
164 QSignalBlocker b18(ui->safeHelperTextCheckbox);
165 ui->safeHelperTextCheckbox->setChecked(mManager->isOn(SETTING::OVERLAY_SAFE_HELPER_TEXT_ON));
166
167 QSignalBlocker b9(ui->highResBox);
168 ui->highResBox->setChecked(mManager->isOn(SETTING::HIGH_RESOLUTION));
169
170 QSignalBlocker b10(ui->backgroundButtons);
171 QString bgName = mManager->getString(SETTING::BACKGROUND_STYLE);
172
173 QSignalBlocker b12(ui->framePoolSizeSpin);
174 ui->framePoolSizeSpin->setValue(mManager->getInt(SETTING::FRAME_POOL_SIZE));
175
176 int buttonIdx = 1;
177 if (bgName == "checkerboard") buttonIdx = 1;
178 else if (bgName == "white") buttonIdx = 2;
179 else if (bgName == "grey") buttonIdx = 3;
180 else if (bgName == "dots") buttonIdx = 4;
181 else if (bgName == "weave") buttonIdx = 5;
182 else Q_ASSERT(false);
183
184 ui->backgroundButtons->button(buttonIdx)->setChecked(true);
185
186 ui->invertScrollDirectionBox->setChecked(mManager->isOn(SETTING::INVERT_SCROLL_ZOOM_DIRECTION));
187}

199void GeneralPage::backgroundChanged(QAbstractButton* button)
200{
201 QString brushName = "white";
202 if (button == ui->checkerBackgroundButton) brushName = "checkerboard";
203 else if (button == ui->whiteBackgroundButton) brushName = "white";
204 else if (button == ui->greyBackgroundButton) brushName = "grey";
205 else if (button == ui->dotsBackgroundButton) brushName = "dots";
206 else if (button == ui->weaveBackgroundButton) brushName = "weave";
207 else Q_UNREACHABLE();
208 mManager->set(SETTING::BACKGROUND_STYLE, brushName);
209}
