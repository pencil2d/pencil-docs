174QString PreferenceManager::getString(SETTING option)
175{
176 int optionId = static_cast<int>(option);
177 if (mIntegerSet.contains(optionId))
178 {
179 return QString::number(mIntegerSet.value(optionId, -1));
180 }
181 else if (mBooleanSet.contains(optionId))
182 {
183 if (mBooleanSet.value(optionId, false))
184 {
185 return "true";
186 }
187 else
188 {
189 return "false";
190 }
191 }
192 return mStringSet.value(optionId);
193}

195void PreferenceManager::set(SETTING option, QString value)
196{
197 QSettings settings(PENCIL2D, PENCIL2D);
198 switch (option)
199 {
200 case SETTING::BACKGROUND_STYLE:
201 settings.setValue(SETTING_BACKGROUND_STYLE, value);

203 case SETTING::ONION_TYPE:
204 settings.setValue(SETTING_ONION_TYPE, value);
205 break;

207 settings.setValue(SETTING_LANGUAGE, value);
208 break;
209 default:
210 break;
211 }
212
213 int optionId = static_cast<int>(option);
214 if (mStringSet[optionId] != value)
215 {
216 mStringSet[optionId] = value;
217 emit optionChanged(option);
