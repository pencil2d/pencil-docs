162 QString PreferenceManager::getString(SETTING option)
163 {
164  int optionId = static_cast<int>(option);
165  if (mIntegerSet.contains(optionId))
166  {
167  return QString::number(mIntegerSet.value(optionId, -1));
168  }
169  else if (mBooleanSet.contains(optionId))
170  {
171  if (mBooleanSet.value(optionId, false))
172  {
173  return "true";
174  }
175  else
176  {
177  return "false";
178  }
179  }
180  return mStringSet.value(optionId);
181 }
