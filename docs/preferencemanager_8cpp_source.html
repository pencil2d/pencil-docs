172QString PreferenceManager::getString(SETTING option)
173{
174 int optionId = static_cast<int>(option);
175 if (mIntegerSet.contains(optionId))
176 {
177 return QString::number(mIntegerSet.value(optionId, -1));
178 }
179 else if (mBooleanSet.contains(optionId))
180 {
181 if (mBooleanSet.value(optionId, false))
182 {
183 return "true";
184 }
185 else
186 {
187 return "false";
188 }
189 }
190 return mStringSet.value(optionId);
191}

193void PreferenceManager::set(SETTING option, QString value)
194{
195 QSettings settings(PENCIL2D, PENCIL2D);
196 switch (option)
197 {
198 case SETTING::BACKGROUND_STYLE:
199 settings.setValue(SETTING_BACKGROUND_STYLE, value);
200 break;
201 case SETTING::ONION_TYPE:
202 settings.setValue(SETTING_ONION_TYPE, value);
203 break;
204 case SETTING::LANGUAGE:
205 settings.setValue(SETTING_LANGUAGE, value);
206 break;
207 default:
208 break;
209 }
210
211 int optionId = static_cast<int>(option);
212 if (mStringSet[optionId] != value)
213 {
214 mStringSet[optionId] = value;
215 emit optionChanged(option);
216 }
