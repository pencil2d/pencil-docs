141 static inline bool compareColor(QRgb newColor, QRgb oldColor, int tolerance, QHash<QRgb, bool> *cache)
142 {
143 // Handle trivial case
144 if (newColor == oldColor) return true;
145
146 if(cache && cache->contains(newColor)) return cache->value(newColor);
147
148 // Get Eulcidian distance between colors
149 // Not an accurate representation of human perception,
150 // but it's the best any image editing program ever does
151 int diffRed = static_cast<int>(qPow(qRed(oldColor) - qRed(newColor), 2));
152 int diffGreen = static_cast<int>(qPow(qGreen(oldColor) - qGreen(newColor), 2));
153 int diffBlue = static_cast<int>(qPow(qBlue(oldColor) - qBlue(newColor), 2));
154 // This may not be the best way to handle alpha since the other channels become less relevant as
155 // the alpha is reduces (ex. QColor(0,0,0,0) is the same as QColor(255,255,255,0))
156 int diffAlpha = static_cast<int>(qPow(qAlpha(oldColor) - qAlpha(newColor), 2));
157
158 bool isSimilar = (diffRed + diffGreen + diffBlue + diffAlpha) <= tolerance;
159
160 if(cache)
161 {
162 Q_ASSERT(cache->contains(isSimilar) ? isSimilar == (*cache)[newColor] : true);
163 (*cache)[newColor] = isSimilar;
164 }
165
166 return isSimilar;
167 }
