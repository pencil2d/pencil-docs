113 void TimeControls::updateUI()
114 {
115  PlaybackManager* playback = mEditor->playback();
116 
117  mPlaybackRangeCheckBox->setChecked(playback->isRangedPlaybackOn()); // don't block this signal since it enables start/end range spinboxes.
118 
119  SignalBlocker b1(mLoopStartSpinBox);
120  mLoopStartSpinBox->setValue(playback->markInFrame());
121 
122  SignalBlocker b2(mLoopEndSpinBox);
123  mLoopEndSpinBox->setValue(playback->markOutFrame());
124 
125  SignalBlocker b3(mFpsBox);
126  mFpsBox->setValue(playback->fps());
127 
128  SignalBlocker b4(mLoopButton);
129  mLoopButton->setChecked(playback->isLooping());

155 void TimeControls::makeConnections()
156 {
157  connect(mPlayButton, &QPushButton::clicked, this, &TimeControls::playButtonClicked);
158  connect(mJumpToEndButton, &QPushButton::clicked, this, &TimeControls::jumpToEndButtonClicked);
159  connect(mJumpToStartButton, &QPushButton::clicked, this, &TimeControls::jumpToStartButtonClicked);
160  connect(mLoopButton, &QPushButton::clicked, this, &TimeControls::loopButtonClicked);
161  connect(mPlaybackRangeCheckBox, &QCheckBox::clicked, this, &TimeControls::playbackRangeClicked);
162 
163  auto spinBoxValueChanged = static_cast<void (QSpinBox::*)(int)>(&QSpinBox::valueChanged);
164  connect(mLoopStartSpinBox, spinBoxValueChanged, this, &TimeControls::loopStartValueChanged);
165  connect(mLoopEndSpinBox, spinBoxValueChanged, this, &TimeControls::loopEndValueChanged);
166 
167  connect(mPlaybackRangeCheckBox, &QCheckBox::toggled, mLoopStartSpinBox, &QSpinBox::setEnabled);
168  connect(mPlaybackRangeCheckBox, &QCheckBox::toggled, mLoopEndSpinBox, &QSpinBox::setEnabled);
169 
170  connect(mSoundButton, &QPushButton::clicked, this, &TimeControls::soundToggled);
171  connect(mSoundButton, &QPushButton::clicked, this, &TimeControls::updateSoundIcon);
172  auto connection = connect(mFpsBox, spinBoxValueChanged, this, &TimeControls::fpsChanged);
173  if(!connection)
174  {
175  // Use "editingFinished" if the "spinBoxValueChanged" signal doesn't work...
176  connect(mFpsBox, &QSpinBox::editingFinished, this, &TimeControls::onFpsEditingFinished);
177  }
178 }
