62 ui->featherSlider->setValue(settings.value("brushFeather", "5").toDouble());
63 ui->featherSpinBox->setValue(settings.value("brushFeather", "5").toDouble());

91void ToolOptionWidget::makeConnectionToEditor(Editor* editor)
92{
93 auto toolManager = editor->tools();
94
95 connect(ui->useBezierBox, &QCheckBox::clicked, toolManager, &ToolManager::setBezier);
96 connect(ui->usePressureBox, &QCheckBox::clicked, toolManager, &ToolManager::setPressure);
97 connect(ui->makeInvisibleBox, &QCheckBox::clicked, toolManager, &ToolManager::setInvisibility);
98 connect(ui->preserveAlphaBox, &QCheckBox::clicked, toolManager, &ToolManager::setPreserveAlpha);
99
100 connect(ui->sizeSlider, &SpinSlider::valueChanged, toolManager, &ToolManager::setWidth);
101 connect(ui->featherSlider, &SpinSlider::valueChanged, toolManager, &ToolManager::setFeather);
102
103 auto spinboxValueChanged = static_cast<void (QDoubleSpinBox::*)(double)>(&QDoubleSpinBox::valueChanged);
104 connect(ui->brushSpinBox, spinboxValueChanged, toolManager, &ToolManager::setWidth);
105 clearFocusOnFinished(ui->brushSpinBox);
106 connect(ui->featherSpinBox, spinboxValueChanged, toolManager, &ToolManager::setFeather);
107 clearFocusOnFinished(ui->featherSpinBox);
108
109 connect(ui->useFeatherBox, &QCheckBox::clicked, toolManager, &ToolManager::setUseFeather);
110
111 connect(ui->vectorMergeBox, &QCheckBox::clicked, toolManager, &ToolManager::setVectorMergeEnabled);
112 connect(ui->useAABox, &QCheckBox::clicked, toolManager, &ToolManager::setAA);
113
114 connect(ui->inpolLevelsCombo, static_cast<void (QComboBox::*)(int)>(&QComboBox::activated), toolManager, &ToolManager::setStabilizerLevel);
115
116 connect(ui->fillContourBox, &QCheckBox::clicked, toolManager, &ToolManager::setUseFillContour);
117
118 connect(ui->showInfoBox, &QCheckBox::clicked, toolManager, &ToolManager::setShowSelectionInfo);
119
120 connect(toolManager, &ToolManager::toolChanged, this, &ToolOptionWidget::onToolChanged);
121 connect(toolManager, &ToolManager::toolPropertyChanged, this, &ToolOptionWidget::onToolPropertyChanged);
122}

124void ToolOptionWidget::onToolPropertyChanged(ToolType, ToolPropertyType ePropertyType)
125{
126 const Properties& p = editor()->tools()->currentTool()->properties;
127
128 switch (ePropertyType)
129 {
130 case WIDTH: setPenWidth(p.width); break;
131 case FEATHER: setPenFeather(p.feather); break;
132 case USEFEATHER: setUseFeather(p.useFeather); break;
133 case PRESSURE: setPressure(p.pressure); break;
134 case INVISIBILITY: setPenInvisibility(p.invisibility); break;
135 case PRESERVEALPHA: setPreserveAlpha(p.preserveAlpha); break;
136 case VECTORMERGE: setVectorMergeEnabled(p.vectorMergeEnabled); break;
137 case ANTI_ALIASING: setAA(p.useAA); break;
138 case STABILIZATION: setStabilizerLevel(p.stabilizerLevel); break;
139 case FILLCONTOUR: setFillContour(p.useFillContour); break;
140 case SHOWSELECTIONINFO: setShowSelectionInfo(p.showSelectionInfo); break;
141 case BEZIER: setBezier(p.bezier_state); break;
142 case CAMERAPATH: { break; }
143 case TOLERANCE: break;
144 case USETOLERANCE: break;
145 case BUCKETFILLEXPAND: break;
146 case USEBUCKETFILLEXPAND: break;
147 case BUCKETFILLLAYERREFERENCEMODE: break;
148 case FILL_MODE: break;
149 default:
150 Q_ASSERT(false);
151 break;
152 }
153}

155void ToolOptionWidget::setVisibility(BaseTool* tool)
156{
157 Q_ASSERT(mBucketOptionsWidget);
158 Q_ASSERT(mCameraOptionsWidget);
159
160 disableAllOptions();
161
162 if (tool->type() == BUCKET)
163 {
164 mBucketOptionsWidget->setHidden(false);
165 return;
166 }
167 else if (tool->type() == CAMERA)
168 {
169 mCameraOptionsWidget->setHidden(false);
170 }
171 else
172 {
173 mCameraOptionsWidget->setHidden(true);
174 mBucketOptionsWidget->setHidden(true);
175 }
176
177 ui->sizeSlider->setVisible(tool->isPropertyEnabled(WIDTH));
178 ui->brushSpinBox->setVisible(tool->isPropertyEnabled(WIDTH));
179 ui->featherSlider->setVisible(tool->isPropertyEnabled(FEATHER));
180 ui->featherSpinBox->setVisible(tool->isPropertyEnabled(FEATHER));
181 ui->useFeatherBox->setVisible(tool->isPropertyEnabled(USEFEATHER));
182 ui->useBezierBox->setVisible(tool->isPropertyEnabled(BEZIER));
183 ui->usePressureBox->setVisible(tool->isPropertyEnabled(PRESSURE));
184 ui->makeInvisibleBox->setVisible(tool->isPropertyEnabled(INVISIBILITY));
185 ui->preserveAlphaBox->setVisible(tool->isPropertyEnabled(PRESERVEALPHA));
186 ui->useAABox->setVisible(tool->isPropertyEnabled(ANTI_ALIASING));
187 ui->stabilizerLabel->setVisible(tool->isPropertyEnabled(STABILIZATION));
188 ui->inpolLevelsCombo->setVisible(tool->isPropertyEnabled(STABILIZATION));
189 ui->fillContourBox->setVisible(tool->isPropertyEnabled(FILLCONTOUR));
190 ui->showInfoBox->setVisible(tool->isPropertyEnabled(SHOWSELECTIONINFO));
191
192 auto currentLayerType = editor()->layers()->currentLayer()->type();
193 auto propertyType = editor()->tools()->currentTool()->type();
194
195 if (currentLayerType == Layer::VECTOR)
196 {
197 switch (propertyType)
198 {
199 case SMUDGE:
200 ui->sizeSlider->setVisible(false);
201 ui->brushSpinBox->setVisible(false);
202 ui->usePressureBox->setVisible(false);
203 ui->featherSlider->setVisible(false);
204 ui->featherSpinBox->setVisible(false);
205 ui->useFeatherBox->setVisible(false);
206 break;
207 case PENCIL:
208 ui->sizeSlider->setVisible(false);
209 ui->brushSpinBox->setVisible(false);
210 ui->usePressureBox->setVisible(false);
211 break;
212 default:
213 ui->sizeSlider->setLabel(tr("Width"));
214 ui->useAABox->setVisible(false);
215 break;
216 }
217 }
218 else
219 {
220 switch (propertyType)
221 {
222 case PENCIL:
223 ui->fillContourBox->setVisible(false);
224 break;
225 case BUCKET:
226 ui->brushSpinBox->setVisible(false);
227 ui->sizeSlider->setVisible(false);
228 break;
229 case SELECT:
230 case MOVE:
231 ui->sizeSlider->setVisible(false);
232 ui->brushSpinBox->setVisible(false);
233 ui->usePressureBox->setVisible(false);
234 ui->featherSlider->setVisible(false);
235 ui->featherSpinBox->setVisible(false);
236 ui->useFeatherBox->setVisible(false);
237 break;
238 default:
239 ui->makeInvisibleBox->setVisible(false);
240 break;
241 }
242 }
243}

307void ToolOptionWidget::setAA(int x)
308{
309 QSignalBlocker b(ui->useAABox);
310 ui->useAABox->setEnabled(true);
311 ui->useAABox->setVisible(false);
312
313 auto layerType = editor()->layers()->currentLayer()->type();
314
315 if (layerType == Layer::BITMAP)
316 {
317 if (x == -1)
318 {
319 ui->useAABox->setEnabled(false);
320 ui->useAABox->setVisible(false);
321 }
322 else
323 {
324 ui->useAABox->setVisible(true);
325 }
326 ui->useAABox->setChecked(x > 0);
327 }
