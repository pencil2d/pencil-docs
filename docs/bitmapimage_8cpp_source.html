166 void BitmapImage::add(BitmapImage* bitmapImage)
167 {
168  QImage* image2 = bitmapImage->image();
169 
170  QRect newBoundaries;
171  if (image()->width() == 0 || image()->height() == 0)
172  {
173  newBoundaries = bitmapImage->mBounds;
174  }
175  else
176  {
177  newBoundaries = mBounds.united(bitmapImage->mBounds);
178  }
179  extend(newBoundaries);
180  QPoint offset = bitmapImage->mBounds.topLeft() - mBounds.topLeft();
181  for (int y = 0; y < image2->height(); y++)
182  {
183  for (int x = 0; x < image2->width(); x++)
184  {
185  QRgb p1 = image()->pixel(offset.x() + x, offset.y() + y);
186  QRgb p2 = image2->pixel(x, y);
187 
188  int a1 = qAlpha(p1);
189  int a2 = qAlpha(p2);
190  int r1 = qRed(p1);
191  int r2 = qRed(p2); // remember that the bitmap format is RGB32 Premultiplied
192  int g1 = qGreen(p1);
193  int g2 = qGreen(p2);
194  int b1 = qBlue(p1);
195  int b2 = qBlue(p2);
196 
197  // unite
198  int a = qMax(a1, a2);
199  int r = qMax(r1, r2);
200  int g = qMax(g1, g2);
201  int b = qMax(b1, b2);
202 
203  QRgb mix = qRgba(r, g, b, a);
204  if (a2 != 0)
205  {
206  mImage->setPixel(offset.x() + x, offset.y() + y, mix);
207  }
208  }
209  }
210  modification();
211 }

546 bool BitmapImage::compareColor(QRgb newColor, QRgb oldColor, int tolerance, QHash<QRgb, bool> *cache)
547 {
548  // Handle trivial case
549  if (newColor == oldColor) return true;
550 
551  if(cache && cache->contains(newColor)) return cache->value(newColor);
552 
553  // Get Eulcidian distance between colors
554  // Not an accurate representation of human perception,
555  // but it's the best any image editing program ever does
556  int diffRed = qPow(qRed(oldColor) - qRed(newColor), 2);
557  int diffGreen = qPow(qGreen(oldColor) - qGreen(newColor), 2);
558  int diffBlue = qPow(qBlue(oldColor) - qBlue(newColor), 2);
559  // This may not be the best way to handle alpha since the other channels become less relevant as
560  // the alpha is reduces (ex. QColor(0,0,0,0) is the same as QColor(255,255,255,0))
561  int diffAlpha = qPow(qAlpha(oldColor) - qAlpha(newColor), 2);
562 
563  bool isSimilar = (diffRed + diffGreen + diffBlue + diffAlpha) <= tolerance;
564 
565  if(cache)
566  {
567  Q_ASSERT(cache->contains(isSimilar) ? isSimilar == (*cache)[newColor] : true);
568  (*cache)[newColor] = isSimilar;
569  }
570 
571  return isSimilar;
572 }

574 // Flood fill
575 // ----- http://lodev.org/cgtutor/floodfill.html
576 void BitmapImage::floodFill(BitmapImage* targetImage,
577  QRect cameraRect,
578  QPoint point,
579  QRgb newColor,
580  int tolerance)
581 {
582  // If the point we are supposed to fill is outside the image and camera bounds, do nothing
583  if(!cameraRect.united(targetImage->bounds()).contains(point))
584  {
585  return;
586  }
587 
588  // Square tolerance for use with compareColor
589  tolerance = qPow(tolerance, 2);
590 
591  QRgb oldColor = targetImage->pixel(point);
592  oldColor = qRgba(qRed(oldColor), qGreen(oldColor), qBlue(oldColor), qAlpha(oldColor));
593 
594  // Preparations
595  QList<QPoint> queue; // queue all the pixels of the filled area (as they are found)
596 
597  BitmapImage* replaceImage = nullptr;
598  QPoint tempPoint;
599  QRgb newPlacedColor = 0;
600  QScopedPointer< QHash<QRgb, bool> > cache(new QHash<QRgb, bool>());
601 
602  int xTemp = 0;
603  bool spanLeft = false;
604  bool spanRight = false;
605 
606  // Extend to size of Camera
607  targetImage->extend(cameraRect);
608  replaceImage = new BitmapImage(cameraRect, Qt::transparent);
609 
610  queue.append(point);
611  // Preparations END
612 
613  while (!queue.empty())
614  {
615  tempPoint = queue.takeFirst();
616 
617  point.setX(tempPoint.x());
618  point.setY(tempPoint.y());
619 
620  xTemp = point.x();
621 
622  newPlacedColor = replaceImage->constScanLine(xTemp, point.y());
623  while (xTemp >= targetImage->topLeft().x() &&
624  compareColor(targetImage->constScanLine(xTemp, point.y()), oldColor, tolerance, cache.data())) xTemp--;
625  xTemp++;
626 
627  spanLeft = spanRight = false;
628  while (xTemp <= targetImage->right() &&
629  compareColor(targetImage->constScanLine(xTemp, point.y()), oldColor, tolerance, cache.data()) &&
630  newPlacedColor != newColor)
631  {
632 
633  // Set pixel color
634  replaceImage->scanLine(xTemp, point.y(), newColor);
635 
636  if (!spanLeft && (point.y() > targetImage->top()) &&
637  compareColor(targetImage->constScanLine(xTemp, point.y() - 1), oldColor, tolerance, cache.data())) {
638  queue.append(QPoint(xTemp, point.y() - 1));
639  spanLeft = true;
640  }
641  else if (spanLeft && (point.y() > targetImage->top()) &&
642  !compareColor(targetImage->constScanLine(xTemp, point.y() - 1), oldColor, tolerance, cache.data())) {
643  spanLeft = false;
644  }
645 
646  if (!spanRight && point.y() < targetImage->bottom() &&
647  compareColor(targetImage->constScanLine(xTemp, point.y() + 1), oldColor, tolerance, cache.data())) {
648  queue.append(QPoint(xTemp, point.y() + 1));
649  spanRight = true;
650 
651  }
652  else if (spanRight && point.y() < targetImage->bottom() &&
653  !compareColor(targetImage->constScanLine(xTemp, point.y() + 1), oldColor, tolerance, cache.data())) {
654  spanRight = false;
655  }
656 
657  Q_ASSERT(queue.count() < (targetImage->width() * targetImage->height()));
658  xTemp++;
659  }
660  }
661 
662  targetImage->paste(replaceImage);
663  targetImage->modification();
664  delete replaceImage;
665 }
