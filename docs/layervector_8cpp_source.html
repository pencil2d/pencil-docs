78 Status LayerVector::saveKeyFrameFile(KeyFrame* keyFrame, QString path)

80  QString theFileName = fileName(keyFrame);

83  VectorImage* vecImage = static_cast<VectorImage*>(keyFrame);

95  DebugDetails dd;
96  dd << __FUNCTION__;
97  dd << QString("KeyFrame.pos() = %1").arg(keyFrame->pos());

100  dd.collect(st.details());
101  return Status(Status::FAIL, dd);
102  }
103 
104  vecImage->setFileName(strFilePath);
105  vecImage->setModified(false);
106  return Status::OK;

152 void LayerVector::loadDomElement(const QDomElement& element, QString dataDirPath, ProgressCallback progressStep)
153 {
154  this->loadBaseDomElement(element);
155 
156  QDomNode imageTag = element.firstChild();
157  while (!imageTag.isNull())
158  {
159  QDomElement imageElement = imageTag.toElement();
160  if (!imageElement.isNull())
161  {
162  if (imageElement.tagName() == "image")
163  {
164  int position;
165  if (!imageElement.attribute("src").isNull())
166  {
167  QString path = dataDirPath + "/" + imageElement.attribute("src"); // the file is supposed to be in the data directory
168  QFileInfo fi(path);
169  if (!fi.exists()) path = imageElement.attribute("src");
170  position = imageElement.attribute("frame").toInt();
171  loadImageAtFrame(path, position);
172  }
173  else
174  {
175  position = imageElement.attribute("frame").toInt();
176  addNewKeyFrameAt(position);
177  getVectorImageAtFrame(position)->loadDomElement(imageElement);
178  }
179  if (imageElement.hasAttribute("opacity"))
180  getVectorImageAtFrame(position)->setOpacity(imageElement.attribute("opacity").toDouble());
181  else
182  getVectorImageAtFrame(position)->setOpacity(1.0);
183  progressStep();
184  }
185  }
186  imageTag = imageTag.nextSibling();
187  }
188 }
