99void MoveTool::pointerPressEvent(PointerEvent* event)
100{
101 Layer* currentLayer = currentPaintableLayer();
102 if (currentLayer == nullptr) return;
103
104 if (mEditor->select()->somethingSelected())
105 {
106 beginInteraction(event->canvasPos(), event->modifiers(), currentLayer);
107 }
108 else if (mEditor->overlays()->anyOverlayEnabled())
109 {
110 LayerCamera* layerCam = mEditor->layers()->getCameraLayerBelow(mEditor->currentLayerIndex());
111 Q_ASSERT(layerCam);
112
113 mPerspMode = mEditor->overlays()->getMoveModeForPoint(event->canvasPos(), layerCam->getViewAtFrame(mEditor->currentFrame()));
114 mEditor->overlays()->setMoveMode(mPerspMode);
115 QPoint mapped = layerCam->getViewAtFrame(mEditor->currentFrame()).map(event->canvasPos()).toPoint();
116 mEditor->overlays()->updatePerspective(mapped);
117 }
118
119 mEditor->updateFrame();
120}

122void MoveTool::pointerMoveEvent(PointerEvent* event)
123{
124 Layer* currentLayer = currentPaintableLayer();
125 if (currentLayer == nullptr) return;
126
127 if (mScribbleArea->isPointerInUse()) // the user is also pressing the mouse (dragging)
128 {
129 transformSelection(event->canvasPos(), event->modifiers());
130
131 if (mEditor->overlays()->anyOverlayEnabled())
132 {
133 LayerCamera* layerCam = mEditor->layers()->getCameraLayerBelow(mEditor->currentLayerIndex());
134 Q_ASSERT(layerCam);
135 mEditor->overlays()->updatePerspective(layerCam->getViewAtFrame(mEditor->currentFrame()).map(event->canvasPos()));
136 }
137 if (mEditor->select()->somethingSelected())
138 {
139 transformSelection(event->canvasPos(), event->modifiers());
140 }
141 }
142 else
143 {
144 // the user is moving the mouse without pressing it
145 // update cursor to reflect selection corner interaction
146 mEditor->select()->setMoveModeForAnchorInRange(event->canvasPos());
147 if (mEditor->overlays()->anyOverlayEnabled())
148 {
149 LayerCamera *layerCam = mEditor->layers()->getCameraLayerBelow(mEditor->currentLayerIndex());
150 Q_ASSERT(layerCam);
151 mPerspMode = mEditor->overlays()->getMoveModeForPoint(event->canvasPos(), layerCam->getViewAtFrame(mEditor->currentFrame()));
152 }
153 mScribbleArea->updateToolCursor();
154
155 if (currentLayer->type() == Layer::VECTOR)
156 {
157 storeClosestVectorCurve(event->canvasPos(), currentLayer);
158 }
159 }
160 mEditor->updateFrame();
161}

163void MoveTool::pointerReleaseEvent(PointerEvent*)
164{
165 mEditor->undoRedo()->record(mUndoSaveState, typeName());
166
167 if (mEditor->overlays()->anyOverlayEnabled())
168 {
169 mEditor->overlays()->setMoveMode(MoveMode::NONE);
170 mPerspMode = MoveMode::NONE;
171 }
172
173 auto selectMan = mEditor->select();
174 if (!selectMan->somethingSelected())
175 return;
176
177 mScribbleArea->updateToolCursor();
178 emit mEditor->frameModified(mEditor->currentFrame());
179}

181void MoveTool::transformSelection(const QPointF& pos, Qt::KeyboardModifiers keyMod)
182{
183 auto selectMan = mEditor->select();
184 if (selectMan->somethingSelected())
185 {
186 int rotationIncrement = 0;
187 if (selectMan->getMoveMode() == MoveMode::ROTATION && keyMod & Qt::ShiftModifier)
188 {
189 rotationIncrement = mRotationIncrement;
190 }
191
192 selectMan->maintainAspectRatio(keyMod == Qt::ShiftModifier);
193 selectMan->alignPositionToAxis(keyMod == Qt::ShiftModifier);
194
195 qreal newAngle = 0;
196 if (selectMan->getMoveMode() == MoveMode::ROTATION) {
197 QPointF anchorPoint = selectMan->currentTransformAnchor();
198 newAngle = selectMan->angleFromPoint(pos, anchorPoint) - mRotatedAngle;
199 }
200
201 selectMan->adjustSelection(pos, mOffset, newAngle, rotationIncrement);
202 }
203 else // there is nothing selected
204 {
205 selectMan->setMoveMode(MoveMode::NONE);
206 }
207}

209void MoveTool::beginInteraction(const QPointF& pos, Qt::KeyboardModifiers keyMod, Layer* layer)
210{
211 auto selectMan = mEditor->select();
212 QRectF selectionRect = selectMan->mySelectionRect();
213 if (!selectionRect.isNull())
214 {
215 mUndoSaveState = mEditor->undoRedo()->state(UndoRedoRecordType::KEYFRAME_MODIFY);
216 mEditor->backup(typeName());
217 }
218
219 if (keyMod != Qt::ShiftModifier)
220 {
221 if (selectMan->isOutsideSelectionArea(pos))
222 {
223 applyTransformation();
224 mEditor->deselectAll();
225 }
226 }
227
228 if (selectMan->getMoveMode() == MoveMode::MIDDLE)
229 {
230 if (keyMod == Qt::ControlModifier) // --- rotation
231 {
232 selectMan->setMoveMode(MoveMode::ROTATION);
233 }
234 }
235
236 if (layer->type() == Layer::VECTOR)
237 {
238 createVectorSelection(pos, keyMod, layer);
239 }
240
241 selectMan->setTransformAnchor(selectMan->getSelectionAnchorPoint());
242 selectMan->setDragOrigin(pos);
243 mOffset = selectMan->myTranslation();
244
245 if(selectMan->getMoveMode() == MoveMode::ROTATION) {
246 mRotatedAngle = selectMan->angleFromPoint(pos, selectMan->currentTransformAnchor()) - selectMan->myRotation();
247 }
248}

255void MoveTool::createVectorSelection(const QPointF& pos, Qt::KeyboardModifiers keyMod, Layer* layer)
256{
257 assert(layer->type() == Layer::VECTOR);
258 LayerVector* vecLayer = static_cast<LayerVector*>(layer);
259 VectorImage* vectorImage = vecLayer->getLastVectorImageAtFrame(mEditor->currentFrame(), 0);
260 if (vectorImage == nullptr) { return; }
261
262 if (!mEditor->select()->closestCurves().empty()) // the user clicks near a curve
263 {
264 setCurveSelected(vectorImage, keyMod);
265 }
266 else if (vectorImage->getLastAreaNumber(pos) > -1)
267 {
268 setAreaSelected(pos, vectorImage, keyMod);
269 }
270}

272void MoveTool::setCurveSelected(VectorImage* vectorImage, Qt::KeyboardModifiers keyMod)
273{
274 auto selectMan = mEditor->select();

278 {

282 selectMan->setSelection(vectorImage->getSelectionRect(), false);

304void MoveTool::storeClosestVectorCurve(const QPointF& pos, Layer* layer)
305{
306 auto selectMan = mEditor->select();

308 VectorImage* pVecImg = layerVector->getLastVectorImageAtFrame(mEditor->currentFrame(), 0);

313void MoveTool::applyTransformation()
314{
315 SelectionManager* selectMan = mEditor->select();
316 mScribbleArea->applyTransformedSelection();
317
318 // When the selection has been applied, a new rect is applied based on the bounding box.
319 // This ensures that if the selection has been rotated, it will still fit the bounds of the image.
320 selectMan->setSelection(selectMan->mapToSelection(QPolygonF(selectMan->mySelectionRect())).boundingRect());
321 mRotatedAngle = 0;
322}

364QCursor MoveTool::cursor(MoveMode mode) const
365{
366 QPixmap cursorPixmap = QPixmap(24, 24);
367
368 cursorPixmap.fill(QColor(255, 255, 255, 0));
369 QPainter cursorPainter(&cursorPixmap);
370 cursorPainter.setRenderHint(QPainter::Antialiasing);
371
372 switch(mode)
373 {
374 case MoveMode::PERSP_LEFT:
375 case MoveMode::PERSP_RIGHT:
376 case MoveMode::PERSP_MIDDLE:
377 case MoveMode::PERSP_SINGLE:
378 {
379 cursorPainter.drawImage(QPoint(6,6),QImage("://icons/general/cursor-move.svg"));
380 break;
381 }
382 case MoveMode::TOPLEFT:
383 case MoveMode::BOTTOMRIGHT:
384 {
385 cursorPainter.drawImage(QPoint(6,6),QImage("://icons/general/cursor-diagonal-left.svg"));
386 break;
387 }
388 case MoveMode::TOPRIGHT:
389 case MoveMode::BOTTOMLEFT:
390 {
391 cursorPainter.drawImage(QPoint(6,6),QImage("://icons/general/cursor-diagonal-right.svg"));
392 break;
393 }
394 case MoveMode::ROTATIONLEFT:
395 case MoveMode::ROTATIONRIGHT:
396 case MoveMode::ROTATION:
397 {
398 cursorPainter.drawImage(QPoint(6,6),QImage("://icons/general/cursor-rotate.svg"));
399 break;
400 }
401 case MoveMode::MIDDLE:
402 case MoveMode::CENTER:
403 {
404 cursorPainter.drawImage(QPoint(6,6),QImage("://icons/general/cursor-move.svg"));
405 break;
406 }
407 default:
408 return Qt::ArrowCursor;
409 }
410 cursorPainter.end();
411
412 return QCursor(cursorPixmap);
413}
