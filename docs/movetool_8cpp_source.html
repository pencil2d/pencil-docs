81 void MoveTool::mouseMoveEvent( QMouseEvent *event )
82 {
83  Layer* layer = mEditor->layers()->currentLayer();
84  if ( layer == NULL ) {
85  return;
86  }
87 
88  if ( layer->type() != Layer::BITMAP && layer->type() != Layer::VECTOR )
89  {
90  return;
91  }
92 
93  if ( event->buttons() & Qt::LeftButton ) // the user is also pressing the mouse (dragging)
94  {
95  if( mScribbleArea->somethingSelected )
96  {
97  QPointF offset = QPointF(mScribbleArea->mOffset.x(),
98  mScribbleArea->mOffset.y());
99 
100  if ( event->modifiers() == Qt::ShiftModifier ) // maintain aspect ratio
101  {
102  offset = maintainAspectRatio(offset.x(), offset.y());
103  }
104 
105  transformSelection(offset.x(), offset.y());
106 
107  mScribbleArea->calculateSelectionTransformation();
108  paintTransformedSelection();
109  }
110  else // there is nothing selected
111  {
112  mScribbleArea->deselectAll();
113  mScribbleArea->mMoveMode = ScribbleArea::NONE;
114  }
115  }
116  else // the user is moving the mouse without pressing it
117  {
118  if ( layer->type() == Layer::VECTOR )
119  {
120  storeClosestVectorCurve();
121  }
122  mScribbleArea->update();
123  }
124 }

126 void MoveTool::pressOperation(QMouseEvent* event, Layer* layer)
127 {
128  if ( layer->isPaintable() )
129  {
130  mEditor->backup( tr( "Move" ) );
131  mScribbleArea->setMoveMode( ScribbleArea::MIDDLE ); // was MIDDLE
132 
133  QRectF selectionRect = mScribbleArea->myTransformedSelection;
134  if ( mScribbleArea->somethingSelected ) // there is an area selection
135  {
136  if ( event->modifiers() != Qt::ShiftModifier && !selectionRect.contains( getCurrentPoint() ) )
137  mScribbleArea->deselectAll();
138 
139  whichTransformationPoint();
140 
141  // calculate new transformation in case click only
142  mScribbleArea->calculateSelectionTransformation();
143 
144  paintTransformedSelection();
145  applyChanges();
146 
147  mScribbleArea->setSelection( mScribbleArea->myTransformedSelection, true );
148  resetSelectionProperties();
149  }
150 
151  if ( mScribbleArea->getMoveMode() == ScribbleArea::MIDDLE )
152  {
153 
154  if ( event->modifiers() == Qt::ControlModifier ) // --- rotation
155  {
156  mScribbleArea->setMoveMode( ScribbleArea::ROTATION );
157  }
158  else if (event->modifiers() == Qt::AltModifier ) // --- symmetry
159  {
160  mScribbleArea->setMoveMode(ScribbleArea::SYMMETRY );
161  }
162 
163  if ( layer->type() == Layer::VECTOR )
164  {
165  actionOnVector(event, layer);
166  }
167  if ( !(mScribbleArea->myTransformedSelection.contains( getLastPoint() )) ) // click is outside the transformed selection with the MOVE tool
168  {
169  applyChanges();
170  }
171  }
172  }
173 }

175 void MoveTool::actionOnVector(QMouseEvent* event, Layer* layer)
176 {
177  VectorImage *vectorImage = ((LayerVector *)layer)->getLastVectorImageAtFrame( mEditor->currentFrame(), 0 );
178  if ( mScribbleArea->mClosestCurves.size() > 0 ) // the user clicks near a curve
179  {
180  // editor->backup();
181  if ( !vectorImage->isSelected( mScribbleArea->mClosestCurves ) )
182  {
183  if ( event->modifiers() != Qt::ShiftModifier )
184  {
185  applyChanges();
186  }
187  vectorImage->setSelected( mScribbleArea->mClosestCurves, true );
188  mScribbleArea->setSelection( vectorImage->getSelectionRect(), true );
189  mScribbleArea->update();
190  }
191  }
192  else
193  {
194  int areaNumber = vectorImage->getLastAreaNumber( getLastPoint() );
195  if ( areaNumber != -1 ) // the user clicks on an area
196  {
197  if ( !vectorImage->isAreaSelected( areaNumber ) )
198  {
199  if ( event->modifiers() != Qt::ShiftModifier )
200  {
201  applyChanges();
202  }
203  vectorImage->setAreaSelected( areaNumber, true );
204  mScribbleArea->setSelection( QRectF( 0, 0, 0, 0 ), true );
205  mScribbleArea->update();
206  }
207  }
208  }

233 void MoveTool::transformSelection(qreal offsetX, qreal offsetY)
234 {
235  switch ( mScribbleArea->mMoveMode )
236  {
237  case ScribbleArea::MIDDLE:
238  if ( QLineF( getLastPressPixel(), getCurrentPixel() ).length() > 4 )
239  {
240  mScribbleArea->myTempTransformedSelection =
241  mScribbleArea->myTransformedSelection.translated( QPointF(offsetX, offsetY) );
242  }
243  break;
244 
245  case ScribbleArea::TOPRIGHT:
246  mScribbleArea->myTempTransformedSelection =
247  mScribbleArea->myTransformedSelection.adjusted( 0, offsetY, offsetX, 0 );
248  break;
249 
250  case ScribbleArea::TOPLEFT:
251  mScribbleArea->myTempTransformedSelection =
252  mScribbleArea->myTransformedSelection.adjusted( offsetX, offsetY, 0, 0 );
253  break;
254 
255  case ScribbleArea::BOTTOMLEFT:
256  mScribbleArea->myTempTransformedSelection =
257  mScribbleArea->myTransformedSelection.adjusted( offsetX, 0, 0, offsetY );
258  break;
259 
260  case ScribbleArea::BOTTOMRIGHT:
261  mScribbleArea->myTempTransformedSelection =
262  mScribbleArea->myTransformedSelection.adjusted( 0, 0, offsetX, offsetY );
263  break;
264 
265  case ScribbleArea::ROTATION:
266  mScribbleArea->myTempTransformedSelection =
267  mScribbleArea->myTransformedSelection; // @ necessary?
268  mScribbleArea->myRotatedAngle = getCurrentPixel().x() - getLastPressPixel().x();
269  break;
270  default:
271  break;
272  }
273  mScribbleArea->update();
274

277 QPointF MoveTool::maintainAspectRatio(qreal offsetX, qreal offsetY)
278 {
279  qreal factor = mScribbleArea->mySelection.width() / mScribbleArea->mySelection.height();
280 
281  if (mScribbleArea->mMoveMode == ScribbleArea::TOPLEFT || mScribbleArea->mMoveMode == ScribbleArea::BOTTOMRIGHT) {
282  offsetY = offsetX / factor;
283  }
284  else if (mScribbleArea->mMoveMode == ScribbleArea::TOPRIGHT || mScribbleArea->mMoveMode == ScribbleArea::BOTTOMLEFT) {
285  offsetY = -(offsetX / factor);
286  }
287  else if (mScribbleArea->mMoveMode == ScribbleArea::MIDDLE) {
288 
289  qreal absX = offsetX;
290  if (absX < 0) {absX = -absX;}
291 
292  qreal absY = offsetY;
293  if (absY < 0) {absY = -absY;}
294 
295  if (absX > absY) {
296  offsetY = 0;
297  }
298  if (absY > absX) {
299  offsetX = 0;
300  }
301  }
302  return QPointF(offsetX,offsetY);
303 }
